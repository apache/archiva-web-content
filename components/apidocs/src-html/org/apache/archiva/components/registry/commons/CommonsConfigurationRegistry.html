053/**
054 * Implementation of the registry component using
055 * <a href="http://commons.apache.org/commons/configuration">Commons Configuration</a>. The use of Commons Configuration
056 * enables a variety of sources to be used, including XML files, properties, JNDI, JDBC, etc.
057 * <p/>
058 * The component can be configured using the {@link #properties} configuration item, the content of which should take
059 * the format of an input to the Commons Configuration
060 * <a href="http://commons.apache.org/commons/configuration/howto_configurationbuilder.html">configuration
061 * builder</a>.
062 */
063@Service( "commons-configuration" )
064public class CommonsConfigurationRegistry
065    implements Registry
066{

068     * The combined configuration instance that houses the registry.
069     */

102    public String dump( )
103    {
104        StringBuilder buffer = new StringBuilder( );
105        buffer.append( "Configuration Dump." );
106        for ( Iterator i = configuration.getKeys( ); i.hasNext( ); )
107        {
108            String key = (String) i.next( );
109            Object value = configuration.getProperty( key );
110            buffer.append( "\n\"" ).append( key ).append( "\" = \"" ).append( value ).append( "\"" );
111        }
112        return buffer.toString( );
113    }

130    public List<Registry> getSubsetList( String key )
131    {
132        List<Registry> subsets = new ArrayList<>( );
133
134        boolean done = false;
135        do
136        {
137            Registry registry = getSubset( key + "(" + subsets.size( ) + ")" );
138            if ( !registry.isEmpty( ) )
139            {
140                subsets.add( registry );
141            }
142            else
143            {
144                done = true;
145            }
146        }
147        while ( !done );
148
149        return subsets;
150    }

152    public Properties getProperties( String key )
153    {
154        Configuration configuration = this.configuration.subset( key );
155
156        Properties properties = new Properties( );
157        for ( Iterator i = configuration.getKeys( ); i.hasNext( ); )
158        {
159            String property = (String) i.next( );
160            List l = configuration.getList( property );
161            StringBuilder sb = new StringBuilder( );
162            for ( Object element : l )
163            {
164                sb.append( (String) element );
165                sb.append( "," );
166            }
167            if ( sb.length( ) > 0 )
168            {
169                sb.deleteCharAt( sb.length( ) - 1 );
170            }
171            properties.setProperty( property, sb.toString( ) );
172        }
173        return properties;
174    }

176    public void save( )
177        throws RegistryException
178    {
179        if ( configuration instanceof FileConfiguration )
180        {
181            FileConfiguration fileConfiguration = (FileConfiguration) configuration;
182            try
183            {
184                fileConfiguration.save( );
185            }
186            catch ( ConfigurationException e )
187            {
188                throw new RegistryException( e.getMessage( ), e );
189            }
190        }
191        else
192        {
193            throw new RegistryException( "Can only save file-based configurations" );
194        }
195    }

221    public Collection<String> getKeys( )
222    {
223        Set<String> keys = new HashSet<String>( );
224
225        for ( Iterator<String> i = configuration.getKeys( ); i.hasNext( ); )
226        {
227            String key = i.next( );
228
229            int index = key.indexOf( '.' );
230            if ( index < 0 )
231            {
232                keys.add( key );
233            }
234            else
235            {
236                keys.add( key.substring( 0, index ) );
237            }
238        }
239
240        return keys;
241    }

248        {
249            keys.add( i.next( ) );
250        }
251
252        return keys;
253    }

260    public void removeSubset( String key )
261    {
262        // create temporary list since removing a key will modify the iterator from configuration
263        List keys = new ArrayList( );
264        for ( Iterator i = configuration.getKeys( key ); i.hasNext( ); )
265        {
266            keys.add( i.next( ) );
267        }
268
269        for ( Iterator i = keys.iterator( ); i.hasNext( ); )
270        {
271            configuration.clearProperty( (String) i.next( ) );
272        }
273    }

326    public void addConfigurationFromResource( String resource, String prefix )
327        throws RegistryException
328    {
329        if ( !( this.configuration instanceof CombinedConfiguration ) )
330        {
331            throw new RegistryException( "This is not a combined configuration so cannot add resource" );
332        }
333        CombinedConfiguration configuration = (CombinedConfiguration) this.configuration;
334        if ( resource.endsWith( ".properties" ) )
335        {
336            try
337            {
338                logger.debug( "Loading properties configuration from classloader resource: {}", resource );
339                configuration.addConfiguration( new PropertiesConfiguration( resource ), null, prefix );
340            }
341            catch ( ConfigurationException e )
342            {
343                throw new RegistryException(
344                    "Unable to add configuration from resource '" + resource + "': " + e.getMessage( ), e );
345            }
346        }
347        else if ( resource.endsWith( ".xml" ) )
348        {
349            try
350            {
351                logger.debug( "Loading XML configuration from classloader resource: {}", resource );
352                configuration.addConfiguration( new XMLConfiguration( resource ), null, prefix );
353            }
354            catch ( ConfigurationException e )
355            {
356                throw new RegistryException(
357                    "Unable to add configuration from resource '" + resource + "': " + e.getMessage( ), e );
358            }
359        }
360        else
361        {
362            throw new RegistryException(
363                "Unable to add configuration from resource '" + resource + "': unrecognised type" );
364        }
365    }

410    @PostConstruct
411    public void initialize( )
412        throws RegistryException
413    {
414        try
415        {
416            CombinedConfiguration configuration;
417            if ( StringUtils.isNotBlank( properties ) )
418            {
419                DefaultConfigurationBuilder builder = new DefaultConfigurationBuilder( );
420                DefaultExpressionEngine expressionEngine = new DefaultExpressionEngine( );
421                expressionEngine.setPropertyDelimiter( propertyDelimiter );
422                builder.setExpressionEngine( expressionEngine );
423
424                StringSubstitutor substitutor = new StringSubstitutor( StringLookupFactory.INSTANCE.systemPropertyStringLookup( ) );
425
426                String interpolatedProps = substitutor.replace( properties );
427
428                logger.debug( "Loading configuration into commons-configuration, xml {}", interpolatedProps );
429                builder.load( new StringReader( interpolatedProps ) );
430                configuration = builder.getConfiguration( false );
431                configuration.setExpressionEngine( expressionEngine );
432                //configuration.set
433            }
434            else
435            {
436                logger.debug( "Creating a default configuration - no configuration was provided" );
437                configuration = new CombinedConfiguration( );
438            }
439
440            configuration.addConfiguration( new SystemConfiguration( ) );
441
442            this.configuration = configuration;
443        }
444        catch ( ConfigurationException e )
445        {
446            throw new RuntimeException( e.getMessage( ), e );
447        }
448    }
