021/**
022 * The edge links a source node to a destination node.
023 * A edge instance is always part of a certain graph instance.
024 *
025 * @param <V> The node implementation.
026 */
027public interface Edge<V extends Node<V>>
028{
029
030    /**
031     * Returns the identifier of this edge. The id must be unique for given graph.
032     *
033     * @return the identifier of this edge
034     */
035    String getId( );
036
037    /**
038     * Returns the label of this edge. The label must not be unique.
039     * If the label was not set, it should return an empty string.
040     *
041     * @return the label of this edge, or a empty string
042     */
043    String getLabel( );
044
045    /**
046     * Sets the label of this edge to the given string.
047     *
048     * @param label the label string, that must not be <code>null</code>
049     * @throws NullPointerException if the label parameter is <code>null</code>
050     */
051    void setLabel( String label ) throws NullPointerException;
052
053    /**
054     * Returns the graph where this edge was created.
055     *
056     * @return the graph instance
057     */
058    Graph<V> getGraph( );
059
060    /**
061     * Returns the node, that is on the source end of this edge.
062     *
063     * @return the source node
064     */
065    V getSource( );
066
067    /**
068     * Returns the node, that is on the destination end of this edge.
069     *
070     * @return the destination node
071     */
072    V getDestination( );
073
074    /**
075     * Returns the weight of this edge. For standard graph implementations the default should be 1.0, but
076     * graph implementations may decide to set another default.
077     *
078     * @return the weight of this edge
079     */
080    double getWeight( );
081
082    /**
083     * Returns the RelationType of the edge. If nothing is set, the {@link StandardRelationType#DEFAULT} should
084     * be returned.
085     *
086     * @return the type of the edge
087     */
088    RelationType getType( );
089
090
091}
