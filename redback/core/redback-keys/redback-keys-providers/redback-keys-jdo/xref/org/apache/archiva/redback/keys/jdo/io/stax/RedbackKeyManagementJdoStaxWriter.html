98      private void writeAuthenticationKeyDatabase(AuthenticationKeyDatabase authenticationKeyDatabase, String tagName, XMLStreamWriter serializer)
99          throws java.io.IOException, XMLStreamException
100     {
101         if ( authenticationKeyDatabase != null )
102         {
103             serializer.writeStartElement( tagName );
104             if ( authenticationKeyDatabase.getKeys() != null && authenticationKeyDatabase.getKeys().size() > 0 )
105             {
106                 serializer.writeStartElement( "keys" );
107                 for ( Iterator iter = authenticationKeyDatabase.getKeys().iterator(); iter.hasNext(); )
108                 {
109                     JdoAuthenticationKey o = (JdoAuthenticationKey) iter.next();
110                     writeJdoAuthenticationKey( o, "key", serializer );
111                 }
112                 serializer.writeEndElement();
113             }
114             serializer.writeEndElement();
115         }

124     public void writeDom(Xpp3Dom dom, XMLStreamWriter serializer)
125         throws XMLStreamException
126     {
127         serializer.writeStartElement( dom.getName() );
128         String[] attributeNames = dom.getAttributeNames();
129         for ( int i = 0; i < attributeNames.length; i++ )
130         {
131             String attributeName = attributeNames[i];
132             serializer.writeAttribute( attributeName, dom.getAttribute( attributeName ) );
133         }
134         Xpp3Dom[] children = dom.getChildren();
135         for ( int i = 0; i < children.length; i++ )
136         {
137             writeDom( children[i], serializer );
138         }
139         String value = dom.getValue();
140         if ( value != null )
141         {
142             serializer.writeCharacters( value );
143         }
144         serializer.writeEndElement();
