41      /**
42       * Method buildDom
43       * 
44       * @param xmlStreamReader
45       */
46      public Xpp3Dom buildDom(XMLStreamReader xmlStreamReader)
47          throws XMLStreamException
48      {
49          java.util.List elements = new java.util.ArrayList();
50          java.util.List values = new java.util.ArrayList();
51          int eventType = xmlStreamReader.getEventType();
52          while ( xmlStreamReader.hasNext() )
53          {
54              if ( eventType == XMLStreamConstants.START_ELEMENT )
55              {
56                  String rawName = xmlStreamReader.getLocalName();
57                  Xpp3Dom childConfiguration = new Xpp3Dom( rawName );
58                  int depth = elements.size();
59                  if ( depth > 0 )
60                  {
61                      Xpp3Dom parent = (Xpp3Dom) elements.get( depth - 1 );
62                      parent.addChild( childConfiguration );
63                  }
64                  elements.add( childConfiguration );
65                  if ( xmlStreamReader.isEndElement() )
66                  {
67                      values.add( null );
68                  }
69                  else
70                  {
71                      values.add( new StringBuffer() );
72                  }
73                  int attributesSize = xmlStreamReader.getAttributeCount();
74                  for ( int i = 0; i < attributesSize; i++ )
75                  {
76                      String name = xmlStreamReader.getAttributeLocalName( i );
77                      String value = xmlStreamReader.getAttributeValue( i );
78                      childConfiguration.setAttribute( name, value );
79                  }
80              }
81              else if ( eventType == XMLStreamConstants.CHARACTERS )
82              {
83                  int depth = values.size() - 1;
84                  StringBuffer valueBuffer = (StringBuffer) values.get( depth );
85                  String text = xmlStreamReader.getText();
86                  text = text.trim();
87                  valueBuffer.append( text );
88              }
89              else if ( eventType == XMLStreamConstants.END_ELEMENT )
90              {
91                  int depth = elements.size() - 1;
92                  Xpp3Dom finishedConfiguration = (Xpp3Dom) elements.remove( depth );
93                  // this Object could be null if it is a singleton tag
94                  Object accumulatedValue = values.remove( depth );
95                  if ( finishedConfiguration.getChildCount() == 0 )
96                  {
97                      if ( accumulatedValue == null )
98                      {
99                          finishedConfiguration.setValue( null );
100                     }
101                     else
102                     {
103                         finishedConfiguration.setValue( accumulatedValue.toString() );
104                     }
105                 }
106                 if ( depth == 0 )
107                 {
108                     return finishedConfiguration;
109                 }
110             }
111             eventType = xmlStreamReader.next();
112         }
113         throw new IllegalStateException( "End of document found before returning to 0 depth" );
114     } //-- Xpp3Dom buildDom(XMLStreamReader)
