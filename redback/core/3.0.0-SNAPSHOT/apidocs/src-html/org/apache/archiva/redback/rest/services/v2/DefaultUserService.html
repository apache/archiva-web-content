199    @Override
200    public UserInfo createUser( User user )
201        throws RedbackServiceException
202    {
203        if ( user == null )
204        {
205            throw new RedbackServiceException( ErrorMessage.of( MessageKeys.ERR_USER_ID_EMPTY ), 422 );
206        }
207        UserInfo result;
208        if ( Arrays.binarySearch( INVALID_CREATE_USER_NAMES, user.getUserId( ) ) >= 0 )
209        {
210            throw new RedbackServiceException( ErrorMessage.of( MessageKeys.ERR_USER_ID_INVALID, user.getUserId( ) ), 422 );
211        }
212
213        try
214        {
215            org.apache.archiva.redback.users.User u = userManager.findUser( user.getUserId( ) );
216            if ( u != null )
217            {
218                httpServletResponse.setHeader( "Location", uriInfo.getAbsolutePathBuilder( ).path( u.getUsername( ) ).build( ).toString( ) );
219                throw new RedbackServiceException(
220                    ErrorMessage.of( MessageKeys.ERR_USER_EXISTS, user.getUserId( ) ), 303 );
221            }
222        }
223        catch ( UserNotFoundException e )
224        {
225            //ignore we just want to prevent non human readable error message from backend :-)
226            log.debug( "User {} does not exist", user.getUserId( ) );
227        }
228        catch ( UserManagerException e )
229        {
230            throw new RedbackServiceException( ErrorMessage.of( MessageKeys.ERR_UNKNOWN, e.getMessage( ) ) );
231        }
232
233        // data validation
234        if ( StringUtils.isEmpty( user.getUserId( ) ) )
235        {
236            throw new RedbackServiceException( ErrorMessage.of( MessageKeys.ERR_USER_ID_EMPTY ), 422 );
237        }
238
239        if ( StringUtils.isEmpty( user.getFullName( ) ) )
240        {
241            throw new RedbackServiceException( ErrorMessage.of( MessageKeys.ERR_USER_FULL_NAME_EMPTY ), 422 );
242        }
243
244        if ( StringUtils.isEmpty( user.getEmail( ) ) )
245        {
246            throw new RedbackServiceException( ErrorMessage.of( MessageKeys.ERR_USER_EMAIL_EMPTY ), 422 );
247        }
248
249        try
250        {
251
252            org.apache.archiva.redback.users.User u =
253                userManager.createUser( user.getUserId( ), user.getFullName( ), user.getEmail( ) );
254            u.setPassword( user.getPassword( ) );
255            u.setLocked( user.isLocked( ) );
256            u.setPasswordChangeRequired( user.isPasswordChangeRequired( ) );
257            u.setValidated( user.isValidated( ) );
258            u = userManager.addUser( u );
259            if ( !user.isPasswordChangeRequired( ) )
260            {
261                u.setPasswordChangeRequired( false );
262                try
263                {
264                    u = userManager.updateUser( u );
265                    log.debug( "User {} created", u.getUsername( ) );
266                }
267                catch ( UserNotFoundException e )
268                {
269                    throw new RedbackServiceException( e.getMessage( ) );
270                }
271            }
272
273            roleManager.assignRole( RedbackRoleConstants.REGISTERED_USER_ROLE_ID, u.getUsername( ) );
274            result = getRestUser( u );
275            httpServletResponse.setStatus( 201 );
276            httpServletResponse.setHeader( "Location", uriInfo.getAbsolutePathBuilder( ).path( user.getUserId( ) ).build( ).toString( ) );
277        }
278        catch ( RoleManagerException rpe )
279        {
280            log.error( "RoleProfile Error: {}", rpe.getMessage( ), rpe );
281            throw new RedbackServiceException( ErrorMessage.of( MessageKeys.ERR_USER_ASSIGN_ROLE ) );
282        }
283        catch ( UserManagerException e )
284        {
285            log.error( "UserManagerException: {}", e.getMessage( ), e );
286            throw new RedbackServiceException( ErrorMessage.of( MessageKeys.ERR_UNKNOWN, e.getMessage( ) ) );
287        }
288        return result;
289    }

291    @Override
292    public void deleteUser( String userId )
293        throws RedbackServiceException
294    {
295        if ( StringUtils.isEmpty( userId ) )
296        {
297            throw new RedbackServiceException( MessageKeys.ERR_USER_ID_EMPTY, 404 );
298        }
299
300        try
301        {
302
303            if ( rbacManager.userAssignmentExists( userId ) )
304            {
305                UserAssignment assignment = rbacManager.getUserAssignment( userId );
306                rbacManager.removeUserAssignment( assignment );
307            }
308
309        }
310        catch ( RbacManagerException e )
311        {
312            log.error( e.getMessage( ), e );
313            throw new RedbackServiceException( ErrorMessage.of( MessageKeys.ERR_RBACMANAGER_FAIL, e.getMessage( ) ) );
314        }
315        try
316        {
317            userManager.deleteUser( userId );
318        }
319        catch ( UserNotFoundException e )
320        {
321            log.error( e.getMessage( ), e );
322            throw new RedbackServiceException( ErrorMessage.of( MessageKeys.ERR_USER_NOT_FOUND ), 404 );
323        }
324        catch ( UserManagerException e )
325        {
326            throw new RedbackServiceException( ErrorMessage.of( MessageKeys.ERR_USERMANAGER_FAIL, e.getMessage( ) ) );
327        }
328        finally
329        {
330            removeFromCache( userId );
331        }
332        httpServletResponse.setStatus( 200 );
333    }

338        throws RedbackServiceException
339    {
340        if ( StringUtils.isEmpty( userId ) )
341        {
342            throw new RedbackServiceException( ErrorMessage.of( MessageKeys.ERR_USER_ID_EMPTY ), 404 );
343        }
344        try
345        {
346            if ( "guest".equals( userId ) )
347            {
348                return getRestUser( userManager.getGuestUser( ) );
349            }
350            org.apache.archiva.redback.users.User user = userManager.findUser( userId );
351            return getRestUser( user );
352        }
353        catch ( UserNotFoundException e )
354        {
355            throw new RedbackServiceException( ErrorMessage.of( MessageKeys.ERR_USER_NOT_FOUND ), 404 );
356        }
357        catch ( UserManagerException e )
358        {
359            throw new RedbackServiceException( new ErrorMessage( e.getMessage( ) ) );
360        }

383    @Override
384    public UserInfo updateMe( SelfUserData user )
385        throws RedbackServiceException
386    {
387        RedbackPrincipal principal = getPrincipal( );
388        if ( principal == null )
389        {
390            throw new RedbackServiceException( ErrorMessage.of( MessageKeys.ERR_AUTH_UNAUTHORIZED_REQUEST ), 401 );
391        }
392
393        // check oldPassword with the current one
394        // only 3 fields to update
395        // ui can limit to not update password
396        org.apache.archiva.redback.users.User foundUser = updateUser( principal.getName( ), realUser -> {
397            try
398            {
399                // current password is only needed, if password change is requested
400                if ( StringUtils.isNotBlank( user.getPassword( ) ) )
401                {
402                    String previousEncodedPassword =
403                        securitySystem.getUserManager( ).findUser( principal.getName( ), false ).getEncodedPassword( );
404
405                    // check oldPassword with the current one
406
407                    PasswordEncoder encoder = securitySystem.getPolicy( ).getPasswordEncoder( );
408
409                    if ( !encoder.isPasswordValid( previousEncodedPassword, user.getCurrentPassword( ) ) )
410                    {
411
412                        return new RedbackServiceException( ErrorMessage.of( MessageKeys.ERR_USER_BAD_PASSWORD ),
413                            Response.Status.BAD_REQUEST.getStatusCode( ) );
414                    }
415                }
416            }
417            catch ( UserNotFoundException e )
418            {
419                return new RedbackServiceException( ErrorMessage.of( MessageKeys.ERR_USER_NOT_FOUND ),
420                    Response.Status.BAD_REQUEST.getStatusCode( ) );
421            }
422            catch ( UserManagerException e )
423            {
424                return new RedbackServiceException( ErrorMessage.of( MessageKeys.ERR_USERMANAGER_FAIL, e.getMessage( ) ) );
425            }
426            // only 3 fields to update
427            if ( StringUtils.isNotBlank( user.getFullName( ) ) )
428            {
429                realUser.setFullName( user.getFullName( ) );
430            }
431            if ( StringUtils.isNotBlank( user.getEmail( ) ) )
432            {
433                realUser.setEmail( user.getEmail( ) );
434            }
435            // ui can limit to not update password
436            if ( StringUtils.isNotBlank( user.getPassword( ) ) )
437            {
438                realUser.setPassword( user.getPassword( ) );
439            }
440            return null;
441        } );
442
443        return getRestUser( foundUser );
444    }

467    @Override
468    public UserInfo updateUser( String userId, User user )
469        throws RedbackServiceException
470    {
471        try
472        {
473            org.apache.archiva.redback.users.User rawUser = userManager.findUser( userId, false );
474            if ( user.getFullName( ) != null )
475                rawUser.setFullName( user.getFullName( ) );
476            if ( user.getEmail( ) != null )
477                rawUser.setEmail( user.getEmail( ) );
478            rawUser.setValidated( user.isValidated( ) );
479            rawUser.setLocked( user.isLocked( ) );
480            if ( !StringUtils.isEmpty( user.getPassword( ) ) )
481                rawUser.setPassword( user.getPassword( ) );
482            rawUser.setPasswordChangeRequired( user.isPasswordChangeRequired( ) );
483
484            org.apache.archiva.redback.users.User updatedUser = userManager.updateUser( rawUser );
485
486            return getRestUser( updatedUser );
487        }
488        catch ( UserNotFoundException e )
489        {
490            throw new RedbackServiceException( ErrorMessage.of( MessageKeys.ERR_USER_NOT_FOUND ), 404 );
491        }
492        catch ( PasswordRuleViolationException e )
493        {
494            List<ErrorMessage> messages = e.getViolations( ).getViolations( ).stream( ).map( m -> ErrorMessage.of( m.getKey( ), m.getArgs( ) ) ).collect( Collectors.toList( ) );
495            throw new RedbackServiceException( messages, 422 );
496        }
497        catch ( UserManagerException e )
498        {
499            throw new RedbackServiceException( ErrorMessage.of( MessageKeys.ERR_USERMANAGER_FAIL, e.getMessage( ) ), 400 );
500        }
501    }

531    public UserInfo createAdminUser( User adminUser )
532        throws RedbackServiceException
533    {
534        UserInfo result;
535        if ( getAdminStatus( ).isExists( ) )
536        {
537            log.warn( "Admin user exists already" );
538            httpServletResponse.setHeader( "Location", uriInfo.getAbsolutePath( ).toString( ) );
539            throw new RedbackServiceException( ErrorMessage.of( MessageKeys.ERR_USER_ADMIN_EXISTS ), 303 );
540        }
541        log.debug( "Creating admin admin user '{}'", adminUser.getUserId( ) );
542        if ( !RedbackRoleConstants.ADMINISTRATOR_ACCOUNT_NAME.equals( adminUser.getUserId( ) ) )
543        {
544            log.error( "Wrong admin user name {}", adminUser.getUserId( ) );
545            throw new RedbackServiceException( ErrorMessage.of( MessageKeys.ERR_USER_ADMIN_BAD_NAME ), 422 );
546        }
547
548        try
549        {
550            org.apache.archiva.redback.users.User user =
551                userManager.createUser( RedbackRoleConstants.ADMINISTRATOR_ACCOUNT_NAME, adminUser.getFullName( ),
552                    adminUser.getEmail( ) );
553            user.setPassword( adminUser.getPassword( ) );
554
555            user.setLocked( false );
556            user.setPasswordChangeRequired( false );
557            user.setPermanent( true );
558            user.setValidated( true );
559
560            userManager.addUser( user );
561            result = getRestUser( user );
562            roleManager.assignRole( "system-administrator", user.getUsername( ) );
563        }
564        catch ( RoleManagerException e )
565        {
566            throw new RedbackServiceException( ErrorMessage.of( MessageKeys.ERR_ROLEMANAGER_FAIL, e.getMessage( ) ), 400 );
567        }
568        catch ( UserManagerException e )
569        {
570            throw new RedbackServiceException( ErrorMessage.of( MessageKeys.ERR_USERMANAGER_FAIL, e.getMessage( ) ), 400 );
571        }
572        httpServletResponse.setStatus( 201 );
573        httpServletResponse.setHeader( "Location", uriInfo.getAbsolutePath( ).toString( ) );
574        return result;
575    }

579        throws RedbackServiceException
580    {
581        try
582        {
583            org.apache.archiva.redback.users.User user = userManager.findUser( config.getString( UserConfigurationKeys.DEFAULT_ADMIN ) );

593        catch ( UserNotFoundException e )
594        {
595            // ignore
596        }
597        catch ( UserManagerException e )

600

603                return new AvailabilityStatus( false );
604            }
605            throw new RedbackServiceException( ErrorMessage.of( MessageKeys.ERR_USERMANAGER_FAIL, e.getMessage( ) ), 400 );
606        }
607        return new AvailabilityStatus( false );
608    }

610    @Override
611    public Response resetPassword( String userId )
612        throws RedbackServiceException
613    {
614        String username = userId;
615        if ( StringUtils.isEmpty( username ) )
616        {
617            throw new RedbackServiceException( new ErrorMessage( "username.cannot.be.empty" ) );
618        }
619
620        UserManager userManager = securitySystem.getUserManager( );
621        KeyManager keyManager = securitySystem.getKeyManager( );
622        UserSecurityPolicy policy = securitySystem.getPolicy( );
623
624        try
625        {
626            org.apache.archiva.redback.users.User user = userManager.findUser( username );
627
628            AuthenticationKey authkey = keyManager.createKey( username, "Password Reset Request",
629                policy.getUserValidationSettings( ).getEmailValidationTimeout( ) );
630
631            String applicationUrl = getBaseUrl( );
632
633            mailer.sendPasswordResetEmail( Arrays.asList( user.getEmail( ) ), authkey, applicationUrl );
634            log.info( "password reset request for username {}", username );
635        }
636        catch ( UserNotFoundException e )
637        {
638            log.info( "Password Reset on non-existant user [{}].", username );
639            throw new RedbackServiceException( ErrorMessage.of( MessageKeys.ERR_USER_NOT_FOUND, userId ), 404 );
640        }
641        catch ( KeyManagerException e )
642        {
643            log.info( "Unable to issue password reset.", e );
644            throw new RedbackServiceException( ErrorMessage.of( MessageKeys.ERR_PASSWD_RESET_FAILED, e.getMessage( ) ), 400 );
645        }
646        catch ( UserManagerException e )
647        {
648            throw new RedbackServiceException( ErrorMessage.of( MessageKeys.ERR_USERMANAGER_FAIL, e.getMessage( ) ), 400 );
649        }
650
651        return Response.ok( ).build( );
652    }

654    @Override
655    public RegistrationKey registerUser( String userId, UserRegistrationRequest userRegistrationRequest )
656        throws RedbackServiceException
657    {
658        User user = userRegistrationRequest.getUser( );
659        if ( user == null )
660        {
661            throw new RedbackServiceException( ErrorMessage.of( MessageKeys.ERR_USER_NOT_FOUND, userId ), 404 );
662
663        }
664
665        UserSecurityPolicy securityPolicy = securitySystem.getPolicy( );
666
667        boolean emailValidationRequired = securityPolicy.getUserValidationSettings( ).isEmailValidationRequired( );
668
669        if ( emailValidationRequired )
670        {
671            validateCredentialsLoose( user );
672        }
673        else
674        {
675            validateCredentialsStrict( user );
676        }
677
678        org.apache.archiva.redback.users.User u = null;
679
680        try
681        {
682
683            // NOTE: Do not perform Password Rules Validation Here.
684
685            if ( userManager.userExists( user.getUserId( ) ) )
686            {
687                throw new RedbackServiceException(
688                    ErrorMessage.of( MessageKeys.ERR_USER_EXISTS, user.getUserId() ));
689            }
690
691            u = userManager.createUser( user.getUserId( ), user.getFullName( ), user.getEmail( ) );
692            u.setPassword( user.getPassword( ) );
693            u.setValidated( false );
694            u.setLocked( false );
695
696            roleManager.assignRole( RedbackRoleConstants.REGISTERED_USER_ROLE_ID, u.getUsername( ) );
697        }
698        catch ( RoleManagerException rpe )
699        {
700            log.error( "RoleProfile Error: {}", rpe.getMessage( ), rpe );
701            throw new RedbackServiceException( ErrorMessage.of( MessageKeys.ERR_REGISTRATION_ROLE_ASSIGNMENT_FAILED, rpe.getMessage( ) ), 400 );
702        }
703        catch ( UserManagerException e )
704        {
705            throw new RedbackServiceException( ErrorMessage.of( MessageKeys.ERR_USERMANAGER_FAIL, e.getMessage( ) ), 400 );
706        }
707
708        if ( emailValidationRequired )
709        {
710            u.setLocked( true );
711
712            try
713            {
714                AuthenticationKey authkey =
715                    securitySystem.getKeyManager( ).createKey( u.getUsername( ), "New User Email Validation",
716                        securityPolicy.getUserValidationSettings( ).getEmailValidationTimeout( ) );
717
718                String baseUrl = userRegistrationRequest.getApplicationUrl( );
719                if ( StringUtils.isBlank( baseUrl ) )
720                {
721                    baseUrl = getBaseUrl( );
722                }
723
724                log.debug( "register user {} with email {} and app url {}", u.getUsername( ), u.getEmail( ), baseUrl );
725
726                mailer.sendAccountValidationEmail( Arrays.asList( u.getEmail( ) ), authkey, baseUrl );
727
728                securityPolicy.setEnabled( false );
729                userManager.addUser( u );
730                return new RegistrationKey( authkey.getKey( ), true );
731
732            }
733            catch ( KeyManagerException e )
734            {
735                log.error( "Unable to register a new user.", e );
736                throw new RedbackServiceException( ErrorMessage.of( MessageKeys.ERR_KEYMANAGER_FAIL, e.getMessage( ) ), 400 );
737            }
738            catch ( UserManagerException e )
739            {
740                throw new RedbackServiceException( ErrorMessage.of( MessageKeys.ERR_USERMANAGER_FAIL, e.getMessage( ) ), 400 );
741            }
742            finally
743            {
744                securityPolicy.setEnabled( true );
745            }
746        }
747        else
748        {
749            try
750            {
751                userManager.addUser( u );
752                return new RegistrationKey( "-1", false );
753            }
754            catch ( UserManagerException e )
755            {
756                throw new RedbackServiceException( ErrorMessage.of( MessageKeys.ERR_USERMANAGER_FAIL, e.getMessage( ) ), 400 );
757            }
758        }
759
760        // FIXME log this event
761        /*
762        AuditEvent event = new AuditEvent( getText( "log.account.create" ) );
763        event.setAffectedUser( username );
764        event.log();
765        */
766

807    public VerificationStatus validateUserRegistration( String userId, String key ) throws RedbackServiceException
808    {
809        String principal = null;
810        try
811        {
812            AuthenticationKey authkey = securitySystem.getKeyManager( ).findKey( key );
813
814            org.apache.archiva.redback.users.User user =
815                securitySystem.getUserManager( ).findUser( authkey.getForPrincipal( ) );
816
817            if ( user.isValidated( ) )
818            {
819                throw new RedbackServiceException( ErrorMessage.of( MessageKeys.ERR_REGISTRATION_USER_VALIDATED ), 404 );
820            }
821            user.setValidated( true );
822            user.setLocked( false );
823            user.setPasswordChangeRequired( true );
824            user.setEncodedPassword( "" );
825            securitySystem.getUserManager( ).updateUser( user );
826            principal = user.getUsername( );
827
828            TokenBasedAuthenticationDataSource authsource = new TokenBasedAuthenticationDataSource( );
829            authsource.setPrincipal( principal );
830            authsource.setToken( authkey.getKey( ) );
831            authsource.setEnforcePasswordChange( false );
832
833            VerificationStatus status = new VerificationStatus( false );
834            SecuritySession authStatus = securitySystem.authenticate( authsource );
835            if ( authStatus.isAuthenticated( ) )
836            {
837                Token accessToken = jwtAuthenticator.generateToken( principal );
838                status.setAccessToken( accessToken.getData( ) );
839                status.setSuccess( true );
840            }
841            else
842            {
843                user.setValidated( false );
844                user.setLocked( true );
845                user.setPasswordChangeRequired( false );
846                securitySystem.getUserManager( ).updateUser( user );
847            }
848
849            log.info( "account validated for user {}", user.getUsername( ) );
850
851            return status;
852        }
853        catch ( MustChangePasswordException e )
854        {
855            throw new RedbackServiceException( ErrorMessage.of( MessageKeys.ERR_AUTH_PASSWORD_CHANGE_REQUIRED ), Response.Status.FORBIDDEN.getStatusCode( ) );
856        }
857        catch ( AccountLockedException e )
858        {
859            throw new RedbackServiceException( ErrorMessage.of( MessageKeys.ERR_AUTH_ACCOUNT_LOCKED ), Response.Status.FORBIDDEN.getStatusCode( ) );
860        }
861        catch ( AuthenticationException e )
862        {
863            throw new RedbackServiceException( ErrorMessage.of( MessageKeys.ERR_AUTH_INVALID_CREDENTIALS ), Response.Status.FORBIDDEN.getStatusCode( ) );
864        }
865        catch ( KeyNotFoundException e )
866        {
867            log.info( "Invalid key requested: {}", key );
868            throw new RedbackServiceException( ErrorMessage.of( MessageKeys.ERR_REGISTRATION_KEY_INVALID ), 404 );
869        }
870        catch ( KeyManagerException e )
871        {
872            throw new RedbackServiceException( ErrorMessage.of( MessageKeys.ERR_KEYMANAGER_FAIL, e.getMessage( ) ), 400 );
873
874        }
875        catch ( UserNotFoundException e )
876        {
877            throw new RedbackServiceException( ErrorMessage.of( MessageKeys.ERR_USER_NOT_FOUND, principal ), 404 );
878
879        }
880        catch ( UserManagerException e )
881        {
882            throw new RedbackServiceException( ErrorMessage.of( MessageKeys.ERR_USERMANAGER_FAIL, e.getMessage( ) ), 400 );
883        }
884    }

920    public RoleTree getRoleTree( final String username ) throws RedbackServiceException
921    {
922        final Map<String, String> roleApplicationMap = roleManager.getModel( ).getApplications( ).stream( )
923            .flatMap( modelApplication -> modelApplication.getRoles( ).stream( ).map( role -> {
924                BaseRoleInfo roleInfo = new BaseRoleInfo( );
925                roleInfo.setId( role.getId( ) );
926                roleInfo.setApplicationId( modelApplication.getId( ) );
927                return roleInfo;
928            } ) ).collect( Collectors.toMap( BaseRoleInfo::getId, BaseRoleInfo::getApplicationId ) );
929
930        try
931        {
932            final Set<String> assignedRoleIds = new HashSet( rbacManager.getUserAssignment( username ).getRoleIds( ) );
933            // We have to reuse the BaseRoleInfo objects, because the roles are not returned starting from the roots
934            final Map<String, BaseRoleInfo> roleIdCache = new HashMap<>( );
935            List<BaseRoleInfo> roleList = rbacManager.getAllRoles( ).stream( ).flatMap( this::flattenRole ).map( role ->
936            {
937                BaseRoleInfo roleInfo = roleIdCache.computeIfAbsent( role.getId( ), s -> new BaseRoleInfo( ) );
938                // Setting the role data, as there may be child role objects that are not completely initialized
939                roleInfo = BaseRoleInfo.of( role, roleInfo );
940                roleInfo.setApplicationId( roleApplicationMap.get( role.getId( ) ) );
941                roleInfo.setAssigned( assignedRoleIds.contains( role.getId( ) ) );
942                roleInfo.setChildren( role.getChildRoleIds( ).stream( )
943                    .map( roleId ->
944                    {
945                        BaseRoleInfo childRoleInfo = roleIdCache.computeIfAbsent( roleId, s -> BaseRoleInfo.ofId( roleId ) );
946                        childRoleInfo.setChild( true );
947                        return childRoleInfo;
948                    } )
949                    .collect( Collectors.toList( ) ) );
950                return roleInfo;
951            } ).collect( Collectors.toList( ) );
952            RoleTree roleTree = new RoleTree( );
953            roleTree.setApplications( getAllApplications( ).stream( ).collect( Collectors.toMap( Application::getId, Function.identity( ) ) ) );
954            roleTree.setRootRoles( roleList.stream( ).filter( BaseRoleInfo::isNotChild ).collect( Collectors.toList( ) ) );
955            roleTree.setUserId( username );
956            return roleTree;
957        }
958        catch ( RbacManagerException e )
959        {
960            throw new RedbackServiceException( ErrorMessage.of( MessageKeys.ERR_RBACMANAGER_FAIL, e.getMessage( ) ) );
961        }
962    }

998    private void changeUserPassword(final String userId, final PasswordChange passwordChange) throws RedbackServiceException
999    {
1000        if ( StringUtils.isEmpty( passwordChange.getCurrentPassword() ) )
1001        {
1002            throw new RedbackServiceException( ErrorMessage.of( MessageKeys.ERR_PASSWORDCHANGE_CURRENT_EMPTY ), 400 );
1003        }
1004        if ( passwordChange.getUserId( ) == null || ( !passwordChange.getUserId( ).equals( userId ) ) )
1005        {
1006            throw new RedbackServiceException( ErrorMessage.of( MessageKeys.ERR_USER_ID_INVALID ), 403 );
1007        }
1008
1009        if ( StringUtils.isEmpty( passwordChange.getNewPassword() ) )
1010        {
1011            throw new RedbackServiceException( ErrorMessage.of(MessageKeys.ERR_PASSWORDCHANGE_NEW_EMPTY), 400 );
1012        }
1013        if ( StringUtils.isEmpty( passwordChange.getNewPasswordConfirmation() ) )
1014        {
1015            throw new RedbackServiceException( ErrorMessage.of( MessageKeys.ERR_PASSWORDCHANGE_CONFIRMATION_EMPTY ),
1016                400 );
1017        }
1018        if ( !StringUtils.equals( passwordChange.getNewPassword(), passwordChange.getNewPasswordConfirmation() ) )
1019        {
1020            throw new RedbackServiceException(ErrorMessage.of( MessageKeys.ERR_PASSWORDCHANGE_BAD_CONFIRMATION ),
1021                403 );
1022        }
1023
1024        try
1025        {
1026            org.apache.archiva.redback.users.User u = securitySystem.getUserManager().findUser( userId );
1027
1028            String previousEncodedPassword = u.getEncodedPassword();
1029
1030            // check oldPassword with the current one
1031
1032            PasswordEncoder encoder = securitySystem.getPolicy().getPasswordEncoder();
1033
1034            if ( !encoder.isPasswordValid( previousEncodedPassword, passwordChange.getCurrentPassword() ) )
1035            {
1036
1037                throw new RedbackServiceException( MessageKeys.ERR_AUTH_INVALID_CREDENTIALS,
1038                    401 );
1039            }
1040
1041            u.setPassword( passwordChange.getNewPassword() );
1042            securitySystem.getUserManager().updateUser( u );
1043        }
1044        catch ( UserNotFoundException e )
1045        {
1046            throw new RedbackServiceException( ErrorMessage.of( MessageKeys.ERR_USER_NOT_FOUND ),
1047                400 );
1048        }
1049        catch ( UserManagerException e )
1050        {
1051            log.info( "UserManagerException: {}", e.getMessage() );
1052            throw new RedbackServiceException( ErrorMessage.of( MessageKeys.ERR_USERMANAGER_FAIL, e.getMessage() ) );
1053        }
1054        catch ( PasswordRuleViolationException e )
1055        {
1056            throw new RedbackServiceException( getPasswordViolationMessages( e ), 401 );
1057        }
1058
1059
1060    }
