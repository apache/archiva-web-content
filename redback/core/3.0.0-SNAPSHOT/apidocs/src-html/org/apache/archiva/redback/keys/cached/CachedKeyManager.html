022import org.apache.archiva.components.cache.Cache;
023import org.apache.archiva.redback.keys.AbstractKeyManager;
024import org.apache.archiva.redback.keys.AuthenticationKey;
025import org.apache.archiva.redback.keys.KeyManager;
026import org.apache.archiva.redback.keys.KeyManagerException;
027import org.apache.archiva.redback.keys.KeyNotFoundException;
028import org.springframework.stereotype.Service;
029
030import javax.annotation.PreDestroy;
031import javax.inject.Inject;
032import javax.inject.Named;
033import java.util.List;
034
035/**
036 * CachedKeyManager
037 *
038 * @author <a href="mailto:joakim@erdfelt.com">Joakim Erdfelt</a>
039 */
040@Service("keyManager#cached")
041public class CachedKeyManager
042    extends AbstractKeyManager
043    implements KeyManager
044{
045    @Inject
046    @Named(value = "keyManager#jpa")
047    private KeyManager keyImpl;
048
049    @Inject
050    @Named(value = "cache#keys")
051    private Cache<String, AuthenticationKey> keysCache;
052
053    public AuthenticationKey addKey( AuthenticationKey key )
054    {
055        if ( key != null )
056        {
057            keysCache.remove( key.getKey() );
058        }
059        return this.keyImpl.addKey( key );
060    }
061
062    public AuthenticationKey createKey( String principal, String purpose, int expirationMinutes )
063        throws KeyManagerException
064    {
065        AuthenticationKey authkey = this.keyImpl.createKey( principal, purpose, expirationMinutes );
066        keysCache.remove( authkey.getKey() );
067        return authkey;
068    }
069
070    public void deleteKey( AuthenticationKey key )
071        throws KeyManagerException
072    {
073        keysCache.remove( key.getKey() );
074        this.keyImpl.deleteKey( key );
075    }
076
077    public void deleteKey( String key )
078        throws KeyManagerException
079    {
080        keysCache.remove( key );
081        this.keyImpl.deleteKey( key );
082    }
083
084    public void eraseDatabase()
085    {
086        try
087        {
088            this.keyImpl.eraseDatabase();
089        }
090        finally
091        {
092            this.keysCache.clear();
093        }
094    }
095
096    public AuthenticationKey findKey( String key )
097        throws KeyManagerException
098    {
099        try
100        {
101            AuthenticationKey authkey = keysCache.get( key );
102            if ( authkey != null )
103            {
104                assertNotExpired( authkey );
105                return authkey;
106            }
107            else
108            {
109                authkey = this.keyImpl.findKey( key );
110                keysCache.put( key, authkey );
111                return authkey;
112            }
113        }
114        catch ( KeyNotFoundException knfe )
115        {
116            // this is done to remove keys that have been expired.
117            // TODO: need to make a listener for the key manager.
118            keysCache.remove( key );
119            throw knfe;
120        }
121    }
122
123    public List<AuthenticationKey> getAllKeys()
124    {
125        log.debug( "NOT CACHED - .getAllKeys()" );
126        return this.keyImpl.getAllKeys();
127    }
128
129    public String getId()
130    {
131        return "Cached Key Manager [" + this.keyImpl.getId() + "]";
132    }
133
134    public KeyManager getKeyImpl()
135    {
136        return keyImpl;
137    }
138
139    public void setKeyImpl( KeyManager keyImpl )
140    {
141        this.keyImpl = keyImpl;
142    }
143
144    public Cache getKeysCache()
145    {
146        return keysCache;
147    }
148
149    public void setKeysCache( Cache<String, AuthenticationKey> keysCache )
150    {
151        this.keysCache = keysCache;
152    }
153
154    public void clearCache() {
155        keysCache.clear();
156    }
157
158    @PreDestroy
159    void shutdown() {
160        clearCache( );
161    }
162}
