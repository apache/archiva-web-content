021import javax.xml.bind.annotation.XmlRootElement;
022import java.util.Base64;
023
024/**
025 * @author Martin Stockhammer <martin_s@apache.org>
026 */
027@XmlRootElement(name="userLogin")
028public class UserLogin extends User
029{
030
031    String authToken;
032    String base64AuthToken;
033
034    public UserLogin( )
035    {
036    }
037
038    public UserLogin( String username, String fullName, String email, boolean validated, boolean locked, String authToken )
039    {
040        super( username, fullName, email, validated, locked );
041        this.authToken = authToken;
042        this.base64AuthToken = Base64.getEncoder( ).encodeToString( authToken.getBytes( ) );
043    }
044
045    public UserLogin( org.apache.archiva.redback.users.User user, String authToken )
046    {
047        super( user );
048        this.authToken = authToken;
049        this.base64AuthToken = Base64.getEncoder( ).encodeToString( authToken.getBytes( ) );
050    }
051
052    public String getAuthToken( )
053    {
054        return authToken;
055    }
056
057    public void setAuthToken( String authToken )
058    {
059        this.authToken = authToken;
060    }
061
062    public String getBase64AuthToken( )
063    {
064        return base64AuthToken;
065    }
066
067    public void setBase64AuthToken( String base64AuthToken )
068    {
069        this.base64AuthToken = base64AuthToken;
070    }
071}
