074    @Override
075    public Role saveRole( Role role )
076        throws RbacManagerException
077    {
078        RBACObjectAssertions.assertValid( "Save Role", role );
079
080        triggerInit();
081
082        roles.put( role.getName(), role );
083
084        fireRbacRoleSaved( role );
085
086        if ( role.getPermissions() != null )
087        {
088            for ( Permission p : role.getPermissions() )
089            {
090                savePermission( p );
091            }
092        }
093
094        return role;
095    }
096
097    @Override
098    public void saveRoles( Collection<Role> roles )
099        throws RbacObjectInvalidException, RbacManagerException
100    {
101        if ( roles == null )
102        {
103            // Nothing to do.
104            return;
105        }
106
107        for ( Role role : roles )
108        {
109            saveRole( role );
110        }
111    }
112
113    private void assertRoleExists( String roleName )
114        throws RbacObjectNotFoundException
115    {
116        if ( !roles.containsKey( roleName ) )
117        {
118            throw new RbacObjectNotFoundException( "Role '" + roleName + "' does not exist." );
119        }
120    }
121
122    @Override
123    public Role getRole( String roleName )
124        throws RbacObjectNotFoundException
125    {
126        triggerInit();
127
128        assertRoleExists( roleName );
129
130        return roles.get( roleName );
131    }
132
133    @Override
134    public Role getRoleById( String id ) throws RbacObjectNotFoundException, RbacManagerException
135    {
136        triggerInit();
137        return roles.values( ).stream( ).filter( role -> StringUtils.equals( role.getId( ), id ) ).findAny( ).orElseThrow(
138            () -> new RbacObjectNotFoundException( "Role with id " + id + " not found" )
139        );
140    }
141
142    @Override
143    public void removeRole( Role role )
144        throws RbacManagerException, RbacObjectNotFoundException
145    {
146        RBACObjectAssertions.assertValid( "Remove Role", role );
147
148        if ( role.isPermanent() )
149        {
150            throw new RbacPermanentException( "Unable to delete permanent role [" + role.getName() + "]" );
151        }
152
153        assertRoleExists( role.getName() );
154
155        fireRbacRoleRemoved( role );
156
157        roles.remove( role.getName() );
158    }

162        throws RbacManagerException

266    @Override
267    public Permission createPermission( String name, String operationName, String resourceIdentifier )
268        throws RbacManagerException
269    {
270        Permission permission;
271
272        try
273        {
274            permission = getPermission( name );
275
276            if ( StringUtils.equals( operationName, permission.getOperation().getName() ) )
277            {
278                throw new RbacManagerException( "Attempted to create a permission named '" + name +
279                                                    "' with an operation named '" + operationName
280                                                    + "', but that overides the existing '" + name +
281                                                    "' permission with operation '"
282                                                    + permission.getOperation().getName() + "'" );
283            }
284
285        }
286        catch ( RbacObjectNotFoundException e )
287        {
288            permission = new MemoryPermission();
289            permission.setName( name );
290
291            permission.setOperation( createOperation( operationName ) );
292            permission.setResource( createResource( resourceIdentifier ) );
293        }
294
295        return permission;
296    }

517    @Override
518    public List<UserAssignment> getUserAssignmentsForRoles( Collection<String> roleIds )
519        throws RbacManagerException
520    {
521
522        List<UserAssignment> allUserAssignments = getAllUserAssignments();
523        List<UserAssignment> userAssignments = new ArrayList<>( allUserAssignments.size( ) );
524
525        for ( UserAssignment ua : allUserAssignments )
526        {
527            for ( String roleId : roleIds )
528            {
529                if ( ua.getRoleIds().contains( roleId ) )
530                {
531                    userAssignments.add( ua );
532                    break;
533                }
534            }
535        }
536
537        return userAssignments;
538    }
