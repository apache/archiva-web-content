020import org.apache.commons.lang3.StringUtils;
021
022/**
023 * RBACObjectAssertions 
024 *
025 * @author <a href="mailto:joakim@erdfelt.com">Joakim Erdfelt</a>
026 *
027 */
028public class RBACObjectAssertions
029{
030    public static void assertValid( Role role )
031        throws RbacObjectInvalidException
032    {
033        assertValid( null, role );
034    }
035
036    public static void assertValid( String scope, Role role )
037        throws RbacObjectInvalidException
038    {
039        if ( role == null )
040        {
041            throw new RbacObjectInvalidException( scope, "Null Role object is invalid." );
042        }
043
044        if ( StringUtils.isEmpty( role.getName() ) )
045        {
046            throw new RbacObjectInvalidException( scope, "Role.name must not be empty." );
047        }
048
049        if ( role.getPermissions() != null )
050        {
051            int i = 0;
052            for ( Permission perm : role.getPermissions() )
053            {
054                assertValid( "Role.permissions[" + i + "]", perm );
055                i++;
056            }
057        }
058    }
059
060    public static void assertValid( Permission permission )
061        throws RbacObjectInvalidException
062    {
063        assertValid( null, permission );
064    }
065
066    public static void assertValid( String scope, Permission permission )
067        throws RbacObjectInvalidException
068    {
069        if ( permission == null )
070        {
071            throw new RbacObjectInvalidException( scope, "Null Permission object is invalid." );
072        }
073
074        if ( StringUtils.isEmpty( permission.getName() ) )
075        {
076            throw new RbacObjectInvalidException( scope, "Permission.name must not be empty." );
077        }
078
079        assertValid( "Permission.operation", permission.getOperation() );
080        assertValid( "Permission.resource", permission.getResource() );
081
082    }
083
084    public static void assertValid( Operation operation )
085        throws RbacObjectInvalidException
086    {
087        assertValid( null, operation );
088    }
089
090    public static void assertValid( String scope, Operation operation )
091        throws RbacObjectInvalidException
092    {
093        if ( operation == null )
094        {
095            throw new RbacObjectInvalidException( scope, "Null Operation object is invalid." );
096        }
097
098        if ( StringUtils.isEmpty( operation.getName() ) )
099        {
100            throw new RbacObjectInvalidException( scope, "Operation.name must not be empty." );
101        }
102    }
103
104    public static void assertValid( Resource resource )
105        throws RbacObjectInvalidException
106    {
107        assertValid( null, resource );
108    }
109
110    public static void assertValid( String scope, Resource resource )
111        throws RbacObjectInvalidException
112    {
113        if ( resource == null )
114        {
115            throw new RbacObjectInvalidException( scope, "Null Resource object is invalid." );
116        }
117
118        if ( StringUtils.isEmpty( resource.getIdentifier() ) )
119        {
120            throw new RbacObjectInvalidException( scope, "Resource.identifier must not be empty." );
121        }
122    }
123
124    public static void assertValid( UserAssignment assignment )
125        throws RbacObjectInvalidException
126    {
127        assertValid( null, assignment );
128    }
129
130    public static void assertValid( String scope, UserAssignment assignment )
131        throws RbacObjectInvalidException
132    {
133        if ( assignment == null )
134        {
135            throw new RbacObjectInvalidException( scope, "Null UserAssigment object is invalid." );
136        }
137
138        if ( StringUtils.isEmpty( assignment.getPrincipal() ) )
139        {
140            throw new RbacObjectInvalidException( scope, "UserAssigment.principal cannot be empty." );
141        }
142
143        if ( assignment.getRoleIds() == null )
144        {
145            throw new RbacObjectInvalidException( scope, "UserAssignment.roles cannot be null." );
146        }
147
148        /*  I don't believe this assertion is valid, a person should be able to be stripped of all roles.
149           -- jesse
150        if ( assignment.getRoleNames().isEmpty() )
151        {
152            throw new RbacObjectInvalidException( scope, "UserAssignment.roles cannot be empty." );
153        }
154          */
155        int i = 0;
156        for ( String name : assignment.getRoleIds() )
157        {
158            if ( StringUtils.isEmpty( name ) )
159            {
160                throw new RbacObjectInvalidException( scope, "UserAssignment.roleid[" + i + "] cannot be empty." );
161            }
162            i++;
163        }
164    }
165
166}
