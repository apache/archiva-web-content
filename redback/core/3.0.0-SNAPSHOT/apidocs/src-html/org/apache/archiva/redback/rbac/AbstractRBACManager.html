417    /**
418     * returns a set of all permissions that are in all active roles for a given
419     * principal
420     *
421     * @param principal
422     * @return
423     * @throws RbacObjectNotFoundException
424     * @throws RbacManagerException
425     */
426    @Override
427    public Set<? extends Permission> getAssignedPermissions( String principal )
428        throws RbacObjectNotFoundException, RbacManagerException
429    {
430
431        UserAssignment ua = getUserAssignment( principal );
432
433        Set<Permission> permissionSet = new HashSet<Permission>();
434
435        if ( ua.getRoleIds() != null )
436        {
437            boolean childRoleNamesUpdated = false;
438
439            Iterator<String> it = ua.getRoleIds().listIterator();
440            while ( it.hasNext() )
441            {
442                String roleId = it.next();
443                try
444                {
445                    Role role = getRoleById( roleId );
446                    gatherUniquePermissions( role, permissionSet );
447                }
448                catch ( RbacObjectNotFoundException e )
449                {
450                    // Found a bad role name. remove it!
451                    it.remove();
452                    childRoleNamesUpdated = true;
453                }
454            }
455
456            if ( childRoleNamesUpdated )
457            {
458                saveUserAssignment( ua );
459            }
460        }
461
462        return permissionSet;
463    }

480    private Map<String, List<? extends Permission>> getPermissionMapByOperation( Collection<? extends Permission> permissions )
481    {
482        Map<String, List<? extends Permission>> tList = new HashMap<>(  );
483        tList.putAll( permissions.stream().collect( Collectors.<Permission, String>groupingBy( p -> p.getOperation( ).getName( ) ) ) );
484        return tList;
485    }
486
487    private void gatherUniquePermissions( Role role, Collection<Permission> coll )
488        throws RbacManagerException
489    {
490        if ( role.getPermissions() != null )
491        {
492            for ( Permission permission : role.getPermissions() )
493            {
494                if ( !coll.contains( permission ) )
495                {
496                    coll.add( permission );
497                }
498            }
499        }
500
501        if ( role.hasChildRoles() )
502        {
503            Map<String, ? extends Role> childRoles = getChildRoleNames( role );
504            Iterator<? extends Role> it = childRoles.values().iterator();
505            while ( it.hasNext() )
506            {
507                Role child = it.next();
508                gatherUniquePermissions( child, coll );
509            }
510        }
511    }

600    /**
601     * get all of the roles that the give role has as a child into a set
602     *
603     * @param role
604     * @param roleSet
605     * @throws RbacObjectNotFoundException
606     * @throws RbacManagerException
607     */
608    private void gatherEffectiveRoles( Role role, Set<Role> roleSet )
609        throws RbacObjectNotFoundException, RbacManagerException
610    {
611        if ( role.hasChildRoles() )
612        {
613            for ( String roleName : role.getChildRoleNames() )
614            {
615                try
616                {
617                    Role crole = getRole( roleName );
618
619                    if ( !roleSet.contains( crole ) )
620                    {
621                        gatherEffectiveRoles( crole, roleSet );
622                    }
623                }
624                catch ( RbacObjectNotFoundException e )
625                {
626                    // the client application might not manage role clean up totally correctly so we want to notify
627                    // of a child role issue and offer a clean up process at some point
628                    log.warn( "dangling child role: {} on {}", roleName, role.getName() );
629                }
630            }
631        }
632
633        if ( !roleSet.contains( role ) )
634        {
635            roleSet.add( role );
636        }
637    }

743    @Override
744    public Map<String, ? extends Role> getChildRoleNames( Role role )
745        throws RbacManagerException
746    {
747        Map<String, Role> childRoles = new HashMap<String, Role>();
748
749        boolean childRoleNamesUpdated = false;
750
751        Iterator<String> it = role.getChildRoleNames().listIterator();
752
753        List<String> updatedChildRoleList = new ArrayList<String>( role.getChildRoleNames().size() );
754
755        while ( it.hasNext() )
756        {
757            String roleName = it.next();
758            try
759            {
760                Role child = getRole( roleName );
761                // archiva can change role manager but LDAP can be non writable so in such case
762                // some roles doesn't exists !!
763                if ( child != null )
764                {
765                    childRoles.put( child.getName(), child );
766                    updatedChildRoleList.add( roleName );
767                }
768                else
769                {
770                    log.warn(
771                        "error searching role with name '{}' probably some issues when migrating your role manager",
772                        roleName );
773                }
774            }
775            catch ( RbacObjectNotFoundException e )
776            {
777                // Found a bad roleName! - trigger new List save
778                //it.remove();
779                childRoleNamesUpdated = true;
780            }
781            catch ( RbacManagerException e )
782            {
783                if ( !( e.getCause() instanceof RbacObjectNotFoundException ) )
784                {
785                    throw e;
786                }
787                childRoleNamesUpdated = true;
788            }
789        }
790
791        if ( childRoleNamesUpdated )
792        {
793            role.setChildRoleNames( updatedChildRoleList );
794            saveRole( role );
795        }
796
797        return childRoles;
798    }

857    @Override
858    public Map<String, ? extends Role> getParentRoleNames( Role role )
859        throws RbacManagerException
860    {
861        Map<String, Role> parentRoles = new HashMap<String, Role>();
862
863        for ( Role r : getAllRoles() )
864        {
865            if ( !r.getName().equals( role.getName() ) )
866            {
867                Set<? extends Role> effectiveRoles = getEffectiveRoles( r );
868                for ( Role currentRole : effectiveRoles )
869                {
870                    if ( currentRole.getName().equals( role.getName() ) )
871                    {
872                        if ( !parentRoles.containsKey( r.getName() ) )
873                        {
874                            parentRoles.put( r.getName(), r );
875                        }
876                    }
877                }
878            }
879        }
880        return parentRoles;
881    }

883    @Override
884    public Map<String, ? extends Role> getParentRoleIds( final Role role ) throws RbacManagerException
885    {
886        return getAllRoles( ).stream( ).filter( r -> !r.getId( ).equals( role.getId( ) ) )
887            .filter( r -> {
888                    try
889                    {
890                        return getEffectiveRoles( r ).stream( ).map( Role::getId ).filter( cRoleId -> cRoleId.equals( role.getId( ) ) ).findAny( ).isPresent( );
891                    }
892                    catch ( RbacManagerException e )
893                    {
894                        return false;
895                    }
896                }
897            ).distinct().collect( Collectors.toMap( Role::getId, Function.identity( ) ) );
898    }
