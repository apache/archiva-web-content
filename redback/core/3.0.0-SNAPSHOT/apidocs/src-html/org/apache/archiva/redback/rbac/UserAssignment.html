022/**
023 * UserAssignment - This the mapping object that takes the principal for a user and associates it with a
024 * set of Roles.
025 * 
026 * This is the many to many mapping object needed by persistence stores.
027 *
028 * @author Jesse McConnell
029 * @author <a href="mailto:joakim@erdfelt.com">Joakim Erdfelt</a>
030 *
031 * TODO expand on javadoc
032 */
033public interface UserAssignment extends Serializable
034{
035
036    
037    /**
038     * The principal for the User that the set of roles is associated with.
039     * 
040     * NOTE: This field is considered the Primary Key for this object.
041     * 
042     * @return the principal for the User.
043     */
044    String getPrincipal();
045
046    /**
047     * Get the roles for this user.
048     * 
049     * @return List of &lt;{@link String}&gt; objects representing the Role Names.
050     */
051    List<String> getRoleNames();
052
053    List<String> getRoleIds();
054
055    /**
056     * Add a rolename to this assignment.
057     * 
058     * @param role the role.
059     */
060    void addRoleName( Role role );
061    
062    /**
063     * Add a rolename to this assignment.
064     * 
065     * @param roleName the role name.
066     */
067    void addRoleName( String roleName );
068
069    void addRoleId( Role role );
070    /**
071     * Adds a role id to this assignment
072     * @param roleId
073     */
074    void addRoleId( String roleId );
075    
076    /**
077     * Remove a rolename from this assignment.
078     * 
079     * @param role the role who's name is to be removed.
080     */
081    void removeRoleName( Role role );
082    
083    /**
084     * Remove a role name from this assignment.
085     * 
086     * @param roleName the role name to be removed.
087     */
088    void removeRoleName( String roleName );
089
090    void removeRoleId( Role role );
091
092    void removeRoleId( String roleId );
093
094    /**
095     * Set the user principal object for this association.
096     * 
097     * NOTE: This field is considered the Primary Key for this object.
098     * 
099     * @param principal
100     */
101    void setPrincipal( String principal );
102
103    /**
104     * Set the roles names for this user.
105     * 
106     * @param roles the List of &lt;{@link String}&gt; objects representing the Role Names.
107     */
108    void setRoleNames( List<String> roles );
109
110    void setRoleIds( List<String> roles );
111    
112    /**
113     * Test to see if the object is a permanent object or not.
114     * 
115     * @return true if the object is permanent.
116     */
117    boolean isPermanent();
118
119    /**
120     * Set flag indicating if the object is a permanent object or not.
121     * 
122     * @param permanent true if the object is permanent.
123     */
124    void setPermanent( boolean permanent );
125}
