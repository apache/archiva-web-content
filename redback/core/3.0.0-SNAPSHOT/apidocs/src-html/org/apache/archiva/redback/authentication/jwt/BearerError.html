021/**
022 * @author Martin Stockhammer <martin_s@apache.org>
023 */
024public enum BearerError
025{
026    INVALID_REQUEST("invalid_request",1024),INVALID_TOKEN("invalid_token", 1025), INSUFFICIENT_SCOPE( "insufficient_scope", 1026 );
027
028    protected String errorString;
029    protected int id;
030    BearerError(String errorString, int id) {
031        this.errorString = errorString;
032        this.id = id;
033    }
034
035    public String getError() {
036        return this.errorString;
037    }
038
039    public int getId() {
040        return this.id;
041    }
042
043    public static BearerError get(int id) {
044        final BearerError[] values = BearerError.values();
045        for ( int i = 0 ; i<values.length; i++) {
046            if (values[i].id==id) {
047                return values[i];
048            }
049        }
050        return null;
051    }
052}
