022import org.apache.archiva.redback.users.User;
023import org.apache.archiva.redback.users.UserManager;
024import org.apache.archiva.redback.users.UserManagerException;
025import org.apache.commons.lang.StringUtils;
026
027
028import java.io.Serializable;
029
030/**
031 * UserCredentials
032 *
033 * @author <a href="mailto:joakim@erdfelt.com">Joakim Erdfelt</a>
034 */
035public abstract class UserCredentials
036    implements Serializable
037{
038    // Potentially Editable Field.
039    private String username;
040
041    // Editable Fields.
042    private String password;
043
044    private String confirmPassword;
045
046    private String fullName;
047
048    private String email;
049
050    // Display Only Fields.
051    private String timestampAccountCreation;
052
053    private String timestampLastLogin;
054
055    private String timestampLastPasswordChange;
056
057    public User createUser( UserManager um )
058        throws UserManagerException
059    {
060        User user = um.createUser( username, fullName, email );
061
062        user.setPassword( password );
063
064        return user;
065    }
066
067    public String toString()
068    {
069        StringBuilder sb = new StringBuilder();
070
071        sb.append( "UserCredentials[" );
072        sb.append( "username=" ).append( username );
073        sb.append( ",fullName=" ).append( fullName );
074        sb.append( ",email=" ).append( email );
075        sb.append( ",password=" );
076        if ( StringUtils.isNotEmpty( password ) )
077        {
078            sb.append( "<***>" );
079        }
080        else
081        {
082            sb.append( "<empty>" );
083        }
084        sb.append( ",confirmPassword=" );
085        if ( StringUtils.isNotEmpty( confirmPassword ) )
086        {
087            sb.append( "<***>" );
088        }
089        else
090        {
091            sb.append( "<empty>" );
092        }
093
094        return sb.append( "]" ).toString();
095    }
096
097    public String getConfirmPassword()
098    {
099        return confirmPassword;
100    }
101
102    public void setConfirmPassword( String confirmPassword )
103    {
104        this.confirmPassword = confirmPassword;
105    }
106
107    public String getEmail()
108    {
109        return email;
110    }
111
112    public void setEmail( String email )
113    {
114        this.email = email;
115    }
116
117    public String getFullName()
118    {
119        return fullName;
120    }
121
122    public void setFullName( String fullName )
123    {
124        this.fullName = fullName;
125    }
126
127    public String getPassword()
128    {
129        return password;
130    }
131
132    public void setPassword( String password )
133    {
134        this.password = password;
135    }
136
137    public String getUsername()
138    {
139        return username;
140    }
141
142    public void setUsername( String username )
143    {
144        this.username = username;
145    }
146
147    public abstract boolean isEdit();
148
149    public String getTimestampAccountCreation()
150    {
151        return timestampAccountCreation;
152    }
153
154    public String getTimestampLastLogin()
155    {
156        return timestampLastLogin;
157    }
158
159    public String getTimestampLastPasswordChange()
160    {
161        return timestampLastPasswordChange;
162    }
163
164    public void setTimestampAccountCreation( String timestampAccountCreation )
165    {
166        this.timestampAccountCreation = timestampAccountCreation;
167    }
168
169    public void setTimestampLastLogin( String timestampLastLogin )
170    {
171        this.timestampLastLogin = timestampLastLogin;
172    }
173
174    public void setTimestampLastPasswordChange( String timestampLastPasswordChange )
175    {
176        this.timestampLastPasswordChange = timestampLastPasswordChange;
177    }
178}
