055/**
056 * Created by Martin Stockhammer on 19.01.17.
057 * <p>
058 * This interceptor tries to check if requests come from a valid origin and
059 * are not generated by another site on behalf of the real client.
060 * <p>
061 * We are using some of the techniques mentioned in
062 * https://www.owasp.org/index.php/Cross-Site_Request_Forgery_(CSRF)_Prevention_Cheat_Sheet
063 * <p>
064 * Try to find Origin and Referer of the request.
065 * Match them to the target address, that may be either statically configured or is determined
066 * by the Host/X-Forwarded-For Header.
067 */
068@Provider
069@Service( "requestValidationInterceptor#rest" )
070public class RequestValidationInterceptor
071    extends AbstractInterceptor
072    implements ContainerRequestFilter
073{

268        // Origin check for Protocol, Host, Port
269        public void checkOrigin()
270        {
271            if ( this.getStatus() == UNKNOWN )
272            {
273                this.status = OK;
274            }
275            if ( !targetUrl.getProtocol().equals( originUrl.getProtocol() ) )
276            {
277                setStatus( F_ORIGIN_PROTOCOL );
278            }
279            if ( !targetHost.equals( originHost ) )
280            {
281                setStatus( F_ORIGIN_HOST );
282            }
283            if ( targetPort != originPort )
284            {
285                setStatus( F_ORIGIN_PORT );
286            }
287        }
288
289        // Referer check only for Host, Port
290        public void checkReferer()
291        {
292            if ( this.getStatus() == UNKNOWN )
293            {
294                this.status = OK;
295            }
296            if ( !targetHost.equals( refererHost ) )
297            {
298                setStatus( F_REFERER_HOST );
299            }
300            if ( targetPort != refererPort )
301            {
302                setStatus( F_REFERER_PORT );
303            }
304        }
305
306        public boolean hasOriginError()
307        {
308            return ( status & ( F_ORIGIN_PROTOCOL | F_ORIGIN_HOST | F_ORIGIN_PORT ) ) > 0;
309        }
310
311        public boolean hasRefererError()
312        {
313            return ( status & ( F_REFERER_HOST | F_REFERER_PORT ) ) > 0;
314        }
315
316        @Override
317        public String toString()
318        {
319            return "Stat=" + status + ", target=" + targetUrl + ", origin=" + originUrl + ", referer=" + refererUrl;
320        }
321    }

329    @PostConstruct
330    public void init()
331    {
332        List<String> baseUrlList = config.getList( UserConfigurationKeys.REST_BASE_URL );
333        if ( baseUrlList != null )
334        {
335            for ( String baseUrlStr : baseUrlList )
336            {
337                if ( !"".equals( baseUrlStr.trim() ) )
338                {
339                    try
340                    {
341                        baseUrl.add( new URL( baseUrlStr ) );
342                        useStaticUrl = true;
343                    }
344                    catch ( MalformedURLException ex )
345                    {
346                        log.error( "Configured baseUrl (rest.baseUrl={}) is invalid. Message: {}", baseUrlStr,
347                            ex.getMessage() );
348                    }
349                }
350            }
351        }
352        denyAbsentHeaders = config.getBoolean( UserConfigurationKeys.REST_CSRF_ABSENTORIGIN_DENY, true );
353        enabled = config.getBoolean( UserConfigurationKeys.REST_CSRF_ENABLED, true );
354        if ( !enabled )
355        {
356            log.info( "CSRF Filter is disabled by configuration" );
357        }
358        else
359        {
360            log.info( "CSRF Filter is enable" );
361        }
362        checkToken = !config.getBoolean( UserConfigurationKeys.REST_CSRF_DISABLE_TOKEN_VALIDATION, false );
363        if ( !checkToken )
364        {
365            log.info( "CSRF Token validation is disabled by configuration" );
366        }
367        else
368        {
369            log.info( "CSRF Token validation is enable" );
370        }
371    }

373    @Override
374    public void filter( ContainerRequestContext containerRequestContext )
375        throws IOException
376    {
377        if ( enabled )
378        {
379            HttpServletRequest request = getRequest();
380            List<URL> targetUrls = getTargetUrl( request );
381            if ( targetUrls == null )
382            {
383                log.error( "Could not verify target URL." );
384                containerRequestContext.abortWith( Response.status( Response.Status.FORBIDDEN ).build() );
385                return;
386            }
387            List<HeaderValidationInfo> validationInfos = new ArrayList<HeaderValidationInfo>();
388            boolean targetMatch = false;
389            boolean noHeader = true;
390            for ( URL targetUrl : targetUrls )
391            {
392                log.trace( "Checking against target URL: {}", targetUrl );
393                HeaderValidationInfo info = checkSourceRequestHeader( new HeaderValidationInfo( targetUrl ), request );
394                // We need only one match
395                noHeader = noHeader && info.getStatus() == info.UNKNOWN;
396                if ( info.getStatus() == info.OK )
397                {
398                    targetMatch = true;
399                    break;
400                }
401                else
402                {
403                    validationInfos.add( info );
404                }
405            }
406            if ( noHeader && denyAbsentHeaders )
407            {
408                log.warn( "Request denied. No Origin or Referer header found and {}=true",
409                    UserConfigurationKeys.REST_CSRF_ABSENTORIGIN_DENY );
410                containerRequestContext.abortWith( Response.status( Response.Status.FORBIDDEN ).build() );
411                return;
412            }
413            if ( !targetMatch )
414            {
415                log.warn( "HTTP Header check failed. Assuming CSRF attack." );
416                for ( HeaderValidationInfo info : validationInfos )
417                {
418                    if ( info.hasOriginError() )
419                    {
420                        log.warn(
421                            "Origin Header does not match: originUrl={}, targetUrl={}. Matches: Host={}, Port={}, Protocol={}",
422                            info.originUrl, info.targetUrl, ( info.getStatus() & info.F_ORIGIN_HOST ) == 0,
423                            ( info.getStatus() & info.F_ORIGIN_PORT ) == 0,
424                            ( info.getStatus() & info.F_ORIGIN_PROTOCOL ) == 0 );
425                    }
426                    if ( info.hasRefererError() )
427                    {
428                        log.warn(
429                            "Referer Header does not match: refererUrl={}, targetUrl={}. Matches: Host={}, Port={}",
430                            info.refererUrl, info.targetUrl, ( info.getStatus() & info.F_REFERER_HOST ) == 0,
431                            ( info.getStatus() & info.F_REFERER_PORT ) == 0 );
432                    }
433                }
434                containerRequestContext.abortWith( Response.status( Response.Status.FORBIDDEN ).build() );
435                return;
436            }
437            if ( checkToken )
438            {
439                checkValidationToken( containerRequestContext, request );
440            }
441        }
442    }

444    /**
445     * Checks the request for a validation token header. It takes the encrypted token, decrypts it
446     * and compares the user information from the token to the logged in user.
447     *
448     * @param containerRequestContext
449     * @param request
450     */
451    private void checkValidationToken( ContainerRequestContext containerRequestContext, HttpServletRequest request )
452    {
453        Message message = JAXRSUtils.getCurrentMessage();
454        RedbackAuthorization redbackAuthorization = getRedbackAuthorization( message );
455        // We check only services that are restricted
456        if ( !redbackAuthorization.noRestriction() )
457        {
458            String tokenString = request.getHeader( X_XSRF_TOKEN );
459            if ( tokenString == null || tokenString.length() == 0 )
460            {
461                log.warn( "No validation token header found: {}", X_XSRF_TOKEN );
462                containerRequestContext.abortWith( Response.status( Response.Status.FORBIDDEN ).build() );
463                return;
464            }
465
466            try
467            {
468                TokenData td = tokenManager.decryptToken( tokenString );
469                AuthenticationResult auth = getAuthenticationResult( message, request );
470                if ( auth == null )
471                {
472                    log.error( "Not authentication data found" );
473                    containerRequestContext.abortWith( Response.status( Response.Status.FORBIDDEN ).build() );
474                    return;
475                }
476                User loggedIn = auth.getUser();
477                if ( loggedIn == null )
478                {
479                    log.error( "User not logged in" );
480                    containerRequestContext.abortWith( Response.status( Response.Status.FORBIDDEN ).build() );
481                    return;
482                }
483                String username = loggedIn.getUsername();
484                if ( !td.isValid() || !td.getUser().equals( username ) )
485                {
486                    log.error( "Invalid data in validation token header {} for user {}: isValid={}, username={}",
487                        X_XSRF_TOKEN, username, td.isValid(), td.getUser() );
488                    containerRequestContext.abortWith( Response.status( Response.Status.FORBIDDEN ).build() );
489                }
490            }
491            catch ( InvalidTokenException e )
492            {
493                log.error( "Token validation failed {}", e.getMessage() );
494                containerRequestContext.abortWith( Response.status( Response.Status.FORBIDDEN ).build() );
495            }
496        }
497        log.debug( "Token validated" );
498    }

500    private HttpServletRequest getRequest()
501    {
502        if ( httpRequest != null )
503        {
504            return httpRequest;

507        {

509            return getHttpServletRequest( message );

513    private List<URL> getTargetUrl( HttpServletRequest request )
514    {
515        if ( useStaticUrl )
516        {
517            return baseUrl;
518        }
519        else
520        {
521            List<URL> urls = new ArrayList<URL>();
522            URL requestUrl;
523            try
524            {
525                requestUrl = new URL( request.getRequestURL().toString() );
526                urls.add( requestUrl );
527            }
528            catch ( MalformedURLException ex )
529            {
530                log.error( "Bad Request URL {}, Message: {}", request.getRequestURL(), ex.getMessage() );
531                return null;
532            }
533            String xforwarded = request.getHeader( X_FORWARDED_HOST );
534            String xforwardedProto = request.getHeader( X_FORWARDED_PROTO );
535            if ( xforwardedProto == null )
536            {
537                xforwardedProto = requestUrl.getProtocol();
538            }
539
540            if ( xforwarded != null && !StringUtils.isEmpty( xforwarded ) )
541            {
542                // X-Forwarded-Host header may contain multiple hosts if there is
543                // more than one proxy between the client and the server
544                String[] forwardedList = xforwarded.split( "\\s*,\\s*" );
545                for ( String hostname : forwardedList )
546                {
547                    try
548                    {
549                        urls.add( new URL( xforwardedProto + "://" + hostname ) );
550                    }
551                    catch ( MalformedURLException ex )
552                    {
553                        log.warn( "X-Forwarded-Host Header is malformed: {}", ex.getMessage() );
554                    }
555                }
556            }
557            return urls;
558        }
559    }

561    private int getPort( final URL url )
562    {
563        return url.getPort() > 0
564            ? url.getPort()
565            : ( "https".equals( url.getProtocol() ) ? DEFAULT_HTTPS : DEFAULT_HTTP );
566    }
567
568    private String getHost( final URL url )
569    {
570        return url.getHost().trim().toLowerCase();
571    }

573    /**
574     * Checks the validation headers. First the Origin header is checked, if this fails
575     * or is absent, the referer header is checked.
576     *
577     * @param info    The info object that must be populated with the targetURL
578     * @param request The HTTP request object
579     * @return A info object with updated status information
580     */
581    private HeaderValidationInfo checkSourceRequestHeader( final HeaderValidationInfo info,
582                                                           final HttpServletRequest request )
583    {
584        String origin = request.getHeader( ORIGIN );
585        if ( origin != null )
586        {
587            try
588            {
589                info.setOriginUrl( new URL( origin ) );
590            }
591            catch ( MalformedURLException e )
592            {
593                log.warn( "Bad origin header found: {}", origin );
594            }
595        }
596        // Check referer if Origin header dos not match or is not available
597        if ( info.getStatus() != info.OK )
598        {
599            String referer = request.getHeader( REFERER );
600            if ( referer != null )
601            {
602                try
603                {
604                    info.setRefererUrl( new URL( referer ) );
605                }
606                catch ( MalformedURLException ex )
607                {
608                    log.warn( "Bad URL in Referer HTTP-Header: {}, Message: {}", referer, ex.getMessage() );
609                }
610            }
611        }
612        return info;
613    }

620    private AuthenticationResult getAuthenticationResult( Message message, HttpServletRequest request )
621    {
622        AuthenticationResult authenticationResult = message.get( AuthenticationResult.class );
623
624        log.debug( "authenticationResult from message: {}", authenticationResult );
625        if ( authenticationResult == null )
626        {
627            try
628            {
629                authenticationResult =
630                    httpAuthenticator.getAuthenticationResult( request, getHttpServletResponse( message ) );
631
632                log.debug( "authenticationResult from request: {}", authenticationResult );
633            }
634            catch ( AuthenticationException e )
635            {
636                log.debug( "failed to authenticate for path {}", message.get( Message.REQUEST_URI ) );
637            }
638            catch ( AccountLockedException e )
639            {
640                log.debug( "account locked for path {}", message.get( Message.REQUEST_URI ) );
641            }
642            catch ( MustChangePasswordException e )
643            {
644                log.debug( "must change password for path {}", message.get( Message.REQUEST_URI ) );
645            }
646        }
647        return authenticationResult;
648    }
