021/**
022 * AbstractUserAssignment useful for common logic that implementors can use. 
023 *
024 * @author <a href="mailto:joakim@erdfelt.com">Joakim Erdfelt</a>
025 *
026 */
027public abstract class AbstractUserAssignment
028    implements UserAssignment
029{
030
031    public void addRoleName( Role role )
032    {
033        addRoleName( role.getName() );
034    }
035
036    @Override
037    public void addRoleId( Role role )
038    {
039        addRoleId( role.getId( ) );
040    }
041
042    public void addRoleName( String roleName )
043    {
044        List<String> names = getRoleNames();
045        if ( !names.contains( roleName ) )
046        {
047            names.add( roleName );
048        }
049        setRoleNames( names );
050    }
051
052    @Override
053    public void addRoleId( String roleId )
054    {
055        final List<String> ids  = getRoleIds( );
056        if (!ids.contains( roleId )) {
057            ids.add( roleId );
058        }
059        setRoleIds( ids );
060    }
061
062    public void removeRoleName( Role role )
063    {
064        removeRoleName( role.getName() );
065    }
066
067    public void removeRoleName( String roleName )
068    {
069        getRoleNames().remove( roleName );
070    }
071
072    @Override
073    public void removeRoleId( Role role )
074    {
075        removeRoleId( role.getId() );
076    }
077
078    @Override
079    public void removeRoleId( String roleId )
080    {
081        getRoleIds( ).remove( roleId );
082    }
083}
