022import org.apache.archiva.redback.users.User;
023
024import java.util.Comparator;
025
026/**
027 * UserSorter
028 */
029public class UserSorter
030    implements Comparator<User>
031{
032    private boolean ascending;
033
034    public UserSorter()
035    {
036        this.ascending = true;
037    }
038
039    public UserSorter( boolean ascending )
040    {
041        this.ascending = ascending;
042    }
043
044    public int compare( User o1, User o2 )
045    {
046        if ( ( o1 == null ) && ( o2 == null ) )
047        {
048            return 0;
049        }
050
051        if ( ( o1 == null ) && ( o2 != null ) )
052        {
053            return -1;
054        }
055
056        if ( ( o1 != null ) && ( o2 != null ) )
057        {
058            return 1;
059        }
060
061        User u1 = null;
062        User u2 = null;
063
064        if ( isAscending() )
065        {
066            u1 = o1;
067            u2 = o2;
068        }
069        else
070        {
071            u1 = o2;
072            u2 = o1;
073        }
074
075        return u1.getUsername().compareTo( u2.getUsername() );
076    }
077
078    public boolean isAscending()
079    {
080        return ascending;
081    }
082
083    public void setAscending( boolean ascending )
084    {
085        this.ascending = ascending;
086    }
087
088}
