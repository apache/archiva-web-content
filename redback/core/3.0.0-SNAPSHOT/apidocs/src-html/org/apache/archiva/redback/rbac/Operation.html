019import java.io.Serializable;
020
021/**
022 * Operation
023 *
024 * In RBAC the operation is an action or functionality that can be linked with a
025 * particular resource into an assignable Permission.  Operations don't exist outside
026 * Permissions.
027 *
028 * @author Jesse McConnell
029 * @author <a href="mailto:joakim@erdfelt.com">Joakim Erdfelt</a>
030 *
031 */
032public interface Operation extends Serializable
033{
034
035
036    /**
037     * Long description of an operation.
038     *
039     * @return String
040     */
041    String getDescription();
042
043    /**
044     * name of the operation that is used in the act of authorization
045     *
046     * 'modify-foo', 'change-password'
047     *
048     * NOTE: This field is considered the Primary Key for this object.
049     *
050     * @return the name of the operation.
051     */
052    String getName();
053
054    /**
055     * @param description
056     */
057    void setDescription( String description );
058
059    /**
060     * Set name of the operation that is used in the act of authorization
061     *
062     * 'modify-foo', 'change-password'
063     *
064     * NOTE: This field is considered the Primary Key for this object.
065     *
066     * @param name
067     */
068    void setName( String name );
069
070    /**
071     * Test to see if the object is a permanent object or not.
072     *
073     * @return true if the object is permanent.
074     */
075    boolean isPermanent();
076
077    /**
078     * Set flag indicating if the object is a permanent object or not.
079     *
080     * @param permanent true if the object is permanent.
081     */
082    void setPermanent( boolean permanent );
083}
