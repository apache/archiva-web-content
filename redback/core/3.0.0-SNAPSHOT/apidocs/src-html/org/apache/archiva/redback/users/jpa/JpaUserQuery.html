022import org.apache.archiva.redback.users.UserQuery;
023
024import java.util.Arrays;
025
026/**
027 * Created by martin on 23.09.16.
028 */
029public class JpaUserQuery implements UserQuery {
030
031    private String username;
032    private String email;
033    private String fullName;
034    private long firstResult=0;
035    private long maxResults=Integer.MAX_VALUE;
036    private boolean ascending=true;
037
038    private String orderBy="username";
039
040    @Override
041    public String getUsername() {
042        return username;
043    }
044
045    @Override
046    public void setUsername(String username) {
047        this.username = username;
048    }
049
050    public boolean hasUsername() {
051        return username != null && !"".equals(username);
052    }
053
054    @Override
055    public String getEmail() {
056        return email;
057    }
058
059    @Override
060    public void setEmail(String email) {
061        this.email = email;
062    }
063
064    public boolean hasEmail() {
065        return email!=null && !"".equals(email);
066    }
067
068    @Override
069    public String getFullName() {
070        return fullName;
071    }
072
073    @Override
074    public void setFullName(String fullName) {
075        this.fullName = fullName;
076    }
077
078    public boolean hasFullName() {
079        return fullName!=null && !"".equals(fullName);
080    }
081
082    @Override
083    public long getFirstResult() {
084        return firstResult;
085    }
086
087    public void setFirstResult(int firstResult) {
088        this.firstResult = firstResult;
089    }
090
091    @Override
092    public long getMaxResults() {
093        return maxResults;
094    }
095
096    public void setMaxResults(int maxResults) {
097        this.maxResults = maxResults;
098    }
099
100    @Override
101    public boolean isAscending() {
102        return ascending;
103    }
104
105    @Override
106    public void setAscending(boolean ascending) {
107        this.ascending = ascending;
108    }
109
110
111    @Override
112    public String getOrderBy() {
113        return orderBy;
114    }
115
116    @Override
117    public void setOrderBy(String orderBy) {
118        if (!UserQuery.ALLOWED_ORDER_FIELDS.contains(orderBy)) {
119            throw new IllegalArgumentException("Order attribute not allowed: "+orderBy);
120        }
121        this.orderBy = orderBy;
122    }
123}
