022import org.apache.archiva.redback.configuration.UserConfiguration;
023import org.apache.archiva.redback.configuration.UserConfigurationKeys;
024import org.slf4j.Logger;
025import org.slf4j.LoggerFactory;
026import org.springframework.stereotype.Service;
027
028import javax.annotation.PostConstruct;
029import javax.inject.Inject;
030import javax.inject.Named;
031import javax.naming.InvalidNameException;
032import javax.naming.ldap.LdapName;
033import javax.naming.ldap.Rdn;
034import javax.naming.spi.ObjectFactory;
035import javax.naming.spi.StateFactory;
036import java.util.Properties;
037
038/**
039 * @author <a href="mailto:trygvis@inamo.no">Trygve Laugst&oslash;l</a>
040 */
041@Service("ldapConnectionFactory#configurable")
042public class ConfigurableLdapConnectionFactory
043    implements LdapConnectionFactory
044{
045
046    private final Logger log = LoggerFactory.getLogger(ConfigurableLdapConnectionFactory.class);
047
048    private String hostname;
049
050    private int port;
051
052    private boolean ssl;
053
054    private String baseDn;
055
056    private String contextFactory;
057
058    private String bindDn;
059
060    private String password;
061
062    private String authenticationMethod;
063
064    private Properties extraProperties;
065
066    private LdapConnectionConfiguration ldapConnectionConfiguration;
067
068    private boolean valid = false;
069
070
071    @Inject
072    @Named(value = "userConfiguration#default")
073    private UserConfiguration userConf;
074
075    // ----------------------------------------------------------------------
076    // Component Lifecycle
077    // ----------------------------------------------------------------------
078    @PostConstruct
079    public void initialize()
080    {
081        try
082        {
083            ldapConnectionConfiguration = new LdapConnectionConfiguration();
084            ldapConnectionConfiguration.setHostname(
085                userConf.getString( UserConfigurationKeys.LDAP_HOSTNAME, hostname ) );
086            ldapConnectionConfiguration.setPort( userConf.getInt( UserConfigurationKeys.LDAP_PORT, port ) );
087            ldapConnectionConfiguration.setSsl( userConf.getBoolean( UserConfigurationKeys.LDAP_SSL, ssl ) );
088            ldapConnectionConfiguration.setBaseDn(
089                userConf.getConcatenatedList( UserConfigurationKeys.LDAP_BASEDN, baseDn ) );
090            ldapConnectionConfiguration.setContextFactory(
091                userConf.getString( UserConfigurationKeys.LDAP_CONTEX_FACTORY, contextFactory ) );
092            ldapConnectionConfiguration.setBindDn(
093                userConf.getConcatenatedList( UserConfigurationKeys.LDAP_BINDDN, bindDn ) );
094            ldapConnectionConfiguration.setPassword(
095                userConf.getString( UserConfigurationKeys.LDAP_PASSWORD, password ) );
096            ldapConnectionConfiguration.setAuthenticationMethod(
097                userConf.getString( UserConfigurationKeys.LDAP_AUTHENTICATION_METHOD, authenticationMethod ) );
098            ldapConnectionConfiguration.setExtraProperties( extraProperties );
099            valid = true;
100        }
101        catch ( InvalidNameException e )
102        {
103            log.error("Error during initialization of LdapConnectionFactory "+e.getMessage(),e);
104        }
105    }
106
107    // ----------------------------------------------------------------------
108    // LdapConnectionFactory Implementation
109    // ----------------------------------------------------------------------
110
111    public DefaultLdapConnection getConnection()
112        throws LdapException
113    {
114        return new DefaultLdapConnection( getLdapConnectionConfiguration(), null );
115    }
116
117    public DefaultLdapConnection getConnection( Rdn subRdn )
118        throws LdapException
119    {
120        return new DefaultLdapConnection( getLdapConnectionConfiguration(), subRdn );
121    }
122
123    public DefaultLdapConnection getConnection( String bindDn, String password )
124        throws LdapException
125    {
126        return new DefaultLdapConnection( getLdapConnectionConfiguration(), bindDn, password );
127    }
128
129    public DefaultLdapConnection getConnection( LdapConnectionConfiguration ldapConnectionConfiguration )
130        throws LdapException
131    {
132        return new DefaultLdapConnection( ldapConnectionConfiguration, null );
133    }
134
135    public LdapName getBaseDnLdapName()
136        throws LdapException
137    {
138        try
139        {
140            return new LdapName( baseDn );
141        }
142        catch ( InvalidNameException e )
143        {
144            throw new LdapException( "The base DN is not a valid name.", e );
145        }
146    }
147
148    public void addObjectFactory( Class<? extends ObjectFactory> objectFactoryClass )
149    {
150        getLdapConnectionConfiguration().getObjectFactories().add( objectFactoryClass );
151    }
152
153    public void addStateFactory( Class<? extends StateFactory> stateFactoryClass )
154    {
155        getLdapConnectionConfiguration().getStateFactories().add( stateFactoryClass );
156    }
157
158    // ----------------------------------------------------------------------
159    //
160    // ----------------------------------------------------------------------
161
162    public String toString()
163    {
164        return "{ConfigurableLdapConnectionFactory: configuration: " + getLdapConnectionConfiguration() + "}";
165    }
166
167    public LdapConnectionConfiguration getLdapConnectionConfiguration()
168    {
169        return ldapConnectionConfiguration;
170    }
171
172    public void setLdapConnectionConfiguration( LdapConnectionConfiguration ldapConnectionConfiguration )
173    {
174        this.ldapConnectionConfiguration = ldapConnectionConfiguration;
175    }
176
177    public String getHostname()
178    {
179        return hostname;
180    }
181
182    public void setHostname( String hostname )
183    {
184        this.hostname = hostname;
185    }
186
187    public int getPort()
188    {
189        return port;
190    }
191
192    public void setPort( int port )
193    {
194        this.port = port;
195    }
196
197    public boolean isSsl()
198    {
199        return ssl;
200    }
201
202    public void setSsl( boolean ssl )
203    {
204        this.ssl = ssl;
205    }
206
207    public String getBaseDn()
208    {
209        return baseDn;
210    }
211
212    public void setBaseDn( String baseDn )
213    {
214        this.baseDn = baseDn;
215    }
216
217    public String getContextFactory()
218    {
219        return contextFactory;
220    }
221
222    public void setContextFactory( String contextFactory )
223    {
224        this.contextFactory = contextFactory;
225    }
226
227    public String getBindDn()
228    {
229        return bindDn;
230    }
231
232    public void setBindDn( String bindDn )
233    {
234        this.bindDn = bindDn;
235    }
236
237    public String getPassword()
238    {
239        return password;
240    }
241
242    public void setPassword( String password )
243    {
244        this.password = password;
245    }
246
247    public String getAuthenticationMethod()
248    {
249        return authenticationMethod;
250    }
251
252    public void setAuthenticationMethod( String authenticationMethod )
253    {
254        this.authenticationMethod = authenticationMethod;
255    }
256
257    public Properties getExtraProperties()
258    {
259        return extraProperties;
260    }
261
262    public void setExtraProperties( Properties extraProperties )
263    {
264        this.extraProperties = extraProperties;
265    }
266
267    public UserConfiguration getUserConf()
268    {
269        return userConf;
270    }
271
272    public void setUserConf( UserConfiguration userConf )
273    {
274        this.userConf = userConf;
275    }
276
277    public boolean isValid() {
278        return valid;
279    }
280}
