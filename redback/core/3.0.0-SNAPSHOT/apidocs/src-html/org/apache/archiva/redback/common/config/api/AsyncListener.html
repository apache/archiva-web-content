022import java.lang.annotation.Documented;
023import java.lang.annotation.Retention;
024import java.lang.annotation.Target;
025
026import static java.lang.annotation.ElementType.METHOD;
027import static java.lang.annotation.ElementType.TYPE;
028import static java.lang.annotation.RetentionPolicy.RUNTIME;
029
030/**
031 *
032 * A annotation that marks a event listener method as asynchronous. That means the listener event methods
033 * are run in a separated thread. How tasks are executed is dependent on the implementation.
034 *
035 * @author Martin Stockhammer <martin_s@apache.org>
036 * @since 3.0
037 */
038
039@Target(value={METHOD, TYPE})
040@Retention(value=RUNTIME)
041@Documented
042public @interface AsyncListener
043{
044    /**
045     * May be set to set the executor. The meaning of this value is implementation specific.
046     * @return The value.
047     */
048    String value() default "";
049}
