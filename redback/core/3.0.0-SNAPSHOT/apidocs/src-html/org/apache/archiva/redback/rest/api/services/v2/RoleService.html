370    /**
371     * This returns the list of assigned users to a given role. The flag "recurse" is a query parameter.
372     * If the query parameter exists and is not set to 'false', or '0', it will recurse all parent roles and return a list of users
373     * assigned to the current role and the parent roles up to the root.
374     * If the query parameter does not exist or is set to 'false' or '0', it will return only the users assigned directly
375     * to the given role.
376     *
377     * @param roleId the role identifier, for which the assigned users are returned
378     * @param recurse if the parameter does not exist or is set to 'false' or '0', only directly assigned users are returned.
379     *                If the parameter value is set to 'parentsOnly', the users assigned to all parent roles up to the root excluding the
380     *                given role are returned.
381     *                Otherwise all users assigned to the given role and all parent roles up to the root are returned.
382     * @param searchTerm the substring query term to search for in the user ids and names
383     * @param offset the offset index in the user list for paging
384     * @param limit the maximum number of users returned
385     * @param orderBy the order attributes for ordering
386     * @param order the order direction 'asc' (ascending), or 'desc' (descending)
387     * @return the list of user objects
388     * @throws RedbackServiceException
389     */
390    @Path("{roleId}/user")
391    @GET
392    @Produces({APPLICATION_JSON})
393    @RedbackAuthorization(permissions = RedbackRoleConstants.USER_MANAGEMENT_RBAC_ADMIN_OPERATION)
394    @Operation( summary = "Returns the users assigned to the given role",
395        parameters = {
396            @Parameter(name = "q", description = "Search term"),
397            @Parameter(name = "offset", description = "The offset of the first element returned"),
398            @Parameter(name = "limit", description = "Maximum number of items to return in the response"),
399            @Parameter(name = "orderBy", description = "List of attribute used for sorting (user_id, fullName, email, created"),
400            @Parameter(name = "order", description = "The sort order. Either ascending (asc) or descending (desc)"),
401            @Parameter(name = "recurse", description = "If not present, or set to 'false' or '0', only users assigned directly to this role are returned."+
402            " If present and set to 'parentsOnly', the list of users assigned to all parents of the given role up to the root."+
403                " If present and set to any other value than 'parentsOnly', 'false' or '0', the users assigned to this role or any parent role in the hierarchy"+
404                " up to the root are returned.")
405        },
406        security = {
407            @SecurityRequirement( name = RedbackRoleConstants.USER_MANAGEMENT_RBAC_ADMIN_OPERATION )
408        },
409        responses = {
410            @ApiResponse( responseCode = "200",
411                description = "If the users could be retrieved"
412            ),
413            @ApiResponse( responseCode = "404", description = "Role instance does not exist",
414                content = @Content(mediaType = APPLICATION_JSON, schema = @Schema(implementation = RedbackRestError.class )) ),
415            @ApiResponse( responseCode = "403", description = "The authenticated user has not the permission for role assignment.",
416                content = @Content(mediaType = APPLICATION_JSON, schema = @Schema(implementation = RedbackRestError.class )) )
417        }
418    )
419    PagedResult<UserInfo> getRoleUsers(@PathParam( "roleId" ) String roleId,
420                                       @QueryParam("recurse") String recurse,
421                                       @QueryParam("q") @DefaultValue( "" ) String searchTerm,
422                                       @QueryParam( "offset" ) @DefaultValue( "0" ) Integer offset,
423                                       @QueryParam( "limit" ) @DefaultValue( value = DEFAULT_PAGE_LIMIT ) Integer limit,
424                                       @QueryParam( "orderBy") @DefaultValue( "id" ) List<String> orderBy,
425                                       @QueryParam("order") @DefaultValue( "asc" ) String order
426                                       ) throws RedbackServiceException;
