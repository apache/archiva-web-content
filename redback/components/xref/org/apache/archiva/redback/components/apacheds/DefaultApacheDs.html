91      public InitialDirContext getAdminContext()
92          throws NamingException
93      {
94          assertIsStarted();
95  
96          Hashtable environment = new Hashtable( configuration.toJndiEnvironment() );
97          environment.put( Context.INITIAL_CONTEXT_FACTORY, ServerContextFactory.class.getName() );
98          environment.put( Context.SECURITY_PRINCIPAL, "uid=admin,ou=system" );
99          environment.put( Context.SECURITY_CREDENTIALS, password );
100         environment.put( Context.SECURITY_AUTHENTICATION, "simple" );
101 //        environment.put( Context.PROVIDER_URL, "dc=hauskeeper,dc=codehaus,dc=org" );
102         return new InitialDirContext( environment );
103     }

130     public void addPartition( Partition partition )
131         throws NamingException
132     {
133         MutablePartitionConfiguration configuration = new MutablePartitionConfiguration();
134 
135         configuration.setId( partition.getName() );
136         configuration.setSuffix( partition.getSuffix() );
137         configuration.setIndexedAttributes( partition.getIndexedAttributes() );
138         configuration.setContextEntry( partition.getContextAttributes() );
139         //configuration.setSynchOnWrite( true );
140         configuration.setCacheSize( 1 );
141         //configuration.setOptimizerEnabled( false );
142         partitionConfigurations.add( configuration );
143     }

145     public Partition addSimplePartition( String name, String[] domainComponents )
146         throws NamingException
147     {
148         if ( domainComponents.length == 0 )
149         {
150             throw new NamingException( "Illegal argument, there has to be at least one domain component." );
151         }
152 
153         StringBuilder suffix = new StringBuilder();
154 
155         for ( int i = 0; i < domainComponents.length; i++ )
156         {
157             String dc = domainComponents[i];
158 
159             suffix.append( "dc=" ).append( dc );
160 
161             if ( i != domainComponents.length - 1 )
162             {
163                 suffix.append( "," );
164             }
165         }
166 
167         // ----------------------------------------------------------------------
168         // The root entry of the partition
169         // ----------------------------------------------------------------------
170 
171         Attributes attributes = new BasicAttributes( true );
172         attributes.put( "dc", domainComponents[0] );
173         Attribute objectClass = new BasicAttribute( "objectClass" );
174         objectClass.add( "top" );
175         objectClass.add( "domain" );
176         objectClass.add( "extensibleObject" );
177         attributes.put( objectClass );
178 
179         Partition partition = new Partition();
180         partition.setName( name );
181         partition.setSuffix( suffix.toString() );
182         partition.setContextAttributes( attributes );
183         HashSet set = new HashSet();
184         set.add( "uid" );
185         set.add( "cn" );
186         partition.setIndexedAttributes( set );
187 
188         addPartition( partition );
189 
190         return partition;
191     }

193     public void startServer()
194         throws Exception
195     {
196         logger.info( "Starting Apache Directory Server server." );
197 
198         logger.info( "ApacheDS basedir: {}", basedir.getAbsolutePath() );
199 
200         File logs = new File( basedir, "logs" );
201 
202         if ( !logs.exists() && !logs.mkdirs() )
203         {
204             throw new Exception( "Could not create logs directory: " + logs.getAbsolutePath() );
205         }
206 
207         Properties environment = new Properties();
208         environment.setProperty( "java.naming.security.authentication", "simple" );
209         environment.setProperty( "java.naming.security.principal", "uid=admin,ou=system" );
210         if ( password != null )
211         {
212             environment.setProperty( "java.naming.security.credentials", password );
213         }
214         MutableServerStartupConfiguration configuration = new MutableServerStartupConfiguration();
215         configuration.setWorkingDirectory( basedir );
216         configuration.setAllowAnonymousAccess( true );
217         //configuration.setEnableNtp( false );
218         //configuration.setEnableKerberos( false );
219         //configuration.setEnableChangePassword( false );
220         LdapConfiguration config = new LdapConfiguration();
221         config.setIpPort( port );
222         configuration.setLdapConfiguration( config );
223         configuration.setEnableNetworking( enableNetworking );
224         configuration.setSynchPeriodMillis( 100 );
225 
226         if ( configuration.getPartitionConfigurations() == null || ( configuration.getPartitionConfigurations() != null
227             && configuration.getPartitionConfigurations().isEmpty() ) )
228         {
229             configuration.setPartitionConfigurations( partitionConfigurations );
230         }
231         Properties env = new Properties();
232         env.setProperty( Context.SECURITY_PRINCIPAL, "uid=admin,ou=system" );
233         if ( password != null )
234         {
235             env.setProperty( Context.SECURITY_CREDENTIALS, password );
236         }
237         env.setProperty( Context.SECURITY_AUTHENTICATION, "simple" );
238         env.setProperty( Context.PROVIDER_URL, "ou=system" );
239         env.setProperty( Context.INITIAL_CONTEXT_FACTORY, ServerContextFactory.class.getName() );
240         env.putAll( configuration.toJndiEnvironment() );
241         InitialDirContext context = new InitialDirContext( env );
242 
243         //Attributes inetAttributes = context.getAttributes( "cn=inetorgperson,ou=schema" );
244 
245         //inetAttributes.remove( "m-disabled" );
246 
247         this.configuration = configuration;
248 
249         logger.info( "Started Apache Directory Server server." );
250 
251         stopped = false;
252     }

254     public void stopServer()
255         throws Exception
256     {
257         if ( stopped )
258         {
259             throw new Exception( "Already stopped." );
260         }
261 
262         logger.info( "Stopping Apache Directory Server server." );
263 
264         sync();
265 
266         stopped = true;
267 
268         Hashtable env = new Hashtable();
269         env.putAll( new ShutdownConfiguration().toJndiEnvironment() );
270         new InitialDirContext( env );
271 
272         logger.info( "Apache Directory Server server stopped." );
273     }

280     public void sync()
281         throws Exception
282     {
283         logger.info( "Sync'ing Apache Directory Server server." );
284 
285         Hashtable env = new Hashtable();
286         env.putAll( new SyncConfiguration().toJndiEnvironment() );
287         new InitialDirContext( env );
288     }

291     public void stop()
292     {
293         try
294         {
295             if ( !stopped )
296             {
297                 stopServer();
298             }
299         }
300         catch ( Exception e )
301         {
302             throw new RuntimeException( "Error while stopping Apache Directory Server server.", e );
303         }
304     }
