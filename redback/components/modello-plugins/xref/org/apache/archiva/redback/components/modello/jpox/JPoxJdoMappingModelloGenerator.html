125     public void generate( Model model, Properties properties ) throws ModelloException
126     {
127         initialize( model, properties );
128 
129         // ----------------------------------------------------------------------
130         // Generate the JDO files
131         // ----------------------------------------------------------------------
132 
133         try
134         {
135             String fileName = properties.getProperty( ModelloParameterConstants.FILENAME, "package.jdo" );
136 
137             JPoxModelMetadata metadata = (JPoxModelMetadata) model.getMetadata( JPoxModelMetadata.ID );
138             File packageJdo = null;
139 
140             if ( metadata.isMappingInPackage() )
141             {
142                 // Use package name.
143                 String packageName = model.getDefaultPackageName( isPackageWithVersion(), getGeneratedVersion() );
144                 String dir = StringUtils.replace( packageName, '.', '/' );
145                 File directory = new File( getOutputDirectory(), dir );
146                 packageJdo = new File( directory, fileName );
147             }
148             else
149             {
150                 // Use the specified location.
151                 File directory = getOutputDirectory();
152                 packageJdo = new File( directory, fileName );
153             }
154 
155             File parent = packageJdo.getParentFile();
156 
157             if ( !parent.exists() )
158             {
159                 if ( !parent.mkdirs() )
160                 {
161                     throw new ModelloException( "Error while creating parent directories for the file " + "'"
162                                     + packageJdo.getAbsolutePath() + "'." );
163                 }
164             }
165 
166             generatePackageJdo( packageJdo, model );
167         }
168         catch ( IOException e )
169         {
170             throw new ModelloException( "Error while writing package.jdo.", e );
171         }
172     }

178     private void generatePackageJdo( File file, Model model ) throws IOException, ModelloException
179     {
180         OutputStreamWriter fileWriter = new OutputStreamWriter( new FileOutputStream( file ), "UTF-8" );
181 
182         PrintWriter printWriter = new PrintWriter( fileWriter );
183 
184         XMLWriter writer = new PrettyPrintXMLWriter( printWriter );
185 
186         Map classes = new HashMap();
187 
188         for ( Iterator it = model.getClasses( getGeneratedVersion() ).iterator(); it.hasNext(); )
189         {
190             ModelClass modelClass = (ModelClass) it.next();
191 
192             JPoxClassMetadata jpoxMetadata = (JPoxClassMetadata) modelClass.getMetadata( JPoxClassMetadata.ID );
193 
194             if ( !jpoxMetadata.isEnabled() )
195             {
196                 // Skip generation of those classes that are not enabled for the jpox plugin.
197                 continue;
198             }
199 
200             String packageName = modelClass.getPackageName( isPackageWithVersion(), getGeneratedVersion() );
201 
202             List list = (List) classes.get( packageName );
203 
204             if ( list == null )
205             {
206                 list = new ArrayList();
207             }
208 
209             list.add( modelClass );
210 
211             classes.put( packageName, list );
212         }
213 
214         printWriter.println( "<?xml version=\"1.0\" encoding=\"UTF-8\"?>" );
215         printWriter.println();
216         printWriter.println( "<!DOCTYPE jdo PUBLIC" );
217         printWriter.println( "  \"-//Sun Microsystems, Inc.//DTD Java Data Objects Metadata 2.0//EN\"" );
218         printWriter.println( "  \"http://java.sun.com/dtd/jdo_2_0.dtd\">" );
219         printWriter.println();
220 
221         writer.startElement( "jdo" );
222 
223         for ( Iterator it = classes.entrySet().iterator(); it.hasNext(); )
224         {
225             Map.Entry entry = (Map.Entry) it.next();
226 
227             List list = (List) entry.getValue();
228 
229             if ( list.size() == 0 )
230             {
231                 continue;
232             }
233 
234             String packageName = (String) entry.getKey();
235 
236             writer.startElement( "package" );
237 
238             writer.addAttribute( "name", packageName );
239 
240             for ( Iterator it2 = list.iterator(); it2.hasNext(); )
241             {
242                 ModelClass modelClass = (ModelClass) it2.next();
243 
244                 writeClass( writer, modelClass );
245             }
246 
247             if ( packageName.equals( model.getDefaultPackageName( isPackageWithVersion(), getGeneratedVersion() ) ) )
248             {
249                 writeModelloMetadataClass( writer );
250             }
251 
252             writer.endElement(); // package
253         }
254 
255         writer.endElement(); // jdo
256 
257         printWriter.println();
258 
259         printWriter.close();
260     }

262     private void writeClass( XMLWriter writer, ModelClass modelClass ) throws ModelloException
263     {
264         JPoxClassMetadata jpoxMetadata = (JPoxClassMetadata) modelClass.getMetadata( JPoxClassMetadata.ID );
265 
266         if ( !jpoxMetadata.isEnabled() )
267         {
268             // Skip generation of those classes that are not enabled for the jpox plugin.
269             return;
270         }
271 
272         writer.startElement( "class" );
273 
274         writer.addAttribute( "name", modelClass.getName() );
275 
276         ModelClass persistenceCapableSuperclass = null;
277 
278         if ( modelClass.hasSuperClass() && modelClass.isInternalSuperClass() )
279         {
280             persistenceCapableSuperclass = getModel().getClass( modelClass.getSuperClass(), getGeneratedVersion() );
281         }
282 
283         if ( persistenceCapableSuperclass != null )
284         {
285             String superPackageName =
286                 persistenceCapableSuperclass.getPackageName( isPackageWithVersion(), getGeneratedVersion() );
287 
288             writer.addAttribute( "persistence-capable-superclass", superPackageName + "."
289                             + persistenceCapableSuperclass.getName() );
290         }
291 
292         writer.addAttribute( "detachable", String.valueOf( jpoxMetadata.isDetachable() ) );
293 
294         writer.addAttribute( "table", getTableName( modelClass, jpoxMetadata ) );
295 
296         // ----------------------------------------------------------------------
297         // If this class has a primary key field mark make jpox manage the id
298         // as a autoincrement variable
299         // ----------------------------------------------------------------------
300 
301         List fields = Collections.unmodifiableList( modelClass.getFields( getGeneratedVersion() ) );
302 
303         // TODO: for now, assume that any primary key will be set in the super class
304         // While it should be possible to have abstract super classes and have the
305         // key defined in the sub class this is not implemented yet.
306 
307         boolean needInheritance = false;
308 
309         if ( persistenceCapableSuperclass == null )
310         {
311             if ( StringUtils.isNotEmpty( jpoxMetadata.getIdentityType() ) )
312             {
313                 String identityType = jpoxMetadata.getIdentityType();
314                 if ( !IDENTITY_TYPES.contains( identityType ) )
315                 {
316                     throw new ModelloException( "The JDO mapping generator does not support the specified "
317                                     + "class identity type '" + identityType + "'. " + "Supported types: "
318                                     + IDENTITY_TYPES );
319                 }
320                 writer.addAttribute( "identity-type", identityType );
321             }
322             else if ( isInstantionApplicationType( modelClass ) )
323             {
324                 writer.addAttribute( "identity-type", "application" );
325             }
326         }
327         else
328         {
329             needInheritance = true;
330         }
331 
332         if ( objectIdClassOverride != null )
333         {
334             if ( StringUtils.isNotEmpty( objectIdClassOverride ) )
335             {
336                 writer.addAttribute( "objectid-class", jpoxMetadata.getIdentityClass() );
337             }
338         }
339         else if ( StringUtils.isNotEmpty( jpoxMetadata.getIdentityClass() ) )
340         {
341             // Use user provided objectId class.
342             writer.addAttribute( "objectid-class", jpoxMetadata.getIdentityClass() );
343         }
344         else
345         {
346             // Calculate the objectId class.
347 
348             List primaryKeys = getPrimaryKeyFields( modelClass );
349 
350             // TODO: write generation support for multi-primary-key support.
351             //       would likely need to write a java class that can supply an ObjectIdentity
352             //       to the jpox/jdo implementation.
353             if ( primaryKeys.size() > 1 )
354             {
355                 throw new ModelloException(
356                                             "The JDO mapping generator does not yet support Object Identifier generation "
357                                                             + "for the " + primaryKeys.size()
358                                                             + " fields specified as <identifier> or "
359                                                             + "with jpox.primary-key=\"true\"" );
360             }
361 
362             if ( primaryKeys.size() == 1 )
363             {
364                 ModelField modelField = (ModelField) primaryKeys.get( 0 );
365                 String objectIdClass = (String) PRIMITIVE_IDENTITY_MAP.get( modelField.getType() );
366 
367                 if ( StringUtils.isNotEmpty( objectIdClass ) )
368                 {
369                     writer.addAttribute( "objectid-class", objectIdClass );
370                 }
371             }
372         }
373 
374         if ( needInheritance )
375         {
376             writer.startElement( "inheritance" );
377 
378             // TODO: The table strategy should be customizable
379             // http://www.jpox.org/docs/1_1/inheritance.html - in particular
380             // the strategy="subclass-table" and strategy="new-table" parts
381 
382             writer.addAttribute( "strategy", "new-table" );
383 
384             writer.endElement();
385         }
386 
387         // ----------------------------------------------------------------------
388         // Write all fields
389         // ----------------------------------------------------------------------
390 
391         for ( Iterator it = fields.iterator(); it.hasNext(); )
392         {
393             ModelField modelField = (ModelField) it.next();
394 
395             writeModelField( writer, modelField );
396         }
397 
398         // Write ignored fields.
399         List ignoredFields = jpoxMetadata.getNotPersisted();
400         if ( ignoredFields != null )
401         {
402             Iterator it = ignoredFields.iterator();
403             while ( it.hasNext() )
404             {
405                 String fieldName = (String) it.next();
406                 writer.startElement( "field" );
407                 writer.addAttribute( "name", fieldName );
408                 writer.addAttribute( "persistence-modifier", "none" );
409                 writer.endElement();
410             }
411         }
412 
413         // ----------------------------------------------------------------------
414         // Write out the "detailed" fetch group. This group will by default
415         // contain all fields in a object. The default fetch group will contain
416         // all the primitives in a class as by JDO defaults.
417         // ----------------------------------------------------------------------
418 
419         List detailedFields = new ArrayList();
420 
421         for ( Iterator it = fields.iterator(); it.hasNext(); )
422         {
423             ModelField field = (ModelField) it.next();
424 
425             if ( field.isPrimitive() )
426             {
427                 continue;
428             }
429 
430             if ( field instanceof ModelAssociation )
431             {
432                 StoreAssociationMetadata storeMetadata = getAssociationMetadata( (ModelAssociation) field );
433 
434                 if ( storeMetadata.isPart() != null && storeMetadata.isPart().booleanValue() )
435                 {
436                     continue;
437                 }
438             }
439 
440             detailedFields.add( field );
441         }
442 
443         // ----------------------------------------------------------------------
444         // Write all fetch groups
445         // ----------------------------------------------------------------------
446 
447         // Write defaut detail fetch group
448         writeFetchGroup( writer, modelClass.getName() + "_detail", detailedFields, true );
449 
450         // Write user fetch groups
451         Map fetchsMap = new HashMap();
452 
453         for ( Iterator it = fields.iterator(); it.hasNext(); )
454         {
455             ModelField field = (ModelField) it.next();
456 
457             JPoxFieldMetadata jpoxFieldMetadata = (JPoxFieldMetadata) field.getMetadata( JPoxFieldMetadata.ID );
458 
459             List names = jpoxFieldMetadata.getFetchGroupNames();
460 
461             if ( names != null )
462             {
463                 for ( Iterator i = names.iterator(); i.hasNext(); )
464                 {
465                     String fetchGroupName = (String) i.next();
466 
467                     List fetchList;
468 
469                     if ( fetchsMap.get( fetchGroupName ) == null )
470                     {
471                         fetchList = new ArrayList();
472                     }
473                     else
474                     {
475                         fetchList = (List) fetchsMap.get( fetchGroupName );
476                     }
477 
478                     fetchList.add( field );
479 
480                     fetchsMap.put( fetchGroupName, fetchList );
481                 }
482             }
483         }
484 
485         for ( Iterator it = fetchsMap.keySet().iterator(); it.hasNext(); )
486         {
487             String fetchName = (String) it.next();
488 
489             writeFetchGroup( writer, fetchName, (List) fetchsMap.get( fetchName ), false );
490         }
491 
492         writer.endElement(); // class
493     }

495     /**
496      * Utility method to obtain the table name from the {@link ModelClass} or {@link JPoxClassMetadata} taking into
497      * account the possibility of prefixes in use at the {@link JPoxModelMetadata} level.
498      * 
499      * @param modelClass the model class to get the class.name from.
500      * @param classMetadata the class metadata to get the class-metadata.table name from.
501      * @return the table name (with possible prefix applied) 
502      * @throws ModelloException if there was a problem with the table name violating a sql reserved word.
503      */
504     private String getTableName( ModelClass modelClass, JPoxClassMetadata classMetadata ) throws ModelloException
505     {
506         JPoxModelMetadata modelMetadata = (JPoxModelMetadata) modelClass.getModel().getMetadata( JPoxModelMetadata.ID );
507 
508         boolean hasPrefix = StringUtils.isNotEmpty( modelMetadata.getTablePrefix() );
509         boolean hasAlternateName = StringUtils.isNotEmpty( classMetadata.getTable() );
510 
511         String prefix = "";
512 
513         if ( hasPrefix )
514         {
515             prefix = modelMetadata.getTablePrefix().trim();
516         }
517 
518         String tableName = null;
519 
520         if ( hasAlternateName )
521         {
522             tableName = prefix + classMetadata.getTable();
523         }
524         else
525         {
526             tableName = prefix + modelClass.getName();
527         }
528 
529         if ( sqlReservedWords.isKeyword( tableName ) )
530         {
531             StringBuffer emsg = new StringBuffer();
532 
533             /* ----------------------------------------------------------------
534              *   SQL Reserved Word Violation: 'ROLES'
535              *   Context: TABLE NAME
536              */
537             emsg.append( EOL ).append( ERROR_LINE ).append( EOL );
538             emsg.append( "  SQL Reserved Word Violation: " ).append( tableName ).append( EOL );
539             emsg.append( "  Context: TABLE NAME" ).append( EOL );
540             emsg.append( " " ).append( EOL );
541 
542             /*   In Model:
543              *     <model jpox.table-prefix="">
544              *       <class jpox.table="">
545              *         <name>JdoRole</name>
546              *       </class>
547              *     </model>
548              */
549             emsg.append( "  In Model:" ).append( EOL );
550             emsg.append( "    <model" );
551             if ( hasPrefix )
552             {
553                 emsg.append( " jpox.table-prefix=\"" ).append( modelMetadata.getTablePrefix() ).append( "\"" );
554             }
555             emsg.append( ">" ).append( EOL );
556             emsg.append( "      <class" );
557             if ( hasAlternateName )
558             {
559                 emsg.append( " jpox.table=\"" ).append( classMetadata.getTable() ).append( "\"" );
560             }
561             emsg.append( ">" ).append( EOL );
562             emsg.append( "        <name>" ).append( modelClass.getName() ).append( "</name>" ).append( EOL );
563             emsg.append( "      </class>" ).append( EOL );
564             emsg.append( "    </model>" ).append( EOL );
565             emsg.append( " " ).append( EOL );
566 
567             /*   Violation Source(s): Oracle (WARNING)
568              *                        SQL 99 (ERROR)
569              *                        
570              *   Severity: ERROR - You must change this name for maximum
571              *             compatibility amoungst JDBC SQL Servers.
572              *             
573              *   Severity: WARNING - You are encouraged to change this name
574              *             for maximum compatibility amoungst JDBC SQL Servers.
575              */
576             boolean hasError = appendKeywordSourceViolations( tableName, emsg );
577 
578             /*   Suggestions: 1) Use a different prefix in
579              *                   <model jpox.table-prefix="DIFFERENT_">
580              *                2) Use a different alternate table name using
581              *                   <class jpox.table="DIFFERENT">
582              *                3) Use a different class name in
583              *                   <class>
584              *                     <name>DIFFERENT</name>
585              *                   </class>
586              * ----------------------------------------------------------------                  
587              */
588             emsg.append( "  Suggestions: 1) Use a different prefix in" ).append( EOL );
589             emsg.append( "                  <model jpox.table-prefix=\"DIFFERENT_\">" ).append( EOL );
590             emsg.append( "               2) Use a different alternate table name using" ).append( EOL );
591             emsg.append( "                  <class jpox.table=\"DIFFERENT\">" ).append( EOL );
592             emsg.append( "               3) Use a different class name in" ).append( EOL );
593             emsg.append( "                  <class>" ).append( EOL );
594             emsg.append( "                    <name>DIFFERENT</name>" ).append( EOL );
595             emsg.append( "                  </class>" ).append( EOL );
596 
597             emsg.append( ERROR_LINE );
598 
599             // Determine possible exception.
600             if ( hasError || modelMetadata.getReservedWordStrictness().equals( JPoxModelMetadata.WARNING ) )
601             {
602                 throw new ModelloException( emsg.toString() );
603             }
604 
605             // No exception. use it. But log it.
606             getLogger().warn( emsg.toString() );
607         }
608 
609         return tableName;
610     }

612     /**
613      * Support method for the {@link #getTableName(ModelClass, JPoxClassMetadata)}, 
614      * {@link #getColumnName(ModelField, JPoxFieldMetadata)}, and 
615      * {@link #getJoinTableName(ModelField, JPoxFieldMetadata)} reserved word tests.
616      * 
617      * @param word the word in violation.
618      * @param emsg the string buffer to append to.
619      * @return if this word has any ERROR severity level violations.
620      */
621     private boolean appendKeywordSourceViolations( String word, StringBuffer emsg )
622     {
623         List sources = sqlReservedWords.getKeywordSourceList( word );
624         boolean hasError = false;
625         emsg.append( "  Violation Source(s): " );
626         for ( Iterator it = sources.iterator(); it.hasNext(); )
627         {
628             KeywordSource source = (KeywordSource) it.next();
629             emsg.append( source.getName() ).append( " (" ).append( source.getSeverity() ).append( ")" );
630             emsg.append( EOL );
631 
632             if ( source.getSeverity().equalsIgnoreCase( "ERROR" ) )
633             {
634                 hasError = true;
635             }
636 
637             if ( it.hasNext() )
638             {
639                 emsg.append( "                       " );
640             }
641         }
642         emsg.append( " " ).append( EOL );
643 
644         emsg.append( "  Severity: " );
645         if ( hasError )
646         {
647             emsg.append( "ERROR - You must change this name for the maximum" ).append( EOL );
648             emsg.append( "            compatibility amoungst JDBC SQL Servers." ).append( EOL );
649         }
650         else
651         {
652             emsg.append( "WARNING - You are encouraged to change this name" ).append( EOL );
653             emsg.append( "            for maximum compatibility amoungst JDBC SQL Servers." ).append( EOL );
654         }
655         emsg.append( " " ).append( EOL );
656 
657         return hasError;
658     }

660     private void writeModelloMetadataClass( XMLWriter writer ) throws ModelloException
661     {
662         writer.startElement( "class" );
663 
664         writer.addAttribute( "name", getModel().getName() + "ModelloMetadata" );
665 
666         writer.addAttribute( "detachable", String.valueOf( true ) );
667 
668         writer.startElement( "field" );
669 
670         writer.addAttribute( "name", "modelVersion" );
671         writer.addAttribute( "null-value", "default" );
672 
673         writer.startElement( "column" );
674 
675         writer.addAttribute( "default-value", getGeneratedVersion().toString() );
676 
677         writer.endElement(); // column
678 
679         writer.endElement(); // field
