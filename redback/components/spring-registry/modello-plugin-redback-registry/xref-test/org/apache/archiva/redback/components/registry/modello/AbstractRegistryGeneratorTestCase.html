37  public abstract class AbstractRegistryGeneratorTestCase

45      protected void prepareTest( String outputType )
46          throws ComponentLookupException, ModelloException, ModelValidationException, IOException, CompilerException
47      {
48          ModelloCore modello = (ModelloCore) container.lookup( ModelloCore.ROLE );
49  
50          Model model = modello.loadModel( ReaderFactory.newXmlReader( getTestFile( "src/test/resources/model.mdo" ) ) );
51  
52          File generatedSources = getTestFile( "target/" + outputType + "/sources" );
53  
54          File classes = getTestFile( "target/" + outputType + "/classes" );
55  
56          FileUtils.deleteDirectory( generatedSources );
57  
58          FileUtils.deleteDirectory( classes );
59  
60          generatedSources.mkdirs();
61  
62          classes.mkdirs();
63  
64          Properties parameters = new Properties();
65  
66          parameters.setProperty( ModelloParameterConstants.OUTPUT_DIRECTORY, generatedSources.getAbsolutePath() );
67  
68          parameters.setProperty( ModelloParameterConstants.VERSION, "1.0.0" );
69  
70          parameters.setProperty( ModelloParameterConstants.PACKAGE_WITH_VERSION, Boolean.toString( false ) );
71  
72          modello.generate( model, "java", parameters );
73  
74          modello.generate( model, outputType, parameters );
75  
76          String registryVersion = System.getProperty( "registryVersion" );
77  
78          addDependency( "org.codehaus.modello", "modello-core", getModelloVersion() );
79          addDependency( "org.apache.archiva.redback.components.registry", "spring-registry-api", registryVersion );
80          addDependency( "org.apache.archiva.redback.components.registry", "spring-registry-commons", registryVersion );
81          addDependency( "org.codehaus.plexus", "plexus-container-default", "1.0-alpha-33" );
82          addDependency( "commons-collections", "commons-collections", "3.1" );
83          addDependency( "commons-configuration", "commons-configuration", "1.8" );
84          addDependency( "commons-lang", "commons-lang", "2.6" );
85          addDependency( "commons-logging", "commons-logging-api", "1.0.4" );
86          addDependency( "org.codehaus.plexus", "plexus-interpolation", "1.14" );
87          addDependency( "org.slf4j", "slf4j-api", System.getProperty( "slf4jVersion" ) );
88          addDependency( "org.slf4j", "slf4j-simple", System.getProperty( "slf4jVersion" ) );
89  
90          if ( "1.5".compareTo( System.getProperty( "java.specification.version" ) ) <= 0 )
91          {
92              // causes a conflict with JDK 1.4 => add this dependency only with JDK 1.5+
93              addDependency( "xerces", "xercesImpl", "2.9.1" );
94          }
95  
96          compile( generatedSources, classes );
97      }
