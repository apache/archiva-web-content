021    import org.apache.archiva.rest.api.services.ArchivaRestServiceException;
022    import org.apache.archiva.web.model.ApplicationRuntimeInfo;
023    import org.apache.archiva.redback.authorization.RedbackAuthorization;
024    
025    import javax.ws.rs.GET;
026    import javax.ws.rs.Path;
027    import javax.ws.rs.PathParam;
028    import javax.ws.rs.Produces;
029    import javax.ws.rs.QueryParam;
030    import javax.ws.rs.core.MediaType;
031    
032    /**
033     * @author Olivier Lamy
034     * @since 1.4-M3
035     */
036    @Path("/runtimeInfoService/")
037    public interface RuntimeInfoService
038    {
039        @Path("archivaRuntimeInfo/{locale}")
040        @GET
041        @Produces({ MediaType.APPLICATION_JSON, MediaType.APPLICATION_XML })
042        @RedbackAuthorization(noRestriction = true)
043        ApplicationRuntimeInfo getApplicationRuntimeInfo( @PathParam("locale") String locale )
044            throws ArchivaRestServiceException;
045    
046    
047        @Path("logMissingI18n")
048        @GET
049        @Produces({ MediaType.APPLICATION_JSON, MediaType.APPLICATION_XML })
050        @RedbackAuthorization(noRestriction = true)
051        Boolean logMissingI18n( @QueryParam("key") String key );
052    }
