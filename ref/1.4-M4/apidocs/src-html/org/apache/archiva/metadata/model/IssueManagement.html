022    import javax.persistence.Entity;
023    import javax.xml.bind.annotation.XmlRootElement;
024    import java.io.Serializable;
025    
026    /**
027     * Information about the issue management system used by the project.
028     *
029     * @todo considering moving this to a facet - avoid referring to it externally
030     */
031    @XmlRootElement(name = "issueManagement")
032    @Entity
033    public class IssueManagement
034        implements Serializable
035    {
036        /**
037         * A simple identifier for the type of issue management server used, eg <tt>jira</tt>, <tt>bugzilla</tt>, etc.
038         */
039        private String system;
040    
041        /**
042         * The base URL of the issue management system.
043         */
044        private String url;
045    
046        public String getUrl()
047        {
048            return url;
049        }
050    
051        public void setUrl( String url )
052        {
053            this.url = url;
054        }
055    
056        public String getSystem()
057        {
058            return system;
059        }
060    
061        public void setSystem( String system )
062        {
063            this.system = system;
064        }
065    }
