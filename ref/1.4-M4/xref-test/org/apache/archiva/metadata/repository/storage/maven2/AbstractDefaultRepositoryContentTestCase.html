421     private void assertArtifactReference( ArtifactReference actualReference, String groupId, String artifactId,
422                                           String version, String classifier, String type )
423     {
424         String expectedId =
425             "ArtifactReference - " + groupId + ":" + artifactId + ":" + version + ":" + classifier + ":" + type;
426 
427         assertNotNull( expectedId + " - Should not be null.", actualReference );
428 
429         assertEquals( expectedId + " - Group ID", groupId, actualReference.getGroupId() );
430         assertEquals( expectedId + " - Artifact ID", artifactId, actualReference.getArtifactId() );
431         if ( StringUtils.isNotBlank( classifier ) )
432         {
433             assertEquals( expectedId + " - Classifier", classifier, actualReference.getClassifier() );
434         }
435         assertEquals( expectedId + " - Version ID", version, actualReference.getVersion() );
436         assertEquals( expectedId + " - Type", type, actualReference.getType() );
437     }

453     /**
454      * Perform a roundtrip through the layout routines to determine success.
455      */
456     private void assertLayout( String path, String groupId, String artifactId, String version, String classifier,
457                                String type )
458         throws LayoutException
459     {
460         ArtifactReference expectedArtifact = createArtifact( groupId, artifactId, version, classifier, type );
461 
462         // --- Artifact Tests.
463 
464         // Artifact to Path 
465         assertEquals( "Artifact <" + expectedArtifact + "> to path:", path, toPath( expectedArtifact ) );
466 
467         // --- Artifact Reference Tests
468 
469         // Path to Artifact Reference.
470         ArtifactReference testReference = toArtifactReference( path );
471         assertArtifactReference( testReference, groupId, artifactId, version, classifier, type );
472 
473         // And back again, using test Reference from previous step.
474         assertEquals( "Artifact <" + expectedArtifact + "> to path:", path, toPath( testReference ) );
475     }

477     protected ArtifactReference createArtifact( String groupId, String artifactId, String version, String classifier,
478                                               String type )
479     {
480         ArtifactReference artifact = new ArtifactReference();
481         artifact.setGroupId( groupId );
482         artifact.setArtifactId( artifactId );
483         artifact.setVersion( version );
484         artifact.setClassifier( classifier );
485         artifact.setType( type );
486         assertNotNull( artifact );
487         return artifact;
488     }
