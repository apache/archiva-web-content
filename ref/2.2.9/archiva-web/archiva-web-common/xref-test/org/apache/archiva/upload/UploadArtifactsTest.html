81          // to help cleaning files on Windows....
82          System.gc();
83          FileUtils.deleteDirectory(appServerBase.toAbsolutePath().toFile());
84          Path confDir = appServerBase.resolve( "conf" );
85          if ( !Files.exists(confDir))
86          {
87              Files.createDirectories(confDir);
88          }

90          Path log4jCfgDst = confDir.resolve( log4jCfg.getFileName( ) );

92          Files.copy( log4jCfg, log4jCfgDst, StandardCopyOption.REPLACE_EXISTING );

97          Path jcrDirectory = appServerBase.resolve("jcr" );
98  
99          if ( Files.exists(jcrDirectory) )
100         {
101             FileUtils.deleteDirectory( jcrDirectory.toAbsolutePath().toFile() );
102         }
103         // We have to activate this to verify the bad path traversal protection. We cannot rely on
104         // the application server only.
105 
106         SLF4JBridgeHandler.removeHandlersForRootLogger();
107         SLF4JBridgeHandler.install();

116         context.addApplicationParameter(applicationParameter);
117         context.addApplicationListener(ContextLoaderListener.class.getName());
118         Tomcat.addServlet(context, "cxf", new CXFServlet());
119         context.addServletMapping("/" + this.getRestServicesPath() + "/*", "cxf");

164     private FileUploadService getUploadService( )
165     {
166         FileUploadService service =
167             JAXRSClientFactory.create( getBaseUrl( ) + "/" + getRestServicesPath( ) + "/archivaUiServices/",
168                 FileUploadService.class,
169                 Collections.singletonList( new JacksonJaxbJsonProvider( ) ) );
170         log.debug( "Service class {}", service.getClass( ).getName( ) );
171         WebClient.client( service ).header( "Authorization", authorizationHeader );
172         WebClient.client( service ).header( "Referer", "http://localhost:" + port );
173 
174         WebClient.client( service ).header( "Referer", "http://localhost" );
175         WebClient.getConfig( service ).getRequestContext( ).put( Message.MAINTAIN_SESSION, true );
176         WebClient.getConfig( service).getRequestContext().put(Message.EXCEPTION_MESSAGE_CAUSE_ENABLED, true);
177         WebClient.getConfig( service).getRequestContext().put(Message.FAULT_STACKTRACE_ENABLED, true);
178         WebClient.getConfig( service).getRequestContext().put(Message.PROPOGATE_EXCEPTION, true);
179         WebClient.getConfig( service).getRequestContext().put("org.apache.cxf.transport.no_io_exceptions", true);
180 
181         // WebClient.client( service ).
182         return service;
183     }

271     @Test
272     public void failUploadAndDeleteFileInOtherDir( ) throws IOException, ArchivaRestServiceException
273     {
274         Path testFile = null;
275         try
276         {
277             FileUploadService service = getUploadService( );
278             Path file = Paths.get( "src/test/repositories/snapshot-repo/org/apache/archiva/archiva-model/1.4-M4-SNAPSHOT/archiva-model-1.4-M4-20130425.081822-1.jar" );
279             Path targetDir = Paths.get( "target/testDelete" ).toAbsolutePath( );
280             if ( !Files.exists( targetDir ) ) Files.createDirectories( targetDir );
281             Path tempDir = SystemUtils.getJavaIoTmpDir( ).toPath( );
282             testFile = Files.createTempFile( targetDir, "TestFile", ".txt" );
283             log.debug( "Test file {}", testFile.toAbsolutePath( ) );
284             log.debug( "Tmp dir {}", tempDir.toAbsolutePath( ) );
285             assertTrue( Files.exists( testFile ) );
286             Path relativePath = tempDir.relativize( testFile.toAbsolutePath( ) );
287             final Attachment fileAttachment = new AttachmentBuilder( ).object( Files.newInputStream( file ) ).contentDisposition( new ContentDisposition( "form-data; filename=\"" + file.getFileName( ).toString( ) + "\"; name=\"files[]\"" ) ).build( );
288             MultipartBody body = new MultipartBody( fileAttachment );
289             service.post( body );
290             String relativePathEncoded = URLEncoder.encode( "../target/" + relativePath, "UTF-8" );
291             log.debug( "Trying to delete with path traversal: {}, {}", relativePath, relativePathEncoded );
292             try
293             {
294                 service.deleteFile( relativePathEncoded );
295             }
296             catch ( ArchivaRestServiceException ex )
297             {
298                 // Expected exception
299             }
300             assertTrue( "File in another directory may not be deleted", Files.exists( testFile ) );
301         }
302         finally
303         {
304             if ( testFile != null )
305             {
306                 Files.deleteIfExists( testFile );
307             }
308         }
309     }

311     @Test
312     public void failSaveFileWithBadParams( ) throws IOException, ArchivaRestServiceException
313     {
314         FileUploadService service = getUploadService( );
315         Path targetFile = Paths.get( "target/test/test-testSave.4" );
316         Path targetPom = Paths.get( "target/test/test-testSave.pom" );
317         try
318         {
319             Path file = Paths.get( "src/test/repositories/snapshot-repo/org/apache/archiva/archiva-model/1.4-M4-SNAPSHOT/archiva-model-1.4-M4-20130425.081822-1.jar" );
320 
321             Path targetDir = Paths.get( "target/appserver-base/test/testSave" ).toAbsolutePath( );
322             Path repoDir = Paths.get("target/appserver-base/repositories/internal/org");
323             log.info("Repo dir {}", repoDir.toAbsolutePath());
324             if (!Files.exists(repoDir)) Files.createDirectories(repoDir);
325             assertTrue(Files.exists(repoDir));
326             if ( !Files.exists( targetDir ) ) Files.createDirectories( targetDir );
327             Files.deleteIfExists( targetFile );
328             Files.deleteIfExists( targetPom );
329             Attachment fileAttachment = new AttachmentBuilder( ).object( Files.newInputStream( file ) ).contentDisposition( new ContentDisposition( "form-data; filename=\"archiva-model-1.2.jar\"; name=\"files[]\"" ) ).build( );
330             MultipartBody body = new MultipartBody( fileAttachment );
331             FileMetadata meta = service.post( body );
332             log.debug( "Metadata {}", meta.toString( ) );
333             assertTrue( service.save( "internal", "org.archiva", "archiva-model", "1.2", "jar", true ) );
334 
335             fileAttachment = new AttachmentBuilder( ).object( Files.newInputStream( file ) ).contentDisposition( new ContentDisposition( "form-data; filename=\"TestFile.FileExt\"; name=\"files[]\"" ) ).build( );
336             body = new MultipartBody( fileAttachment );
337             meta = service.post( body );
338             log.debug( "Metadata {}", meta.toString( ) );
339             try {
340                 service.save("internal", "org", URLEncoder.encode("../../../test", "UTF-8"), URLEncoder.encode("testSave", "UTF-8"), "4", true);
341                 fail("Error expected, if the content contains bad characters.");
342             } catch (ClientErrorException e) {
343                 assertEquals(422, e.getResponse().getStatus());
344             }
345             assertFalse( Files.exists( Paths.get( "target/test-testSave.4" ) ) );
346         }
347         finally
348         {
349             // service.clearUploadedFiles( );
350             Files.deleteIfExists( targetFile );
351             Files.deleteIfExists( targetPom );
352         }
