109     @SuppressWarnings( "unchecked" )
110     public void prepare()
111         throws RepositoryAdminException
112     {
113         remoteRepositories = new ArrayList<RemoteRepository>( getRemoteRepositoryAdmin().getRemoteRepositories() );
114         managedRepositories = new ArrayList<ManagedRepository>( getManagedRepositoryAdmin().getManagedRepositories() );
115         repositoryToGroupMap = getRepositoryGroupAdmin().getRepositoryToGroupMap();
116 
117         Collections.sort( managedRepositories, new RepositoryComparator() );
118         Collections.sort( remoteRepositories, new RepositoryComparator() );
119 
120         repositoryStatistics = new HashMap<String, RepositoryStatistics>();
121         RepositorySession repositorySession = repositorySessionFactory.createSession();
122         try
123         {
124             MetadataRepository metadataRepository = repositorySession.getRepository();
125             for ( ManagedRepository repo : managedRepositories )
126             {
127                 RepositoryStatistics stats = null;
128                 try
129                 {
130                     stats = repositoryStatisticsManager.getLastStatistics( metadataRepository, repo.getId() );
131                 }
132                 catch ( MetadataRepositoryException e )
133                 {
134                     addActionError(
135                         "Error retrieving statistics for repository " + repo.getId() + " - consult application logs" );
136                     log.warn( "Error retrieving repository statistics: " + e.getMessage(), e );
137                 }
138                 if ( stats != null )
139                 {
140                     repositoryStatistics.put( repo.getId(), stats );
141                 }
142             }
143         }
144         finally
145         {
146             repositorySession.close();
147         }
148     }

156             {
157                 managedRepositoriesList.add( repoConfig );
158             }
159         }
160         return managedRepositoriesList;
161     }
