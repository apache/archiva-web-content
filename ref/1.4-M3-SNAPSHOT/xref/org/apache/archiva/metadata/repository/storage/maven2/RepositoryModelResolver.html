102     public ModelSource resolveModel( String groupId, String artifactId, String version )
103         throws UnresolvableModelException
104     {
105         String filename = artifactId + "-" + version + ".pom";
106         // TODO: we need to convert 1.0-20091120.112233-1 type paths to baseVersion for the below call - add a test
107 
108         File model = pathTranslator.toFile( basedir, groupId, artifactId, version, filename );
109 
110         if ( !model.exists() )
111         {
112 
113             // is a SNAPSHOT ? so we can try to find locally before asking remote repositories.
114             if ( StringUtils.contains( version, VersionUtil.SNAPSHOT ) )
115             {
116                 File localSnapshotModel = findTimeStampedSnapshotPom( groupId, artifactId, version, model.getParent() );
117                 if ( localSnapshotModel != null )
118                 {
119                     return new FileModelSource( localSnapshotModel );
120                 }
121 
122             }
123 
124             for ( RemoteRepository remoteRepository : remoteRepositories )
125             {
126                 try
127                 {
128                     boolean success = getModelFromProxy( remoteRepository, groupId, artifactId, version, filename );
129                     if ( success && model.exists() )
130                     {
131                         log.info(
132                             "Model '" + model.getAbsolutePath() + "' successfully retrieved from remote repository '"
133                                 + remoteRepository.getId() + "'" );
134                         break;
135                     }
136                 }
137                 catch ( ResourceDoesNotExistException e )
138                 {
139                     log.info( "An exception was caught while attempting to retrieve model '" + model.getAbsolutePath()
140                                   + "' from remote repository '" + remoteRepository.getId() + "'.Reason:"
141                                   + e.getMessage() );
142                 }
143                 catch ( Exception e )
144                 {
145                     log.warn( "An exception was caught while attempting to retrieve model '" + model.getAbsolutePath()
146                                   + "' from remote repository '" + remoteRepository.getId() + "'.", e.getMessage() );
147 
148                     continue;
149                 }
150             }
151         }
152 
153         return new FileModelSource( model );
154     }

156     protected File findTimeStampedSnapshotPom( String groupId, String artifactId, String version,
157                                                String parentDirectory )
158     {
159 
160         // reading metadata if there
161         File mavenMetadata = new File( parentDirectory, METADATA_FILENAME );
162         if ( mavenMetadata.exists() )
163         {
164             try
165             {
166                 ArchivaRepositoryMetadata archivaRepositoryMetadata = MavenMetadataReader.read( mavenMetadata );
167                 SnapshotVersion snapshotVersion = archivaRepositoryMetadata.getSnapshotVersion();
168                 if ( snapshotVersion != null )
169                 {
170                     String lastVersion = snapshotVersion.getTimestamp();
171                     int buildNumber = snapshotVersion.getBuildNumber();
172                     String snapshotPath =
173                         StringUtils.replaceChars( groupId, '.', '/' ) + '/' + artifactId + '/' + version + '/'
174                             + artifactId + '-' + StringUtils.remove( version, "-SNAPSHOT" ) + '-' + lastVersion + '-'
175                             + buildNumber + ".pom";
176                     if ( log.isDebugEnabled() )
177                     {
178                         log.debug( "use snapshot path {} for maven coordinate {}", snapshotPath,
179                                    groupId + ":" + artifactId + ":" + version );
180                     }
181                     File model = new File( basedir, snapshotPath );
182                     //model = pathTranslator.toFile( basedir, groupId, artifactId, lastVersion, filename );
183                     if ( model.exists() )
184                     {
185                         return model;
186                     }
187                 }
188             }
189             catch ( XMLException e )
190             {
191                 log.warn( "fail to read {}, {}", mavenMetadata.getAbsolutePath(), e.getCause() );
192             }
193         }
194 
195         return null;
196     }

211     // FIXME: we need to do some refactoring, we cannot re-use the proxy components of archiva-proxy in maven2-repository
212     // because it's causing a cyclic dependency
213     private boolean getModelFromProxy( RemoteRepository remoteRepository, String groupId, String artifactId,
214                                        String version, String filename )
215         throws AuthorizationException, TransferFailedException, ResourceDoesNotExistException, WagonFactoryException,
216         XMLException
217     {
218         boolean success = false;
219         File tmpMd5 = null;
220         File tmpSha1 = null;
221         File tmpResource = null;
222         String artifactPath = pathTranslator.toPath( groupId, artifactId, version, filename );
223         File resource = new File( targetRepository.getLocation(), artifactPath );
224 
225         File workingDirectory = createWorkingDirectory( targetRepository.getLocation() );
226         try
227         {
228             Wagon wagon = null;
229             try
230             {
231                 String protocol = getProtocol( remoteRepository.getUrl() );
232                 final NetworkProxy networkProxy = this.networkProxyMap.get( remoteRepository.getId() );
233 
234                 wagon = ( networkProxy != null && networkProxy.isUseNtlm() ) ? wagonFactory.getWagon(
235                     "wagon#" + protocol + "-ntlm" ) : wagonFactory.getWagon( "wagon#" + protocol );
236                 wagon = wagonFactory.getWagon( "wagon#" + protocol );
237                 if ( wagon == null )
238                 {
239                     throw new RuntimeException( "Unsupported remote repository protocol: " + protocol );
240                 }
241 
242                 boolean connected = connectToRepository( wagon, remoteRepository );
243                 if ( connected )
244                 {
245                     tmpResource = new File( workingDirectory, filename );
246 
247                     if ( VersionUtil.isSnapshot( version ) )
248                     {
249                         // get the metadata first!
250                         File tmpMetadataResource = new File( workingDirectory, METADATA_FILENAME );
251 
252                         String metadataPath =
253                             StringUtils.substringBeforeLast( artifactPath, "/" ) + "/" + METADATA_FILENAME;
254 
255                         wagon.get( metadataPath, tmpMetadataResource );
256 
257                         log.debug( "Successfully downloaded metadata." );
258 
259                         ArchivaRepositoryMetadata metadata = MavenMetadataReader.read( tmpMetadataResource );
260 
261                         // re-adjust to timestamp if present, otherwise retain the original -SNAPSHOT filename
262                         SnapshotVersion snapshotVersion = metadata.getSnapshotVersion();
263                         String timestampVersion = version;
264                         if ( snapshotVersion != null )
265                         {
266                             timestampVersion = timestampVersion.substring( 0, timestampVersion.length()
267                                 - 8 ); // remove SNAPSHOT from end
268                             timestampVersion = timestampVersion + snapshotVersion.getTimestamp() + "-"
269                                 + snapshotVersion.getBuildNumber();
270 
271                             filename = artifactId + "-" + timestampVersion + ".pom";
272 
273                             artifactPath = pathTranslator.toPath( groupId, artifactId, version, filename );
274 
275                             log.debug( "New artifactPath : " + artifactPath );
276                         }
277                     }
278 
279                     log.info( "Retrieving " + artifactPath + " from " + remoteRepository.getName() );
280 
281                     wagon.get( artifactPath, tmpResource );
282 
283                     log.debug( "Downloaded successfully." );
284 
285                     tmpSha1 = transferChecksum( wagon, remoteRepository, artifactPath, tmpResource, workingDirectory,
286                                                 ".sha1" );
287                     tmpMd5 = transferChecksum( wagon, remoteRepository, artifactPath, tmpResource, workingDirectory,
288                                                ".md5" );
289                 }
290             }
291             finally
292             {
293                 if ( wagon != null )
294                 {
295                     try
296                     {
297                         wagon.disconnect();
298                     }
299                     catch ( ConnectionException e )
300                     {
301                         log.warn( "Unable to disconnect wagon.", e );
302                     }
303                 }
304             }
305 
306             if ( resource != null )
307             {
308                 synchronized ( resource.getAbsolutePath().intern() )
309                 {
310                     File directory = resource.getParentFile();
311                     moveFileIfExists( tmpMd5, directory );
312                     moveFileIfExists( tmpSha1, directory );
313                     moveFileIfExists( tmpResource, directory );
314                     success = true;
315                 }
316             }
317         }
318         finally
319         {
320             FileUtils.deleteQuietly( workingDirectory );
321         }
322 
323         // do we still need to execute the consumers?
324 
325         return success;
326     }

328     /**
329      * Using wagon, connect to the remote repository.
330      *
331      * @param wagon the wagon instance to establish the connection on.
332      * @return true if the connection was successful. false if not connected.
333      */
334     private boolean connectToRepository( Wagon wagon, RemoteRepository remoteRepository )
335     {
336         boolean connected;
337 
338         final NetworkProxy proxyConnector = this.networkProxyMap.get( remoteRepository.getId() );
339         ProxyInfo networkProxy = null;
340         if ( proxyConnector != null )
341         {
342             networkProxy = new ProxyInfo();
343             networkProxy.setType( proxyConnector.getProtocol() );
344             networkProxy.setHost( proxyConnector.getHost() );
345             networkProxy.setPort( proxyConnector.getPort() );
346             networkProxy.setUserName( proxyConnector.getUsername() );
347             networkProxy.setPassword( proxyConnector.getPassword() );
348 
349             String msg = "Using network proxy " + networkProxy.getHost() + ":" + networkProxy.getPort()
350                 + " to connect to remote repository " + remoteRepository.getUrl();
351             if ( networkProxy.getNonProxyHosts() != null )
352             {
353                 msg += "; excluding hosts: " + networkProxy.getNonProxyHosts();
354             }
355 
356             if ( StringUtils.isNotBlank( networkProxy.getUserName() ) )
357             {
358                 msg += "; as user: " + networkProxy.getUserName();
359             }
360 
361             log.debug( msg );
362         }
363 
364         AuthenticationInfo authInfo = null;
365         String username = remoteRepository.getUserName();
366         String password = remoteRepository.getPassword();
367 
368         if ( StringUtils.isNotBlank( username ) && StringUtils.isNotBlank( password ) )
369         {
370             log.debug( "Using username {} to connect to remote repository {}", username, remoteRepository.getUrl() );
371             authInfo = new AuthenticationInfo();
372             authInfo.setUserName( username );
373             authInfo.setPassword( password );
374         }
375 
376         // Convert seconds to milliseconds
377         int timeoutInMilliseconds = remoteRepository.getTimeout() * 1000;
378         // FIXME olamy having 2 config values
379         // Set timeout
380         wagon.setReadTimeout( timeoutInMilliseconds );
381         wagon.setTimeout( timeoutInMilliseconds );
382 
383         try
384         {
385             org.apache.maven.wagon.repository.Repository wagonRepository =
386                 new org.apache.maven.wagon.repository.Repository( remoteRepository.getId(), remoteRepository.getUrl() );
387             if ( networkProxy != null )
388             {
389                 wagon.connect( wagonRepository, authInfo, networkProxy );
390             }
391             else
392             {
393                 wagon.connect( wagonRepository, authInfo );
394             }
395             connected = true;
396         }
397         catch ( ConnectionException e )
398         {
399             log.error( "Could not connect to " + remoteRepository.getName() + ": " + e.getMessage() );
400             connected = false;
401         }
402         catch ( AuthenticationException e )
403         {
404             log.error( "Could not connect to " + remoteRepository.getName() + ": " + e.getMessage() );
405             connected = false;
406         }
407 
408         return connected;
409     }

438     private void moveFileIfExists( File fileToMove, File directory )
439     {
440         if ( fileToMove != null && fileToMove.exists() )
441         {
442             File newLocation = new File( directory, fileToMove.getName() );
443             if ( newLocation.exists() && !newLocation.delete() )
444             {
445                 throw new RuntimeException(
446                     "Unable to overwrite existing target file: " + newLocation.getAbsolutePath() );
447             }
448 
449             newLocation.getParentFile().mkdirs();
450             if ( !fileToMove.renameTo( newLocation ) )
451             {
452                 log.warn( "Unable to rename tmp file to its final name... resorting to copy command." );
453 
454                 try
455                 {
456                     FileUtils.copyFile( fileToMove, newLocation );
457                 }
458                 catch ( IOException e )
459                 {
460                     if ( newLocation.exists() )
461                     {
462                         log.error( "Tried to copy file " + fileToMove.getName() + " to " + newLocation.getAbsolutePath()
463                                        + " but file with this name already exists." );
464                     }
465                     else
466                     {
467                         throw new RuntimeException(
468                             "Cannot copy tmp file " + fileToMove.getAbsolutePath() + " to its final location", e );
469                     }
470                 }
471                 finally
472                 {
473                     FileUtils.deleteQuietly( fileToMove );
474                 }
475             }
476         }
