31  /**
32   * Generate Plexus Registry output mechanism for model 'Configuration'.

34  public class ConfigurationRegistryWriter
35  {
36      public void write( Configuration model, Registry registry )
37      {
38          writeConfiguration( "", model, registry );

41      private void writeConfiguration( String prefix, Configuration value, Registry registry )
42      {
43          if ( value != null )
44          {
45              if ( value.getVersion() != null
46   )
47              {
48                  String name = "version";
49                  registry.setString( prefix + name, value.getVersion() );
50              }
51              if ( value.getRepositories() != null && value.getRepositories().size() > 0
52   )
53              {
54                  registry.removeSubset( prefix + "repositories" );
55  
56                  int count = 0;
57                  for ( Iterator iter = value.getRepositories().iterator(); iter.hasNext(); count++ )
58                  {
59                      String name = "repositories.repository(" + count + ")";
60                      V1RepositoryConfiguration o = ( V1RepositoryConfiguration ) iter.next();
61                      writeV1RepositoryConfiguration( prefix + name + ".", o, registry );
62                  }
63              }
64              if ( value.getRepositoryGroups() != null && value.getRepositoryGroups().size() > 0
65   )
66              {
67                  registry.removeSubset( prefix + "repositoryGroups" );
68  
69                  int count = 0;
70                  for ( Iterator iter = value.getRepositoryGroups().iterator(); iter.hasNext(); count++ )
71                  {
72                      String name = "repositoryGroups.repositoryGroup(" + count + ")";
73                      RepositoryGroupConfiguration o = ( RepositoryGroupConfiguration ) iter.next();
74                      writeRepositoryGroupConfiguration( prefix + name + ".", o, registry );
75                  }
76              }
77              if ( value.getManagedRepositories() != null && value.getManagedRepositories().size() > 0
78   )
79              {
80                  registry.removeSubset( prefix + "managedRepositories" );
81  
82                  int count = 0;
83                  for ( Iterator iter = value.getManagedRepositories().iterator(); iter.hasNext(); count++ )
84                  {
85                      String name = "managedRepositories.managedRepository(" + count + ")";
86                      ManagedRepositoryConfiguration o = ( ManagedRepositoryConfiguration ) iter.next();
87                      writeManagedRepositoryConfiguration( prefix + name + ".", o, registry );
88                  }
89              }
90              if ( value.getRemoteRepositories() != null && value.getRemoteRepositories().size() > 0
91   )
92              {
93                  registry.removeSubset( prefix + "remoteRepositories" );
94  
95                  int count = 0;
96                  for ( Iterator iter = value.getRemoteRepositories().iterator(); iter.hasNext(); count++ )
97                  {
98                      String name = "remoteRepositories.remoteRepository(" + count + ")";
99                      RemoteRepositoryConfiguration o = ( RemoteRepositoryConfiguration ) iter.next();
100                     writeRemoteRepositoryConfiguration( prefix + name + ".", o, registry );
101                 }
102             }
103             if ( value.getProxyConnectors() != null && value.getProxyConnectors().size() > 0
104  )
105             {
106                 registry.removeSubset( prefix + "proxyConnectors" );
107 
108                 int count = 0;
109                 for ( Iterator iter = value.getProxyConnectors().iterator(); iter.hasNext(); count++ )
110                 {
111                     String name = "proxyConnectors.proxyConnector(" + count + ")";
112                     ProxyConnectorConfiguration o = ( ProxyConnectorConfiguration ) iter.next();
113                     writeProxyConnectorConfiguration( prefix + name + ".", o, registry );
114                 }
115             }
116             if ( value.getNetworkProxies() != null && value.getNetworkProxies().size() > 0
117  )
118             {
119                 registry.removeSubset( prefix + "networkProxies" );
120 
121                 int count = 0;
122                 for ( Iterator iter = value.getNetworkProxies().iterator(); iter.hasNext(); count++ )
123                 {
124                     String name = "networkProxies.networkProxy(" + count + ")";
125                     NetworkProxyConfiguration o = ( NetworkProxyConfiguration ) iter.next();
126                     writeNetworkProxyConfiguration( prefix + name + ".", o, registry );
127                 }
128             }
129             if ( value.getLegacyArtifactPaths() != null && value.getLegacyArtifactPaths().size() > 0
130  )
131             {
132                 registry.removeSubset( prefix + "legacyArtifactPaths" );
133 
134                 int count = 0;
135                 for ( Iterator iter = value.getLegacyArtifactPaths().iterator(); iter.hasNext(); count++ )
136                 {
137                     String name = "legacyArtifactPaths.legacyArtifactPath(" + count + ")";
138                     LegacyArtifactPath o = ( LegacyArtifactPath ) iter.next();
139                     writeLegacyArtifactPath( prefix + name + ".", o, registry );
140                 }
141             }
142             if ( value.getRepositoryScanning() != null
143  )
144             {
145                 writeRepositoryScanningConfiguration( prefix + "repositoryScanning.", value.getRepositoryScanning(), registry );
146             }
147             if ( value.getWebapp() != null
148  )
149             {
150                 writeWebappConfiguration( prefix + "webapp.", value.getWebapp(), registry );
151             }
152             if ( value.getOrganisationInfo() != null
153  )
154             {
155                 writeOrganisationInformation( prefix + "organisationInfo.", value.getOrganisationInfo(), registry );
156             }
157             if ( value.getNetworkConfiguration() != null
158  )
159             {
160                 writeNetworkConfiguration( prefix + "networkConfiguration.", value.getNetworkConfiguration(), registry );
161             }
162         }
