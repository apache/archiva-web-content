021    import org.codehaus.jackson.map.DeserializationConfig;
022    import org.codehaus.jackson.map.ObjectMapper;
023    import org.slf4j.Logger;
024    import org.slf4j.LoggerFactory;
025    import org.springframework.stereotype.Service;
026    
027    import javax.inject.Inject;
028    
029    /**
030     * class to setup Jackson Json configuration
031     * @since 1.4-M3
032     * @author Olivier Lamy
033     */
034    @Service("archivaJacksonJsonConfigurator")
035    public class JacksonJsonConfigurator
036    {
037        private Logger log = LoggerFactory.getLogger( getClass() );
038    
039        @Inject
040        public JacksonJsonConfigurator( ObjectMapper objectMapper )
041        {
042            log.info( "configure jackson ObjectMapper" );
043            objectMapper.configure( DeserializationConfig.Feature.FAIL_ON_UNKNOWN_PROPERTIES, false );
044        }
045    }
