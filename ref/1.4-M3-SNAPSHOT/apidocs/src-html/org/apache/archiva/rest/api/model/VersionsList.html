021    import javax.xml.bind.annotation.XmlRootElement;
022    import java.io.Serializable;
023    import java.util.List;
024    
025    /**
026     * @author Olivier Lamy
027     */
028    @XmlRootElement( name = "versionsList" )
029    public class VersionsList
030        implements Serializable
031    {
032    
033        private List<String> versions;
034    
035        public VersionsList()
036        {
037            // no op
038        }
039    
040        public VersionsList( List<String> versions )
041        {
042            this.versions = versions;
043        }
044    
045        public List<String> getVersions()
046        {
047            return versions;
048        }
049    
050        public void setVersions( List<String> versions )
051        {
052            this.versions = versions;
053        }
054    
055        @Override
056        public String toString()
057        {
058            final StringBuilder sb = new StringBuilder();
059            sb.append( "VersionsList" );
060            sb.append( "{versions=" ).append( versions );
061            sb.append( '}' );
062            return sb.toString();
063        }
064    }
