022    import org.apache.archiva.scheduler.ArchivaTaskScheduler;
023    import org.apache.archiva.redback.components.taskqueue.TaskQueue;
024    import org.apache.archiva.redback.components.taskqueue.TaskQueueException;
025    import org.slf4j.Logger;
026    import org.slf4j.LoggerFactory;
027    import org.springframework.stereotype.Service;
028    
029    import javax.inject.Inject;
030    import javax.inject.Named;
031    
032    /**
033     * Default implementation of a scheduling component for archiva.
034     *
035     * @todo TODO - consider just folding in, not really scheduled
036     */
037    @Service("archivaTaskScheduler#indexing")
038    public class IndexingArchivaTaskScheduler
039        implements ArchivaTaskScheduler<ArtifactIndexingTask>
040    {
041        private Logger log = LoggerFactory.getLogger( IndexingArchivaTaskScheduler.class );
042    
043        /**
044         *
045         */
046        @Inject
047        @Named(value = "taskQueue#indexing")
048        private TaskQueue indexingQueue;
049    
050        public void queueTask( ArtifactIndexingTask task )
051            throws TaskQueueException
052        {
053            indexingQueue.put( task );
054        }
055    
056    }
