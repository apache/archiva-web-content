041        private Configuration readConfiguration( String prefix, Registry registry )
042        {
043            Configuration value = new Configuration();
044    
045            String version = registry.getString( prefix + "version", value.getVersion() );
046            value.setVersion( version );
047            java.util.List repositories = new java.util.ArrayList/*<V1RepositoryConfiguration>*/();
048            List repositoriesSubsets = registry.getSubsetList( prefix + "repositories.repository" );
049            for ( Iterator i = repositoriesSubsets.iterator(); i.hasNext(); )
050            {
051                V1RepositoryConfiguration v = readV1RepositoryConfiguration( "", (Registry) i.next() );
052                repositories.add( v );
053            }
054            value.setRepositories( repositories );
055            java.util.List repositoryGroups = new java.util.ArrayList/*<RepositoryGroupConfiguration>*/();
056            List repositoryGroupsSubsets = registry.getSubsetList( prefix + "repositoryGroups.repositoryGroup" );
057            for ( Iterator i = repositoryGroupsSubsets.iterator(); i.hasNext(); )
058            {
059                RepositoryGroupConfiguration v = readRepositoryGroupConfiguration( "", (Registry) i.next() );
060                repositoryGroups.add( v );
061            }
062            value.setRepositoryGroups( repositoryGroups );
063            java.util.List managedRepositories = new java.util.ArrayList/*<ManagedRepositoryConfiguration>*/();
064            List managedRepositoriesSubsets = registry.getSubsetList( prefix + "managedRepositories.managedRepository" );
065            for ( Iterator i = managedRepositoriesSubsets.iterator(); i.hasNext(); )
066            {
067                ManagedRepositoryConfiguration v = readManagedRepositoryConfiguration( "", (Registry) i.next() );
068                managedRepositories.add( v );
069            }
070            value.setManagedRepositories( managedRepositories );
071            java.util.List remoteRepositories = new java.util.ArrayList/*<RemoteRepositoryConfiguration>*/();
072            List remoteRepositoriesSubsets = registry.getSubsetList( prefix + "remoteRepositories.remoteRepository" );
073            for ( Iterator i = remoteRepositoriesSubsets.iterator(); i.hasNext(); )
074            {
075                RemoteRepositoryConfiguration v = readRemoteRepositoryConfiguration( "", (Registry) i.next() );
076                remoteRepositories.add( v );
077            }
078            value.setRemoteRepositories( remoteRepositories );
079            java.util.List proxyConnectors = new java.util.ArrayList/*<ProxyConnectorConfiguration>*/();
080            List proxyConnectorsSubsets = registry.getSubsetList( prefix + "proxyConnectors.proxyConnector" );
081            for ( Iterator i = proxyConnectorsSubsets.iterator(); i.hasNext(); )
082            {
083                ProxyConnectorConfiguration v = readProxyConnectorConfiguration( "", (Registry) i.next() );
084                proxyConnectors.add( v );
085            }
086            value.setProxyConnectors( proxyConnectors );
087            java.util.List networkProxies = new java.util.ArrayList/*<NetworkProxyConfiguration>*/();
088            List networkProxiesSubsets = registry.getSubsetList( prefix + "networkProxies.networkProxy" );
089            for ( Iterator i = networkProxiesSubsets.iterator(); i.hasNext(); )
090            {
091                NetworkProxyConfiguration v = readNetworkProxyConfiguration( "", (Registry) i.next() );
092                networkProxies.add( v );
093            }
094            value.setNetworkProxies( networkProxies );
095            java.util.List legacyArtifactPaths = new java.util.ArrayList/*<LegacyArtifactPath>*/();
096            List legacyArtifactPathsSubsets = registry.getSubsetList( prefix + "legacyArtifactPaths.legacyArtifactPath" );
097            for ( Iterator i = legacyArtifactPathsSubsets.iterator(); i.hasNext(); )
098            {
099                LegacyArtifactPath v = readLegacyArtifactPath( "", (Registry) i.next() );
100                legacyArtifactPaths.add( v );
101            }
102            value.setLegacyArtifactPaths( legacyArtifactPaths );
103            RepositoryScanningConfiguration repositoryScanning = readRepositoryScanningConfiguration( prefix + "repositoryScanning.", registry );
104            value.setRepositoryScanning( repositoryScanning );
105            WebappConfiguration webapp = readWebappConfiguration( prefix + "webapp.", registry );
106            value.setWebapp( webapp );
107            OrganisationInformation organisationInfo = readOrganisationInformation( prefix + "organisationInfo.", registry );
108            value.setOrganisationInfo( organisationInfo );
109            NetworkConfiguration networkConfiguration = readNetworkConfiguration( prefix + "networkConfiguration.", registry );
110            value.setNetworkConfiguration( networkConfiguration );
111    
112            return value;
113        }
