022    import com.opensymphony.xwork2.util.ValueStack;
023    import org.apache.struts2.components.Component;
024    
025    import javax.servlet.http.HttpServletRequest;
026    import javax.servlet.http.HttpServletResponse;
027    import javax.servlet.jsp.JspException;
028    
029    import org.apache.struts2.views.annotations.StrutsTag;
030    import org.apache.struts2.views.annotations.StrutsTagAttribute;
031    import org.apache.struts2.views.jsp.ComponentTagSupport;
032    
033    /**
034     * GroupIdLink 
035     *
036     *
037     */
038    @StrutsTag(name = "groupIdLink", tldBodyContent = "empty", tldTagClass = "org.apache.archiva.web.tags.GroupIdLinkTag", description = "Render a groupId as a set of Links")
039    public class GroupIdLinkTag
040        extends ComponentTagSupport
041    {
042        private String var_; // stores EL-based property
043    
044        private String var; // stores the evaluated object.
045    
046        private boolean includeTop = false;
047    
048        @Override
049        public Component getBean(ValueStack valueStack, HttpServletRequest request, HttpServletResponse response) {
050            return new GroupIdLink( valueStack, request, response );
051        }
052    
053        @Override
054        public void release()
055        {
056            var_ = null;
057            var = null;
058            includeTop = false;
059    
060            super.release();
061        }
062    
063        @Override
064        public int doEndTag()
065            throws JspException
066        {
067            evaluateExpressions();
068            
069            GroupIdLink groupIdLink = (GroupIdLink)component;
070    
071            groupIdLink.setGroupId( var );
072            groupIdLink.setIncludeTop( includeTop );
073    
074            return super.doEndTag();
075        }
076    
077        private void evaluateExpressions()
078            throws JspException
079        {
080            ExpressionTool exprTool = new ExpressionTool( pageContext, this, "groupIdLink" );
081            
082            var = exprTool.optionalString( "var", var_, "" );
083        }
084    
085        @StrutsTagAttribute(description = "The GroupID String", type = "String", defaultValue = "", required = true, rtexprvalue = true)
086        public void setVar( String value )
087        {
088            this.var_ = value;
089        }
090    
091        @StrutsTagAttribute(description = "Boolean indicating if 'top' link should be created or not.", type = "String", defaultValue = "", required = false, rtexprvalue = true)
092        public void setIncludeTop( boolean includeTop )
093        {
094            this.includeTop = includeTop;
095        }
096    }
