022    import org.apache.archiva.admin.model.RepositoryCommonValidator;
023    import org.apache.archiva.admin.model.managed.ManagedRepositoryAdmin;
024    import org.apache.archiva.audit.Auditable;
025    import org.apache.archiva.security.common.ArchivaRoleConstants;
026    import org.apache.archiva.web.action.AbstractActionSupport;
027    import org.apache.archiva.redback.rbac.Resource;
028    import org.apache.archiva.redback.integration.interceptor.SecureAction;
029    import org.apache.archiva.redback.integration.interceptor.SecureActionBundle;
030    import org.apache.archiva.redback.integration.interceptor.SecureActionException;
031    
032    import javax.inject.Inject;
033    
034    /**
035     * Abstract AdminRepositories Action base.
036     * <p/>
037     * Base class for all repository administrative functions.
038     * This should be neutral to the type of action (add/edit/delete) and type of repo (managed/remote)
039     *
040     *
041     */
042    public abstract class AbstractRepositoriesAdminAction
043        extends AbstractActionSupport
044        implements SecureAction, Auditable
045    {
046    
047        @Inject
048        private ManagedRepositoryAdmin managedRepositoryAdmin;
049    
050        @Inject
051        private RepositoryCommonValidator repositoryCommonValidator;
052    
053    
054        public SecureActionBundle getSecureActionBundle()
055            throws SecureActionException
056        {
057            SecureActionBundle bundle = new SecureActionBundle();
058    
059            bundle.setRequiresAuthentication( true );
060            bundle.addRequiredAuthorization( ArchivaRoleConstants.OPERATION_MANAGE_CONFIGURATION, Resource.GLOBAL );
061    
062            return bundle;
063        }
064    
065    
066    
067        public ManagedRepositoryAdmin getManagedRepositoryAdmin()
068        {
069            return managedRepositoryAdmin;
070        }
071    
072        public void setManagedRepositoryAdmin( ManagedRepositoryAdmin managedRepositoryAdmin )
073        {
074            this.managedRepositoryAdmin = managedRepositoryAdmin;
075        }
076    
077        public RepositoryCommonValidator getRepositoryCommonValidator()
078        {
079            return repositoryCommonValidator;
080        }
081    
082        public void setRepositoryCommonValidator( RepositoryCommonValidator repositoryCommonValidator )
083        {
084            this.repositoryCommonValidator = repositoryCommonValidator;
085        }
086    }
