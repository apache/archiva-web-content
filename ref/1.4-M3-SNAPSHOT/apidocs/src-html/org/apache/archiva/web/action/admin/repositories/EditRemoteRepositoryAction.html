022    import com.opensymphony.xwork2.Preparable;
023    import org.apache.archiva.admin.model.RepositoryAdminException;
024    import org.apache.archiva.admin.model.beans.RemoteRepository;
025    import org.apache.archiva.scheduler.indexing.DownloadRemoteIndexException;
026    import org.apache.archiva.scheduler.indexing.DownloadRemoteIndexScheduler;
027    import org.apache.commons.lang.StringUtils;
028    import org.springframework.context.annotation.Scope;
029    import org.springframework.stereotype.Controller;
030    
031    import javax.inject.Inject;
032    
033    /**
034     * EditRemoteRepositoryAction
035     *
036     *
037     */
038    @Controller( "editRemoteRepositoryAction" )
039    @Scope( "prototype" )
040    public class EditRemoteRepositoryAction
041        extends AbstractRemoteRepositoriesAction
042        implements Preparable
043    {
044        /**
045         * The model for this action.
046         */
047        private RemoteRepository repository;
048    
049        /**
050         * The repository id to edit.
051         */
052        private String repoid;
053    
054        private boolean now, fullDownload;
055    
056        @Inject
057        private DownloadRemoteIndexScheduler downloadRemoteIndexScheduler;
058    
059        public void prepare()
060            throws RepositoryAdminException
061        {
062            if ( StringUtils.isNotBlank( repoid ) )
063            {
064                this.repository = getRemoteRepositoryAdmin().getRemoteRepository( repoid );
065            }
066            setNetworkProxies( getNetworkProxyAdmin().getNetworkProxies() );
067        }
068    
069        public String input()
070        {
071            if ( StringUtils.isBlank( repoid ) )
072            {
073                addActionError( "Edit failure, unable to edit a repository with a blank repository id." );
074                return ERROR;
075            }
076    
077            return INPUT;
078        }
079    
080        public String commit()
081        {
082            String result = SUCCESS;
083            try
084            {
085                getRemoteRepositoryAdmin().updateRemoteRepository( getRepository(), getAuditInformation() );
086            }
087            catch ( RepositoryAdminException e )
088            {
089                addActionError( "RepositoryAdminException: " + e.getMessage() );
090                result = INPUT;
091            }
092    
093            return result;
094        }
095    
096        public String downloadRemoteIndex()
097        {
098            try
099            {
100                downloadRemoteIndexScheduler.scheduleDownloadRemote( repoid, now, fullDownload );
101            }
102            catch ( DownloadRemoteIndexException e )
103            {
104                addActionError( "DownloadRemoteIndexException: " + e.getMessage() );
105                return INPUT;
106            }
107            return SUCCESS;
108        }
109    
110        public RemoteRepository getRepository()
111        {
112            return repository;
113        }
114    
115        public void setRepository( RemoteRepository repository )
116        {
117            this.repository = repository;
118        }
119    
120        public String getRepoid()
121        {
122            return repoid;
123        }
124    
125        public void setRepoid( String repoid )
126        {
127            this.repoid = repoid;
128        }
129    
130        public boolean isNow()
131        {
132            return now;
133        }
134    
135        public void setNow( boolean now )
136        {
137            this.now = now;
138        }
139    
140        public boolean isFullDownload()
141        {
142            return fullDownload;
143        }
144    
145        public void setFullDownload( boolean fullDownload )
146        {
147            this.fullDownload = fullDownload;
148        }
149    }
