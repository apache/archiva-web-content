022    import com.opensymphony.xwork2.Preparable;
023    import org.apache.archiva.admin.model.RepositoryAdminException;
024    import org.apache.archiva.admin.model.beans.RemoteRepository;
025    import org.apache.commons.lang.StringUtils;
026    import org.springframework.context.annotation.Scope;
027    import org.springframework.stereotype.Controller;
028    
029    /**
030     * DeleteRemoteRepositoryAction
031     *
032     *
033     */
034    @Controller( "deleteRemoteRepositoryAction" )
035    @Scope( "prototype" )
036    public class DeleteRemoteRepositoryAction
037        extends AbstractRemoteRepositoriesAction
038        implements Preparable
039    {
040        private RemoteRepository repository;
041    
042        private String repoid;
043    
044        public void prepare()
045            throws RepositoryAdminException
046        {
047            if ( StringUtils.isNotBlank( repoid ) )
048            {
049                this.repository = getRemoteRepositoryAdmin().getRemoteRepository( repoid );
050            }
051        }
052    
053        public String confirmDelete()
054        {
055            if ( StringUtils.isBlank( repoid ) )
056            {
057                addActionError( "Unable to delete remote repository: repository id was blank." );
058                return ERROR;
059            }
060    
061            return INPUT;
062        }
063    
064        public String delete()
065        {
066            String result = SUCCESS;
067            RemoteRepository existingRepository = repository;
068            if ( existingRepository == null )
069            {
070                addActionError( "A repository with that id does not exist" );
071                return ERROR;
072            }
073    
074            try
075            {
076                getRemoteRepositoryAdmin().deleteRemoteRepository( existingRepository.getId(), getAuditInformation() );
077            }
078            catch ( RepositoryAdminException e )
079            {
080                addActionError( "RepositoryAdminException: " + e.getMessage() );
081                result = ERROR;
082            }
083            return result;
084        }
085    
086    
087        public RemoteRepository getRepository()
088        {
089            return repository;
090        }
091    
092        public void setRepository( RemoteRepository repository )
093        {
094            this.repository = repository;
095        }
096    
097        public String getRepoid()
098        {
099            return repoid;
100        }
101    
102        public void setRepoid( String repoid )
103        {
104            this.repoid = repoid;
105        }
106    }
