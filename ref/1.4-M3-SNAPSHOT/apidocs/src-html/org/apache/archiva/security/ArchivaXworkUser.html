022    import java.util.Map;
023    
024    import org.apache.archiva.redback.users.UserManager;
025    import org.apache.archiva.redback.system.SecuritySession;
026    import org.apache.archiva.redback.system.SecuritySystemConstants;
027    import org.apache.archiva.redback.users.User;
028    
029    /**
030     * ArchivaXworkUser
031     *
032     *
033     */
034    public final class ArchivaXworkUser
035    {
036        private ArchivaXworkUser()
037        {
038            // no touchy
039        }
040        
041        public static String getActivePrincipal( Map<String, ?> sessionMap )
042        {
043            if ( sessionMap == null )
044            {
045                return UserManager.GUEST_USERNAME;
046            }
047            
048            SecuritySession securitySession =
049                (SecuritySession) sessionMap.get( SecuritySystemConstants.SECURITY_SESSION_KEY );
050    
051            if ( securitySession == null )
052            {
053                return UserManager.GUEST_USERNAME;
054            }
055    
056            User user = securitySession.getUser();
057            if ( user == null )
058            {
059                return UserManager.GUEST_USERNAME;
060            }
061    
062            return (String) user.getPrincipal();
063        }
064    }
