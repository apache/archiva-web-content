149        public ProjectVersionMetadata readProjectVersionMetadata( String repoId, String namespace, String projectId,
150                                                                  String projectVersion )
151            throws RepositoryStorageMetadataNotFoundException, RepositoryStorageMetadataInvalidException,
152            RepositoryStorageRuntimeException
153        {
154            try
155            {
156                ManagedRepository managedRepository = managedRepositoryAdmin.getManagedRepository( repoId );
157    
158                String artifactVersion = projectVersion;
159    
160                File basedir = new File( managedRepository.getLocation() );
161                if ( VersionUtil.isSnapshot( projectVersion ) )
162                {
163                    File metadataFile =
164                        pathTranslator.toFile( basedir, namespace, projectId, projectVersion, METADATA_FILENAME );
165                    try
166                    {
167                        ArchivaRepositoryMetadata metadata = MavenMetadataReader.read( metadataFile );
168    
169                        // re-adjust to timestamp if present, otherwise retain the original -SNAPSHOT filename
170                        SnapshotVersion snapshotVersion = metadata.getSnapshotVersion();
171                        if ( snapshotVersion != null )
172                        {
173                            artifactVersion =
174                                artifactVersion.substring( 0, artifactVersion.length() - 8 ); // remove SNAPSHOT from end
175                            artifactVersion =
176                                artifactVersion + snapshotVersion.getTimestamp() + "-" + snapshotVersion.getBuildNumber();
177                        }
178                    }
179                    catch ( XMLException e )
180                    {
181                        // unable to parse metadata - log it, and continue with the version as the original SNAPSHOT version
182                        log.warn( "Invalid metadata: " + metadataFile + " - " + e.getMessage() );
183                    }
184                }
185    
186                // TODO: won't work well with some other layouts, might need to convert artifact parts to ID by path translator
187                String id = projectId + "-" + artifactVersion + ".pom";
188                File file = pathTranslator.toFile( basedir, namespace, projectId, projectVersion, id );
189    
190                if ( !file.exists() )
191                {
192                    // metadata could not be resolved
193                    throw new RepositoryStorageMetadataNotFoundException(
194                        "The artifact's POM file '" + file.getAbsolutePath() + "' was missing" );
195                }
196    
197                // TODO: this is a workaround until we can properly resolve using proxies as well - this doesn't cache
198                //       anything locally!
199                List<RemoteRepository> remoteRepositories = new ArrayList<RemoteRepository>();
200                Map<String, NetworkProxy> networkProxies = new HashMap<String, NetworkProxy>();
201    
202                Map<String, List<ProxyConnector>> proxyConnectorsMap = proxyConnectorAdmin.getProxyConnectorAsMap();
203                List<ProxyConnector> proxyConnectors = proxyConnectorsMap.get( repoId );
204                if ( proxyConnectors != null )
205                {
206                    for ( ProxyConnector proxyConnector : proxyConnectors )
207                    {
208                        RemoteRepository remoteRepoConfig =
209                            remoteRepositoryAdmin.getRemoteRepository( proxyConnector.getTargetRepoId() );
210    
211                        if ( remoteRepoConfig != null )
212                        {
213                            remoteRepositories.add( remoteRepoConfig );
214    
215                            NetworkProxy networkProxyConfig =
216                                networkProxyAdmin.getNetworkProxy( proxyConnector.getProxyId() );
217    
218                            if ( networkProxyConfig != null )
219                            {
220                                // key/value: remote repo ID/proxy info
221                                networkProxies.put( proxyConnector.getTargetRepoId(), networkProxyConfig );
222                            }
223                        }
224                    }
225                }
226    
227                ModelBuildingRequest req =
228                    new DefaultModelBuildingRequest().setProcessPlugins( false ).setPomFile( file ).setTwoPhaseBuilding(
229                        false ).setValidationLevel( ModelBuildingRequest.VALIDATION_LEVEL_MINIMAL );
230    
231                //MRM-1607. olamy this will resolve jdk profiles on the current running archiva jvm
232                req.setSystemProperties( System.getProperties() );
233    
234                // MRM-1411
235                req.setModelResolver(
236                    new RepositoryModelResolver( managedRepository, pathTranslator, wagonFactory, remoteRepositories,
237                                                 networkProxies, managedRepository ) );
238    
239                Model model;
240                try
241                {
242                    model = builder.build( req ).getEffectiveModel();
243                }
244                catch ( ModelBuildingException e )
245                {
246                    String msg = "The artifact's POM file '" + file + "' was invalid: " + e.getMessage();
247    
248                    List<ModelProblem> modelProblems = e.getProblems();
249                    for ( ModelProblem problem : modelProblems )
250                    {
251                        // MRM-1411, related to MRM-1335
252                        // this means that the problem was that the parent wasn't resolved!
253                        // olamy really hackhish but fail with java profile so use error message
254                        // || ( StringUtils.startsWith( problem.getMessage(), "Failed to determine Java version for profile" ) )
255                        // but setTwoPhaseBuilding(true) fix that
256                        if ( ( problem.getException() instanceof FileNotFoundException && e.getModelId() != null &&
257                            !e.getModelId().equals( problem.getModelId() ) ) )
258                        {
259                            log.warn( "The artifact's parent POM file '" + file + "' cannot be resolved. " +
260                                          "Using defaults for project version metadata.." );
261    
262                            ProjectVersionMetadata metadata = new ProjectVersionMetadata();
263                            metadata.setId( projectVersion );
264    
265                            MavenProjectFacet facet = new MavenProjectFacet();
266                            facet.setGroupId( namespace );
267                            facet.setArtifactId( projectId );
268                            facet.setPackaging( "jar" );
269                            metadata.addFacet( facet );
270    
271                            String errMsg =
272                                "Error in resolving artifact's parent POM file. " + ( problem.getException() == null
273                                    ? problem.getMessage()
274                                    : problem.getException().getMessage() );
275                            RepositoryProblemFacet repoProblemFacet = new RepositoryProblemFacet();
276                            repoProblemFacet.setRepositoryId( repoId );
277                            repoProblemFacet.setId( repoId );
278                            repoProblemFacet.setMessage( errMsg );
279                            repoProblemFacet.setProblem( errMsg );
280                            repoProblemFacet.setProject( projectId );
281                            repoProblemFacet.setVersion( projectVersion );
282                            repoProblemFacet.setNamespace( namespace );
283    
284                            metadata.addFacet( repoProblemFacet );
285    
286                            return metadata;
287                        }
288                    }
289    
290                    throw new RepositoryStorageMetadataInvalidException( "invalid-pom", msg, e );
291                }
292    
293                // Check if the POM is in the correct location
294                boolean correctGroupId = namespace.equals( model.getGroupId() );
295                boolean correctArtifactId = projectId.equals( model.getArtifactId() );
296                boolean correctVersion = projectVersion.equals( model.getVersion() );
297                if ( !correctGroupId || !correctArtifactId || !correctVersion )
298                {
299                    StringBuilder message = new StringBuilder( "Incorrect POM coordinates in '" + file + "':" );
300                    if ( !correctGroupId )
301                    {
302                        message.append( "\nIncorrect group ID: " ).append( model.getGroupId() );
303                    }
304                    if ( !correctArtifactId )
305                    {
306                        message.append( "\nIncorrect artifact ID: " ).append( model.getArtifactId() );
307                    }
308                    if ( !correctVersion )
309                    {
310                        message.append( "\nIncorrect version: " ).append( model.getVersion() );
311                    }
312    
313                    throw new RepositoryStorageMetadataInvalidException( "mislocated-pom", message.toString() );
314                }
315    
316                ProjectVersionMetadata metadata = new ProjectVersionMetadata();
317                metadata.setCiManagement( convertCiManagement( model.getCiManagement() ) );
318                metadata.setDescription( model.getDescription() );
319                metadata.setId( projectVersion );
320                metadata.setIssueManagement( convertIssueManagement( model.getIssueManagement() ) );
321                metadata.setLicenses( convertLicenses( model.getLicenses() ) );
322                metadata.setMailingLists( convertMailingLists( model.getMailingLists() ) );
323                metadata.setDependencies( convertDependencies( model.getDependencies() ) );
324                metadata.setName( model.getName() );
325                metadata.setOrganization( convertOrganization( model.getOrganization() ) );
326                metadata.setScm( convertScm( model.getScm() ) );
327                metadata.setUrl( model.getUrl() );
328    
329                MavenProjectFacet facet = new MavenProjectFacet();
330                facet.setGroupId( model.getGroupId() != null ? model.getGroupId() : model.getParent().getGroupId() );
331                facet.setArtifactId( model.getArtifactId() );
332                facet.setPackaging( model.getPackaging() );
333                if ( model.getParent() != null )
334                {
335                    MavenProjectParent parent = new MavenProjectParent();
336                    parent.setGroupId( model.getParent().getGroupId() );
337                    parent.setArtifactId( model.getParent().getArtifactId() );
338                    parent.setVersion( model.getParent().getVersion() );
339                    facet.setParent( parent );
340                }
341                metadata.addFacet( facet );
342    
343                return metadata;
344            }
345            catch ( RepositoryAdminException e )
346            {
347                throw new RepositoryStorageRuntimeException( "repo-admin", e.getMessage() );
348            }
349        }

356        private List<org.apache.archiva.metadata.model.Dependency> convertDependencies( List<Dependency> dependencies )
357        {
358            List<org.apache.archiva.metadata.model.Dependency> l =
359                new ArrayList<org.apache.archiva.metadata.model.Dependency>();
360            for ( Dependency dependency : dependencies )
361            {
362                org.apache.archiva.metadata.model.Dependency newDependency =
363                    new org.apache.archiva.metadata.model.Dependency();
364                newDependency.setArtifactId( dependency.getArtifactId() );
365                newDependency.setClassifier( dependency.getClassifier() );
366                newDependency.setGroupId( dependency.getGroupId() );
367                newDependency.setOptional( dependency.isOptional() );
368                newDependency.setScope( dependency.getScope() );
369                newDependency.setSystemPath( dependency.getSystemPath() );
370                newDependency.setType( dependency.getType() );
371                newDependency.setVersion( dependency.getVersion() );
372                l.add( newDependency );
373            }
374            return l;
375        }

466        private static Collection<String> getSortedFiles( File dir, Filter<String> filter )
467        {
468            List<String> fileNames;
469            String[] files = dir.list( new DirectoryFilter( filter ) );
470            if ( files != null )
471            {
472                fileNames = new ArrayList<String>( Arrays.asList( files ) );
473                Collections.sort( fileNames );
474            }
475            else
476            {
477                fileNames = Collections.emptyList();
478            }
479            return fileNames;
480        }

497        public Collection<String> listNamespaces( String repoId, String namespace, Filter<String> filter )
498            throws RepositoryStorageRuntimeException
499        {
500            File dir = pathTranslator.toFile( getRepositoryBasedir( repoId ), namespace );
501    
502            // scan all the directories which are potential namespaces. Any directories known to be projects are excluded
503            List<String> namespaces = new ArrayList<String>();
504            File[] files = dir.listFiles( new DirectoryFilter( filter ) );
505            if ( files != null )
506            {
507                for ( File file : files )
508                {
509                    if ( !isProject( file, filter ) )
510                    {
511                        namespaces.add( file.getName() );
512                    }
513                }
514            }
515            Collections.sort( namespaces );
516            return namespaces;
517        }

519        public Collection<String> listProjects( String repoId, String namespace, Filter<String> filter )
520            throws RepositoryStorageRuntimeException
521        {
522            File dir = pathTranslator.toFile( getRepositoryBasedir( repoId ), namespace );
523    
524            // scan all directories in the namespace, and only include those that are known to be projects
525            List<String> projects = new ArrayList<String>();
526            File[] files = dir.listFiles( new DirectoryFilter( filter ) );
527            if ( files != null )
528            {
529                for ( File file : files )
530                {
531                    if ( isProject( file, filter ) )
532                    {
533                        projects.add( file.getName() );
534                    }
535                }
536            }
537            Collections.sort( projects );
538            return projects;
539        }

551        public Collection<ArtifactMetadata> readArtifactsMetadata( String repoId, String namespace, String projectId,
552                                                                   String projectVersion, Filter<String> filter )
553            throws RepositoryStorageRuntimeException
554        {
555            File dir = pathTranslator.toFile( getRepositoryBasedir( repoId ), namespace, projectId, projectVersion );
556    
557            // all files that are not metadata and not a checksum / signature are considered artifacts
558            File[] files = dir.listFiles( new ArtifactDirectoryFilter( filter ) );
559    
560            List<ArtifactMetadata> artifacts = new ArrayList<ArtifactMetadata>();
561            if ( files != null )
562            {
563                for ( File file : files )
564                {
565                    ArtifactMetadata metadata = getArtifactFromFile( repoId, namespace, projectId, projectVersion, file );
566                    artifacts.add( metadata );
567                }
568            }
569            return artifacts;
570        }

582        private ArtifactMetadata getArtifactFromFile( String repoId, String namespace, String projectId,
583                                                      String projectVersion, File file )
584        {
585            ArtifactMetadata metadata =
586                pathTranslator.getArtifactFromId( repoId, namespace, projectId, projectVersion, file.getName() );
587    
588            populateArtifactMetadataFromFile( metadata, file );
589    
590            return metadata;
591        }
592    
593        private static void populateArtifactMetadataFromFile( ArtifactMetadata metadata, File file )
594        {
595            metadata.setWhenGathered( new Date() );
596            metadata.setFileLastModified( file.lastModified() );
597            ChecksummedFile checksummedFile = new ChecksummedFile( file );
598            try
599            {
600                metadata.setMd5( checksummedFile.calculateChecksum( ChecksumAlgorithm.MD5 ) );
601            }
602            catch ( IOException e )
603            {
604                log.error( "Unable to checksum file " + file + ": " + e.getMessage() );
605            }
606            try
607            {
608                metadata.setSha1( checksummedFile.calculateChecksum( ChecksumAlgorithm.SHA1 ) );
609            }
610            catch ( IOException e )
611            {
612                log.error( "Unable to checksum file " + file + ": " + e.getMessage() );
613            }
614            metadata.setSize( file.length() );
615        }

617        private boolean isProject( File dir, Filter<String> filter )
618        {
619            // scan directories for a valid project version subdirectory, meaning this must be a project directory
620            File[] files = dir.listFiles( new DirectoryFilter( filter ) );
621            if ( files != null )
622            {
623                for ( File file : files )
624                {
625                    if ( isProjectVersion( file ) )
626                    {
627                        return true;
628                    }
629                }
630            }
631    
632            // if a metadata file is present, check if this is the "artifactId" directory, marking it as a project
633            ArchivaRepositoryMetadata metadata = readMetadata( dir );
634            if ( metadata != null && dir.getName().equals( metadata.getArtifactId() ) )
635            {
636                return true;
637            }
638    
639            return false;
640        }

642        private boolean isProjectVersion( File dir )
643        {
644            final String artifactId = dir.getParentFile().getName();
645            final String projectVersion = dir.getName();
646    
647            // check if there is a POM artifact file to ensure it is a version directory
648            File[] files;
649            if ( VersionUtil.isSnapshot( projectVersion ) )
650            {
651                files = dir.listFiles( new PomFilenameFilter( artifactId, projectVersion ) );
652            }
653            else
654            {
655                final String pomFile = artifactId + "-" + projectVersion + ".pom";
656                files = dir.listFiles( new PomFileFilter( pomFile ) );
657            }
658            if ( files != null && files.length > 0 )
659            {
660                return true;
661            }
662    
663            // if a metadata file is present, check if this is the "version" directory, marking it as a project version
664            ArchivaRepositoryMetadata metadata = readMetadata( dir );
665            if ( metadata != null && projectVersion.equals( metadata.getVersion() ) )
666            {
667                return true;
668            }
669    
670            return false;
671        }
672    
673        private ArchivaRepositoryMetadata readMetadata( File directory )
674        {
675            ArchivaRepositoryMetadata metadata = null;
676            File metadataFile = new File( directory, METADATA_FILENAME );
677            if ( metadataFile.exists() )
678            {
679                try
680                {
681                    metadata = MavenMetadataReader.read( metadataFile );
682                }
683                catch ( XMLException e )
684                {
685                    // ignore missing or invalid metadata
686                }
687            }
688            return metadata;
689        }

691        private static class DirectoryFilter
692            implements FilenameFilter
693        {
694            private final Filter<String> filter;
695    
696            public DirectoryFilter( Filter<String> filter )
697            {
698                this.filter = filter;
699            }
700    
701            public boolean accept( File dir, String name )
702            {
703                if ( !filter.accept( name ) )
704                {
705                    return false;
706                }
707                else if ( name.startsWith( "." ) )
708                {
709                    return false;
710                }
711                else if ( !new File( dir, name ).isDirectory() )
712                {
713                    return false;
714                }
715                return true;
716            }
717        }

719        private class ArtifactDirectoryFilter
720            implements FilenameFilter
721        {
722            private final Filter<String> filter;
723    
724            public ArtifactDirectoryFilter( Filter<String> filter )
725            {
726                this.filter = filter;
727            }
728    
729            public boolean accept( File dir, String name )
730            {
731                // TODO compare to logic in maven-repository-layer
732                if ( !filter.accept( name ) )
733                {
734                    return false;
735                }
736                else if ( name.startsWith( "." ) )
737                {
738                    return false;
739                }
740                else if ( name.endsWith( ".md5" ) || name.endsWith( ".sha1" ) || name.endsWith( ".asc" ) )
741                {
742                    return false;
743                }
744                else if ( name.equals( METADATA_FILENAME ) )
745                {
746                    return false;
747                }
748                else if ( new File( dir, name ).isDirectory() )
749                {
750                    return false;
751                }
752                // some files from remote repositories can have name like maven-metadata-archiva-vm-all-public.xml
753                else if ( StringUtils.startsWith( name, METADATA_FILENAME_START ) && StringUtils.endsWith( name, ".xml" ) )
754                {
755                    return false;
756                }
757    
758                return true;
759    
760            }
761        }

763        private static class PomFilenameFilter
764            implements FilenameFilter
765        {
766    
767            private final String artifactId, projectVersion;
768    
769            private PomFilenameFilter( String artifactId, String projectVersion )
770            {
771                this.artifactId = artifactId;
772                this.projectVersion = projectVersion;
773            }
774    
775            public boolean accept( File dir, String name )
776            {
777                if ( name.startsWith( artifactId + "-" ) && name.endsWith( ".pom" ) )
778                {
779                    String v = name.substring( artifactId.length() + 1, name.length() - 4 );
780                    v = VersionUtil.getBaseVersion( v );
781                    if ( v.equals( projectVersion ) )
782                    {
783                        return true;
784                    }
785                }
786                return false;
787            }
788        }
