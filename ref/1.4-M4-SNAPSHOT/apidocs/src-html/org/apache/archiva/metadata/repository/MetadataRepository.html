022    import org.apache.archiva.metadata.model.ArtifactMetadata;
023    import org.apache.archiva.metadata.model.MetadataFacet;
024    import org.apache.archiva.metadata.model.ProjectMetadata;
025    import org.apache.archiva.metadata.model.ProjectVersionMetadata;
026    import org.apache.archiva.metadata.model.ProjectVersionReference;
027    
028    import java.util.Collection;
029    import java.util.Date;
030    import java.util.List;
031    
032    public interface MetadataRepository
033    {
034        /**
035         * Update metadata for a particular project in the metadata repository, or create it if it does not already exist.
036         *
037         * @param repositoryId the repository the project is in
038         * @param project      the project metadata to create or update
039         */
040        void updateProject( String repositoryId, ProjectMetadata project )
041            throws MetadataRepositoryException;
042    
043        void updateArtifact( String repositoryId, String namespace, String projectId, String projectVersion,
044                             ArtifactMetadata artifactMeta )
045            throws MetadataRepositoryException;
046    
047        void updateProjectVersion( String repositoryId, String namespace, String projectId,
048                                   ProjectVersionMetadata versionMetadata )
049            throws MetadataRepositoryException;
050    
051        void updateNamespace( String repositoryId, String namespace )
052            throws MetadataRepositoryException;
053    
054        List<String> getMetadataFacets( String repositoryId, String facetId )
055            throws MetadataRepositoryException;
056    
057        MetadataFacet getMetadataFacet( String repositoryId, String facetId, String name )
058            throws MetadataRepositoryException;
059    
060        void addMetadataFacet( String repositoryId, MetadataFacet metadataFacet )
061            throws MetadataRepositoryException;
062    
063        void removeMetadataFacets( String repositoryId, String facetId )
064            throws MetadataRepositoryException;
065    
066        void removeMetadataFacet( String repositoryId, String facetId, String name )
067            throws MetadataRepositoryException;
068    
069        List<ArtifactMetadata> getArtifactsByDateRange( String repositoryId, Date startTime, Date endTime )
070            throws MetadataRepositoryException;
071    
072        // TODO: remove from API, just use configuration
073        Collection<String> getRepositories()
074            throws MetadataRepositoryException;
075    
076        List<ArtifactMetadata> getArtifactsByChecksum( String repositoryId, String checksum )
077            throws MetadataRepositoryException;
078    
079        void removeArtifact( String repositoryId, String namespace, String project, String version, String id )
080            throws MetadataRepositoryException;
081    
082        /**
083         * used for deleting timestamped version of SNAPSHOT artifacts
084         *
085         * @param artifactMetadata the artifactMetadata with the timestamped version (2.0-20120618.214135-2)
086         * @param baseVersion      the base version of the snapshot (2.0-SNAPSHOT)
087         * @throws MetadataRepositoryException
088         * @since 1.4-M3
089         */
090        void removeArtifact( ArtifactMetadata artifactMetadata, String baseVersion )
091            throws MetadataRepositoryException;
092    
093        /**
094         * @param repositoryId
095         * @param namespace
096         * @param project
097         * @param version
098         * @param metadataFacet will remove artifacts which have this {@link MetadataFacet} using equals
099         * @throws MetadataRepositoryException
100         * @since 1.4-M3
101         */
102        void removeArtifact( String repositoryId, String namespace, String project, String version,
103                             MetadataFacet metadataFacet )
104            throws MetadataRepositoryException;
105    
106        /**
107         * Delete a repository's metadata. This includes all associated metadata facets.
108         *
109         * @param repositoryId the repository to delete
110         */
111        void removeRepository( String repositoryId )
112            throws MetadataRepositoryException;
113    
114        /**
115         * @param repositoryId
116         * @param namespace    (groupId for maven )
117         * @throws MetadataRepositoryException
118         * @since 1.4-M3
119         */
120        void removeNamespace( String repositoryId, String namespace )
121            throws MetadataRepositoryException;
122    
123        List<ArtifactMetadata> getArtifacts( String repositoryId )
124            throws MetadataRepositoryException;
125    
126        ProjectMetadata getProject( String repoId, String namespace, String projectId )
127            throws MetadataResolutionException;
128    
129        ProjectVersionMetadata getProjectVersion( String repoId, String namespace, String projectId, String projectVersion )
130            throws MetadataResolutionException;
131    
132        Collection<String> getArtifactVersions( String repoId, String namespace, String projectId, String projectVersion )
133            throws MetadataResolutionException;
134    
135        /**
136         * Retrieve project references from the metadata repository. Note that this is not built into the content model for
137         * a project version as a reference may be present (due to reverse-lookup of dependencies) before the actual
138         * project is, and we want to avoid adding a stub model to the content repository.
139         *
140         * @param repoId         the repository ID to look within
141         * @param namespace      the namespace of the project to get references to
142         * @param projectId      the identifier of the project to get references to
143         * @param projectVersion the version of the project to get references to
144         * @return a list of project references
145         */
146        Collection<ProjectVersionReference> getProjectReferences( String repoId, String namespace, String projectId,
147                                                                  String projectVersion )
148            throws MetadataResolutionException;
149    
150        Collection<String> getRootNamespaces( String repoId )
151            throws MetadataResolutionException;
152    
153        Collection<String> getNamespaces( String repoId, String namespace )
154            throws MetadataResolutionException;
155    
156        Collection<String> getProjects( String repoId, String namespace )
157            throws MetadataResolutionException;
158    
159        Collection<String> getProjectVersions( String repoId, String namespace, String projectId )
160            throws MetadataResolutionException;
161    
162        /**
163         * @param repoId
164         * @param namespace
165         * @param projectId
166         * @param projectVersion
167         * @throws MetadataResolutionException
168         * @since 1.4-M4
169         */
170        void removeProjectVersion( String repoId, String namespace, String projectId, String projectVersion )
171            throws MetadataRepositoryException;
172    
173        Collection<ArtifactMetadata> getArtifacts( String repoId, String namespace, String projectId,
174                                                   String projectVersion )
175            throws MetadataResolutionException;
176    
177        /**
178         * remove a project
179         * @param repositoryId
180         * @param namespace
181         * @param projectId
182         * @throws MetadataRepositoryException
183         * @since 1.4-M4
184         */
185        void removeProject( String repositoryId, String namespace, String projectId )
186            throws MetadataRepositoryException;
187    
188    
189        void save();
190    
191        void close()
192            throws MetadataRepositoryException;
193    
194        void revert();
195    
196        boolean canObtainAccess( Class<?> aClass );
197    
198        Object obtainAccess( Class<?> aClass )
199            throws MetadataRepositoryException;
200    }
