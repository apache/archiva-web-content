021    import javax.xml.bind.annotation.XmlRootElement;
022    import java.io.Serializable;
023    import java.util.List;
024    
025    /**
026     * @author Olivier Lamy
027     * @since 1.4-M1
028     */
029    @XmlRootElement( name = "groupIdList" )
030    public class GroupIdList
031        implements Serializable
032    {
033        private List<String> groupIds;
034    
035        public GroupIdList()
036        {
037            // no op
038        }
039    
040        public GroupIdList( List<String> groupIds )
041        {
042            this.groupIds = groupIds;
043        }
044    
045        public List<String> getGroupIds()
046        {
047            return groupIds;
048        }
049    
050        public void setGroupIds( List<String> groupIds )
051        {
052            this.groupIds = groupIds;
053        }
054    }
