042    private Configuration readConfiguration( String prefix, Registry registry )
043    {
044        Configuration value = new Configuration();
045
046        String version = registry.getString( prefix + "version", value.getVersion() );
047        value.setVersion( version );
048        java.util.List repositories = new java.util.ArrayList/*<V1RepositoryConfiguration>*/();

107        value.setWebapp( webapp );

111        value.setNetworkConfiguration( networkConfiguration );
