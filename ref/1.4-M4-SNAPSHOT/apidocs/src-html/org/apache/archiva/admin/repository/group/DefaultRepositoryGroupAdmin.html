048@Service("repositoryGroupAdmin#default")

054    private Logger log = LoggerFactory.getLogger( getClass() );

056    private static final Pattern REPO_GROUP_ID_PATTERN = Pattern.compile( "[A-Za-z0-9\\._\\-]+" );
057
058    @Inject
059    private ManagedRepositoryAdmin managedRepositoryAdmin;
060
061    public List<RepositoryGroup> getRepositoriesGroups()
062        throws RepositoryAdminException
063    {
064        List<RepositoryGroup> repositoriesGroups =
065            new ArrayList<RepositoryGroup>( getArchivaConfiguration().getConfiguration().getRepositoryGroups().size() );
066
067        for ( RepositoryGroupConfiguration repositoryGroupConfiguration : getArchivaConfiguration().getConfiguration().getRepositoryGroups() )
068        {
069            repositoriesGroups.add( new RepositoryGroup( repositoryGroupConfiguration.getId(), new ArrayList<String>(
070                repositoryGroupConfiguration.getRepositories() ) ).mergedIndexPath(
071                repositoryGroupConfiguration.getMergedIndexPath() ) );
072        }
073
074        return repositoriesGroups;
075    }
076
077    public RepositoryGroup getRepositoryGroup( String repositoryGroupId )
078        throws RepositoryAdminException
079    {
080        List<RepositoryGroup> repositoriesGroups = getRepositoriesGroups();
081        for ( RepositoryGroup repositoryGroup : repositoriesGroups )
082        {
083            if ( StringUtils.equals( repositoryGroupId, repositoryGroup.getId() ) )
084            {
085                return repositoryGroup;
086            }
087        }
088        return null;
089    }
090
091    public Boolean addRepositoryGroup( RepositoryGroup repositoryGroup, AuditInformation auditInformation )
092        throws RepositoryAdminException
093    {
094        validateRepositoryGroup( repositoryGroup, false );
095        validateManagedRepositoriesExists( repositoryGroup.getRepositories() );
096        RepositoryGroupConfiguration repositoryGroupConfiguration = new RepositoryGroupConfiguration();
097        repositoryGroupConfiguration.setId( repositoryGroup.getId() );
098        repositoryGroupConfiguration.setRepositories( repositoryGroup.getRepositories() );
099        repositoryGroupConfiguration.setMergedIndexPath( repositoryGroup.getMergedIndexPath() );
100        Configuration configuration = getArchivaConfiguration().getConfiguration();
101        configuration.addRepositoryGroup( repositoryGroupConfiguration );
102        saveConfiguration( configuration );
103        triggerAuditEvent( repositoryGroup.getId(), null, AuditEvent.ADD_REPO_GROUP, auditInformation );
104        return Boolean.TRUE;
105    }
106
107    public Boolean deleteRepositoryGroup( String repositoryGroupId, AuditInformation auditInformation )
108        throws RepositoryAdminException
109    {
110        Configuration configuration = getArchivaConfiguration().getConfiguration();
111        RepositoryGroupConfiguration repositoryGroupConfiguration =
112            configuration.getRepositoryGroupsAsMap().get( repositoryGroupId );
113        if ( repositoryGroupConfiguration == null )
114        {
115            throw new RepositoryAdminException(
116                "repositoryGroup with id " + repositoryGroupId + " doesn't not exists so cannot remove" );
117        }
118        configuration.removeRepositoryGroup( repositoryGroupConfiguration );
119        triggerAuditEvent( repositoryGroupId, null, AuditEvent.DELETE_REPO_GROUP, auditInformation );
120        return Boolean.TRUE;
121    }
122
123    public Boolean updateRepositoryGroup( RepositoryGroup repositoryGroup, AuditInformation auditInformation )
124        throws RepositoryAdminException
125    {
126        return updateRepositoryGroup( repositoryGroup, auditInformation, true );
127    }
128
129    private Boolean updateRepositoryGroup( RepositoryGroup repositoryGroup, AuditInformation auditInformation,
130                                           boolean triggerAuditEvent )
131        throws RepositoryAdminException
132    {
133        validateRepositoryGroup( repositoryGroup, true );
134        validateManagedRepositoriesExists( repositoryGroup.getRepositories() );
135        Configuration configuration = getArchivaConfiguration().getConfiguration();
136
137        RepositoryGroupConfiguration repositoryGroupConfiguration =
138            configuration.getRepositoryGroupsAsMap().get( repositoryGroup.getId() );
139
140        configuration.removeRepositoryGroup( repositoryGroupConfiguration );
141
142        repositoryGroupConfiguration.setRepositories( repositoryGroup.getRepositories() );
143        repositoryGroupConfiguration.setMergedIndexPath( repositoryGroup.getMergedIndexPath() );
144        configuration.addRepositoryGroup( repositoryGroupConfiguration );
145
146        saveConfiguration( configuration );
147        if ( triggerAuditEvent )
148        {
149            triggerAuditEvent( repositoryGroup.getId(), null, AuditEvent.MODIFY_REPO_GROUP, auditInformation );
150        }
151        return Boolean.TRUE;
152    }

155    public Boolean addRepositoryToGroup( String repositoryGroupId, String repositoryId,
156                                         AuditInformation auditInformation )
157        throws RepositoryAdminException
158    {
159        RepositoryGroup repositoryGroup = getRepositoryGroup( repositoryGroupId );
160        if ( repositoryGroup == null )
161        {
162            throw new RepositoryAdminException(
163                "repositoryGroup with id " + repositoryGroupId + " doesn't not exists so cannot add repository to it" );
164        }
165
166        if ( repositoryGroup.getRepositories().contains( repositoryId ) )
167        {
168            throw new RepositoryAdminException(
169                "repositoryGroup with id " + repositoryGroupId + " already contain repository with id" + repositoryId );
170        }
171        validateManagedRepositoriesExists( Arrays.asList( repositoryId ) );
172
173        repositoryGroup.addRepository( repositoryId );
174        updateRepositoryGroup( repositoryGroup, auditInformation, false );
175        triggerAuditEvent( repositoryGroup.getId(), null, AuditEvent.ADD_REPO_TO_GROUP, auditInformation );
176        return Boolean.TRUE;
177    }
178
179    public Boolean deleteRepositoryFromGroup( String repositoryGroupId, String repositoryId,
180                                              AuditInformation auditInformation )
181        throws RepositoryAdminException
182    {
183        RepositoryGroup repositoryGroup = getRepositoryGroup( repositoryGroupId );
184        if ( repositoryGroup == null )
185        {
186            throw new RepositoryAdminException( "repositoryGroup with id " + repositoryGroupId
187                                                    + " doesn't not exists so cannot remove repository from it" );
188        }
189
190        if ( !repositoryGroup.getRepositories().contains( repositoryId ) )
191        {
192            throw new RepositoryAdminException(
193                "repositoryGroup with id " + repositoryGroupId + " doesn't not contains repository with id"
194                    + repositoryId );
195        }
196
197        repositoryGroup.removeRepository( repositoryId );
198        updateRepositoryGroup( repositoryGroup, auditInformation, false );
199        triggerAuditEvent( repositoryGroup.getId(), null, AuditEvent.DELETE_REPO_FROM_GROUP, auditInformation );
200        return Boolean.TRUE;
201    }

203    public Map<String, RepositoryGroup> getRepositoryGroupsAsMap()
204        throws RepositoryAdminException
205    {
206        List<RepositoryGroup> repositoriesGroups = getRepositoriesGroups();
207        Map<String, RepositoryGroup> map = new HashMap<String, RepositoryGroup>( repositoriesGroups.size() );
208        for ( RepositoryGroup repositoryGroup : repositoriesGroups )
209        {
210            map.put( repositoryGroup.getId(), repositoryGroup );
211        }
212        return map;
213    }

215    public Map<String, List<String>> getGroupToRepositoryMap()
216        throws RepositoryAdminException
217    {
218
219        java.util.Map<String, java.util.List<String>> map = new java.util.HashMap<String, java.util.List<String>>();
220
221        for ( ManagedRepository repo : getManagedRepositoryAdmin().getManagedRepositories() )
222        {
223            for ( RepositoryGroup group : getRepositoriesGroups() )
224            {
225                if ( !group.getRepositories().contains( repo.getId() ) )
226                {
227                    String groupId = group.getId();
228                    java.util.List<String> repos = map.get( groupId );
229                    if ( repos == null )
230                    {
231                        repos = new ArrayList<String>();
232                        map.put( groupId, repos );
233                    }
234                    repos.add( repo.getId() );
235                }
236            }
237        }
238        return map;
239    }

241    public Map<String, List<String>> getRepositoryToGroupMap()
242        throws RepositoryAdminException
243    {
244        java.util.Map<String, java.util.List<String>> map = new java.util.HashMap<String, java.util.List<String>>();
245
246        for ( RepositoryGroup group : getRepositoriesGroups() )
247        {
248            for ( String repositoryId : group.getRepositories() )
249            {
250                java.util.List<String> groups = map.get( repositoryId );
251                if ( groups == null )
252                {
253                    groups = new ArrayList<String>();
254                    map.put( repositoryId, groups );
255                }
256                groups.add( group.getId() );
257            }
258        }
259        return map;
260    }

262    public Boolean validateRepositoryGroup( RepositoryGroup repositoryGroup, boolean updateMode )
263        throws RepositoryAdminException
264    {
265        String repoGroupId = repositoryGroup.getId();
266        if ( StringUtils.isBlank( repoGroupId ) )
267        {
268            throw new RepositoryAdminException( "repositoryGroup id cannot be empty" );
269        }
270
271        if ( repoGroupId.length() > 100 )
272        {
273            throw new RepositoryAdminException(
274                "Identifier [" + repoGroupId + "] is over the maximum limit of 100 characters" );
275
276        }
277
278        Matcher matcher = REPO_GROUP_ID_PATTERN.matcher( repoGroupId );
279        if ( !matcher.matches() )
280        {
281            throw new RepositoryAdminException(
282                "Invalid character(s) found in identifier. Only the following characters are allowed: alphanumeric, '.', '-' and '_'" );
283        }
284
285        Configuration configuration = getArchivaConfiguration().getConfiguration();
286
287        if ( configuration.getRepositoryGroupsAsMap().containsKey( repoGroupId ) )
288        {
289            if ( !updateMode )
290            {
291                throw new RepositoryAdminException( "Unable to add new repository group with id [" + repoGroupId
292                                                        + "], that id already exists as a repository group." );
293            }
294        }
295        else if ( configuration.getManagedRepositoriesAsMap().containsKey( repoGroupId ) )
296        {
297            throw new RepositoryAdminException( "Unable to add new repository group with id [" + repoGroupId
298                                                    + "], that id already exists as a managed repository." );
299        }
300        else if ( configuration.getRemoteRepositoriesAsMap().containsKey( repoGroupId ) )
301        {
302            throw new RepositoryAdminException( "Unable to add new repository group with id [" + repoGroupId
303                                                    + "], that id already exists as a remote repository." );
304        }
305
306        return Boolean.TRUE;
307    }
308
309    private void validateManagedRepositoriesExists( List<String> managedRepositoriesIds )
310        throws RepositoryAdminException
311    {
312        for ( String id : managedRepositoriesIds )
313        {
314            if ( getManagedRepositoryAdmin().getManagedRepository( id ) == null )
315            {
316                throw new RepositoryAdminException(
317                    "managedRepository with id " + id + " not exists so cannot be used in a repositoryGroup" );
318            }
319        }
320    }
