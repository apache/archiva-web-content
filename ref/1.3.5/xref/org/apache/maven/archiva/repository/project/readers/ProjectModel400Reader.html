57      public ArchivaProjectModel read( File pomFile )
58          throws XMLException
59      {
60          XMLReader xml = new XMLReader( "project", pomFile );
61  
62          ArchivaProjectModel model = new ArchivaProjectModel();
63  
64          if ( !"http://maven.apache.org/POM/4.0.0".equals( xml.getDefaultNamespaceURI() ) )
65          {
66              // No namespace defined
67              // TODO: Output to monitor the problem with the Namespace.
68          }
69  
70          xml.removeNamespaces();
71  
72          Element project = xml.getElement( "//project" );
73  
74          model.setGroupId( project.elementTextTrim( "groupId" ) );
75          model.setArtifactId( project.elementTextTrim( "artifactId" ) );
76          model.setVersion( project.elementTextTrim( "version" ) );
77          model.setName( project.elementTextTrim( "name" ) );
78          model.setDescription( project.elementTextTrim( "description" ) );
79          model.setUrl( project.elementTextTrim( "url" ) );
80  
81          model.setPackaging( StringUtils.defaultIfEmpty( project.elementTextTrim( "packaging" ), "jar" ) );
82  
83          model.setParentProject( getParentProject( xml ) );
84  
85          model.setMailingLists( getMailingLists( xml ) );
86          model.setCiManagement( getCiManagement( xml ) );
87          model.setIndividuals( getIndividuals( xml ) );
88          model.setIssueManagement( getIssueManagement( xml ) );
89          model.setLicenses( getLicenses( xml ) );
90          model.setOrganization( getOrganization( xml ) );
91          model.setScm( getSCM( xml ) );
92          model.setRepositories( getRepositories( xml ) );
93  
94          model.setDependencies( getDependencies( xml ) );
95          model.setDependencyManagement( getDependencyManagement( xml ) );
96          model.setPlugins( getPlugins( xml ) );
97          model.setReports( getReports( xml ) );
98          model.setProperties( getProperties( xml.getElement( "//project/properties" ) ) );
99  
100         model.setBuildExtensions( getBuildExtensions( xml ) );
101 
102         model.setRelocation( getRelocation( xml ) );
103 
104         model.setOrigin( "filesystem" );
105 
106         return model;
107     }

169     private List<Dependency> getDependencyList( XMLReader xml, String parts[] )
170         throws XMLException
171     {
172         List<Dependency> dependencyList = new ArrayList<Dependency>();
173 
174         Element project = xml.getElement( "//project" );
175 
176         Element depsParent = project;
177 
178         for ( String part : parts )
179         {
180             depsParent = depsParent.element( part );
181             if ( depsParent == null )
182             {
183                 return dependencyList;
184             }
185         }
186 
187         Iterator<Element> it = depsParent.elementIterator( "dependency" );
188         while ( it.hasNext() )
189         {
190             Element elemDependency = it.next();
191             Dependency dependency = new Dependency();
192 
193             dependency.setGroupId( elemDependency.elementTextTrim( "groupId" ) );
194             dependency.setArtifactId( elemDependency.elementTextTrim( "artifactId" ) );
195             dependency.setVersion( elemDependency.elementTextTrim( "version" ) );
196 
197             dependency.setClassifier( StringUtils.defaultString( elemDependency.elementTextTrim( "classifier" ) ) );
198             dependency.setType( StringUtils.defaultIfEmpty( elemDependency.elementTextTrim( "type" ), "jar" ) );
199             dependency.setScope( StringUtils.defaultIfEmpty( elemDependency.elementTextTrim( "scope" ), "compile" ) );
200             // Not for v4.0.0 -> dependency.setUrl( elemDependency.elementTextTrim("url") );
201             dependency.setOptional( toBoolean( elemDependency.elementTextTrim( "optional" ), false ) );
202             if ( DependencyScope.isSystemScoped( dependency ) )
203             {
204                 dependency.setSystemPath( elemDependency.elementTextTrim( "systemPath" ) );
205             }
206 
207             dependency.setExclusions( getExclusions( elemDependency ) );
208 
209             if ( dependencyList.contains( dependency ) )
210             {
211                 // TODO: throw into monitor as "duplicate dependency" issue.
212             }
213 
214             dependencyList.add( dependency );
215         }
216 
217         return dependencyList;
218     }

261     private List<Individual> getIndividuals( XMLReader xml, boolean isCommitor, String xpathExpr )
262         throws XMLException
263     {
264         List<Individual> ret = new ArrayList<Individual>();
265 
266         List<Element> modelPersonList = xml.getElementList( xpathExpr );
267 
268         Iterator<Element> iter = modelPersonList.iterator();
269         while ( iter.hasNext() )
270         {
271             Element elemPerson = iter.next();
272             Individual individual = new Individual();
273 
274             if ( isCommitor )
275             {
276                 individual.setPrincipal( elemPerson.elementTextTrim( "id" ) );
277             }
278 
279             individual.setCommitor( isCommitor );
280             individual.setEmail( elemPerson.elementTextTrim( "email" ) );
281             individual.setName( elemPerson.elementTextTrim( "name" ) );
282             individual.setOrganization( elemPerson.elementTextTrim( "organization" ) );
283             individual.setOrganizationUrl( elemPerson.elementTextTrim( "organizationUrl" ) );
284             individual.setUrl( elemPerson.elementTextTrim( "url" ) );
285             individual.setTimezone( elemPerson.elementTextTrim( "timezone" ) );
286             individual.setIndividualEmail( elemPerson.elementTextTrim( "email" ) );
287 
288             // Roles
289             Element elemRoles = elemPerson.element( "roles" );
290             if ( elemRoles != null )
291             {
292                 List<Element> roleNames = elemRoles.elements( "role" );
293                 Iterator<Element> itRole = roleNames.iterator();
294                 while ( itRole.hasNext() )
295                 {
296                     Element role = itRole.next();
297                     individual.addRole( role.getTextTrim() );
298                 }
299             }
300 
301             // Properties
302             individual.setProperties( getProperties( elemPerson.element( "properties" ) ) );
303 
304             ret.add( individual );
305         }
306 
307         return ret;
308     }

367             mlist.setUnsubscribeAddress( elemMailingList.elementTextTrim( "unsubscribe" ) );
368             mlist.setPostAddress( elemMailingList.elementTextTrim( "post" ) );
369             mlist.setMainArchiveUrl( elemMailingList.elementTextTrim( "archive" ) );

371             Element elemOtherArchives = elemMailingList.element( "otherArchives" );
372             if ( elemOtherArchives != null )
373             {

375                 List<Element> others = elemOtherArchives.elements( "otherArchive" );

533     private boolean toBoolean( String value, boolean defaultValue )
534     {
535         if ( StringUtils.equalsIgnoreCase( value, "true" ) )
536         {
537             return true;
538         }
539         else if ( StringUtils.equalsIgnoreCase( value, "false" ) )
540         {
541             return false;
542         }
543         else
544         {
545             // If unset, or not "true" or "false".
