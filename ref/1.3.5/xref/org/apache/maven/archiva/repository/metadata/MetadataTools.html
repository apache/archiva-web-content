128     /**
129      * Gather the set of snapshot versions found in a particular versioned reference.
130      *
131      * @return the Set of snapshot artifact versions found.
132      * @throws LayoutException
133      * @throws ContentNotFoundException 
134      */
135     public Set<String> gatherSnapshotVersions( ManagedRepositoryContent managedRepository, VersionedReference reference )
136         throws LayoutException, IOException, ContentNotFoundException
137     {
138         Set<String> foundVersions = managedRepository.getVersions( reference );
139 
140         // Next gather up the referenced 'latest' versions found in any proxied repositories
141         // maven-metadata-${proxyId}.xml files that may be present.
142 
143         // Does this repository have a set of remote proxied repositories?
144         Set<String> proxiedRepoIds = this.proxies.get( managedRepository.getId() );
145 
146         if ( CollectionUtils.isNotEmpty( proxiedRepoIds ) )
147         {
148             String baseVersion = VersionUtil.getBaseVersion( reference.getVersion() );
149             baseVersion = baseVersion.substring( 0, baseVersion.indexOf( VersionUtil.SNAPSHOT ) - 1 );
150 
151             // Add in the proxied repo version ids too.
152             Iterator<String> it = proxiedRepoIds.iterator();
153             while ( it.hasNext() )
154             {
155                 String proxyId = it.next();
156 
157                 ArchivaRepositoryMetadata proxyMetadata = readProxyMetadata( managedRepository, reference, proxyId );
158                 if ( proxyMetadata == null )
159                 {
160                     // There is no proxy metadata, skip it.
161                     continue;
162                 }
163 
164                 // Is there some snapshot info?
165                 SnapshotVersion snapshot = proxyMetadata.getSnapshotVersion();
166                 if ( snapshot != null )
167                 {
168                     String timestamp = snapshot.getTimestamp();
169                     int buildNumber = snapshot.getBuildNumber();
170 
171                     // Only interested in the timestamp + buildnumber.
172                     if ( StringUtils.isNotBlank( timestamp ) && ( buildNumber > 0 ) )
173                     {
174                         foundVersions.add( baseVersion + "-" + timestamp + "-" + buildNumber );
175                     }
176                 }
177             }
178         }
179 
180         return foundVersions;
181     }

183     /**
184      * Take a path to a maven-metadata.xml, and attempt to translate it to a VersionedReference.
185      *
186      * @param path
187      * @return
188      */
189     public VersionedReference toVersionedReference( String path )
190         throws RepositoryMetadataException
191     {
192         if ( !path.endsWith( "/" + MAVEN_METADATA ) )
193         {
194             throw new RepositoryMetadataException( "Cannot convert to versioned reference, not a metadata file. " );
195         }
196 
197         VersionedReference reference = new VersionedReference();
198 
199         String normalizedPath = StringUtils.replace( path, "\\", "/" );
200         String pathParts[] = StringUtils.split( normalizedPath, '/' );
201 
202         int versionOffset = pathParts.length - 2;
203         int artifactIdOffset = versionOffset - 1;
204         int groupIdEnd = artifactIdOffset - 1;
205 
206         reference.setVersion( pathParts[versionOffset] );
207 
208         if ( !hasNumberAnywhere( reference.getVersion() ) )
209         {
210             // Scary check, but without it, all paths are version references;
211             throw new RepositoryMetadataException(
212                                                    "Not a versioned reference, as version id on path has no number in it." );
213         }
214 
215         reference.setArtifactId( pathParts[artifactIdOffset] );
216 
217         StringBuffer gid = new StringBuffer();
218         for ( int i = 0; i <= groupIdEnd; i++ )
219         {
220             if ( i > 0 )
221             {
222                 gid.append( "." );
223             }
224             gid.append( pathParts[i] );
225         }
226 
227         reference.setGroupId( gid.toString() );
228 
229         return reference;
230     }
231 
232     private boolean hasNumberAnywhere( String version )
233     {
234         return StringUtils.indexOfAny( version, NUMS ) != ( -1 );
235     }

272     public String toPath( ProjectReference reference )
273     {
274         StringBuffer path = new StringBuffer();
275 
276         path.append( formatAsDirectory( reference.getGroupId() ) ).append( PATH_SEPARATOR );
277         path.append( reference.getArtifactId() ).append( PATH_SEPARATOR );
278         path.append( MAVEN_METADATA );
279 
280         return path.toString();
281     }

283     public String toPath( VersionedReference reference )
284     {
285         StringBuffer path = new StringBuffer();
286 
287         path.append( formatAsDirectory( reference.getGroupId() ) ).append( PATH_SEPARATOR );
288         path.append( reference.getArtifactId() ).append( PATH_SEPARATOR );
289         if ( reference.getVersion() != null )
290         {
291             // add the version only if it is present
292             path.append( VersionUtil.getBaseVersion( reference.getVersion() ) ).append( PATH_SEPARATOR );
293         }
294         path.append( MAVEN_METADATA );
295 
296         return path.toString();
297     }

324     {

326 
327         int idx = path.lastIndexOf( "/" );
328         if ( idx > 0 )
329         {
330             ret.append( path.substring( 0, idx + 1 ) );
331         }
332 
333         // TODO: need to filter out 'bad' characters from the proxy id.
334         ret.append( "maven-metadata-" ).append( proxyId ).append( ".xml" );

424     public void updateMetadata( ManagedRepositoryContent managedRepository, String logicalResource) throws RepositoryMetadataException
425     {
426         final File metadataFile = new File(managedRepository.getRepoRoot(), logicalResource);
427         ArchivaRepositoryMetadata metadata = null;
428         
429         //Gather and merge all metadata available
430         List<ArchivaRepositoryMetadata> metadatas = getMetadatasForManagedRepository(managedRepository, logicalResource);
431         for (ArchivaRepositoryMetadata proxiedMetadata : metadatas)
432         {
433             if (metadata == null)
434             {
435                 metadata = proxiedMetadata;
436                 continue;
437             }
438             metadata = RepositoryMetadataMerge.merge(metadata, proxiedMetadata);
439         }
440         
441         if (metadata == null)
442         {
443             log.debug("No metadata to update for " + logicalResource);
444             return;
445         }
446         
447         Set<String> availableVersions = new HashSet<String>();
448         List<String> metadataAvailableVersions = metadata.getAvailableVersions();
449         if (metadataAvailableVersions != null)
450         {
451             availableVersions.addAll(metadataAvailableVersions);
452         }
453         availableVersions = findPossibleVersions(availableVersions, metadataFile.getParentFile());
454 
455         if (availableVersions.size() > 0)
456         {
457             updateMetadataVersions(availableVersions, metadata);
458         }
459         
460         RepositoryMetadataWriter.write(metadata, metadataFile);
461         
462         ChecksummedFile checksum = new ChecksummedFile( metadataFile );
463         checksum.fixChecksums( algorithms );
464     }

466     /**
467      * Skims the parent directory of a metadata in vain hope of finding 
468      * subdirectories that contain poms.
469      * 
470      * @param metadataParentDirectory
471      * @return origional set plus newley found versions
472      */
473     private Set<String> findPossibleVersions(Set<String> versions, File metadataParentDirectory)
474     {
475         Set<String> result = new HashSet<String>(versions);
476         for (File directory : metadataParentDirectory.listFiles())
477         {
478             if (directory.isDirectory())
479             {
480                 for (File possiblePom : directory.listFiles())
481                 {
482                     if (possiblePom.getName().endsWith(".pom"))
483                     {
484                         result.add(directory.getName());
485                     }
486                 }
487             }
488         }
489         return result;
490     }

492     private List<ArchivaRepositoryMetadata> getMetadatasForManagedRepository( ManagedRepositoryContent managedRepository, String logicalResource )
493     {
494         List<ArchivaRepositoryMetadata> metadatas = new ArrayList<ArchivaRepositoryMetadata>();
495         File file = new File(managedRepository.getRepoRoot(), logicalResource);
496         if (file.exists())
497         {
498             try
499             {
500                ArchivaRepositoryMetadata existingMetadata = RepositoryMetadataReader.read(file);
501                if (existingMetadata != null)
502                {
503                     metadatas.add(existingMetadata);
504                }
505             }
506             catch (RepositoryMetadataException e)
507             {
508                 log.debug("Could not read metadata at " + file.getAbsolutePath() + ". Metadata will be removed.");
509                 FileUtils.deleteQuietly(file);
510             }
511         }
512         
513         Set<String> proxyIds = proxies.get(managedRepository.getId());
514         if (proxyIds != null)
515         {
516             for (String proxyId : proxyIds)
517             {
518                 ArchivaRepositoryMetadata proxyMetadata = readProxyMetadata( managedRepository, logicalResource, proxyId );
519                 if (proxyMetadata != null)
520                 {
521                     metadatas.add(proxyMetadata);
522                 }
523             }
524         }
525         
526         return metadatas;
527     }

530     /**
531      * Update the metadata to represent the all versions/plugins of
532      * the provided groupId:artifactId project or group reference,
533      * based off of information present in the repository,
534      * the maven-metadata.xml files, and the proxy/repository specific
535      * metadata file contents.
536      *
537      * We must treat this as a group or a project metadata file as there is no way to know in advance
538      *
539      * @deprecated 
540      * @param managedRepository the managed repository where the metadata is kept.
541      * @param reference         the reference to update.
542      * @throws LayoutException
543      * @throws RepositoryMetadataException
544      * @throws IOException
545      * @throws ContentNotFoundException 
546      */
547     public void updateMetadata( ManagedRepositoryContent managedRepository, ProjectReference reference )
548         throws LayoutException, RepositoryMetadataException, IOException, ContentNotFoundException
549     {
550         File metadataFile = new File( managedRepository.getRepoRoot(), toPath( reference ) );
551 
552         long lastUpdated = getExistingLastUpdated( metadataFile );
553 
554         ArchivaRepositoryMetadata metadata = new ArchivaRepositoryMetadata();
555         metadata.setGroupId( reference.getGroupId() );
556         metadata.setArtifactId( reference.getArtifactId() );
557 
558         // Gather up all versions found in the managed repository.
559         Set<String> allVersions = managedRepository.getVersions( reference );
560 
561         // Gather up all plugins found in the managed repository.
562         // TODO: do we know this information instead?
563 //        Set<Plugin> allPlugins = managedRepository.getPlugins( reference );
564         Set<Plugin> allPlugins;
565         if ( metadataFile.exists() )
566         {
567             allPlugins = new LinkedHashSet<Plugin>( RepositoryMetadataReader.read( metadataFile ).getPlugins() );
568         }
569         else
570         {
571             allPlugins = new LinkedHashSet<Plugin>();
572         }
573 
574         // Does this repository have a set of remote proxied repositories?
575         Set<String> proxiedRepoIds = this.proxies.get( managedRepository.getId() );
576 
577         if ( CollectionUtils.isNotEmpty( proxiedRepoIds ) )
578         {
579             // Add in the proxied repo version ids too.
580             Iterator<String> it = proxiedRepoIds.iterator();
581             while ( it.hasNext() )
582             {
583                 String proxyId = it.next();
584 
585                 ArchivaRepositoryMetadata proxyMetadata = readProxyMetadata( managedRepository, reference, proxyId );
586                 if ( proxyMetadata != null )
587                 {
588                     allVersions.addAll( proxyMetadata.getAvailableVersions() );
589                     allPlugins.addAll( proxyMetadata.getPlugins() );
590                     long proxyLastUpdated = getLastUpdated( proxyMetadata );
591 
592                     lastUpdated = Math.max( lastUpdated, proxyLastUpdated );
593                 }
594             }
595         }
596 
597         if ( !allVersions.isEmpty() )
598         {
599             updateMetadataVersions( allVersions ,metadata );
600         }
601         else
602         {
603             // Add the plugins to the metadata model.
604             metadata.setPlugins( new ArrayList<Plugin>( allPlugins ) );
605 
606             // artifact ID was actually the last part of the group
607             metadata.setGroupId( metadata.getGroupId() + "." + metadata.getArtifactId() );
608             metadata.setArtifactId( null );
609         }
610 
611         if ( lastUpdated > 0 )
612         {
613             metadata.setLastUpdatedTimestamp( toLastUpdatedDate( lastUpdated ) );
614         }
615 
616         // Save the metadata model to disk.
617         RepositoryMetadataWriter.write( metadata, metadataFile );
618         ChecksummedFile checksum = new ChecksummedFile( metadataFile );
619         checksum.fixChecksums( algorithms );
620     }

622     private void updateMetadataVersions(Collection<String> allVersions, ArchivaRepositoryMetadata metadata) 
623     {
624         // Sort the versions
625         List<String> sortedVersions = new ArrayList<String>(allVersions);
626         Collections.sort(sortedVersions, VersionComparator.getInstance());
627 
628         // Split the versions into released and snapshots.
629         List<String> releasedVersions = new ArrayList<String>();
630         List<String> snapshotVersions = new ArrayList<String>();
631 
632         for (String version : sortedVersions) 
633         {
634             if (VersionUtil.isSnapshot(version)) 
635             {
636                 snapshotVersions.add(version);
637             }
638             else 
639             {
640                 releasedVersions.add(version);
641             }
642         }
643 
644         Collections.sort(releasedVersions, VersionComparator.getInstance());
645         Collections.sort(snapshotVersions, VersionComparator.getInstance());
646 
647         String latestVersion = sortedVersions.get(sortedVersions.size() - 1);
648         String releaseVersion = null;
649 
650         if (CollectionUtils.isNotEmpty(releasedVersions)) 
651         {
652             releaseVersion = releasedVersions.get(releasedVersions.size() - 1);
653         }
654 
655         // Add the versions to the metadata model.
656         metadata.setAvailableVersions(sortedVersions);
657 
658         metadata.setLatestVersion(latestVersion);
659         metadata.setReleasedVersion(releaseVersion);
660     }

670     private long toLastUpdatedLong( String timestampString )
671     {
672         try
673         {
674             Date date = lastUpdatedFormat.parse( timestampString );
675             Calendar cal = Calendar.getInstance( DateUtils.UTC_TIME_ZONE );
676             cal.setTime( date );
677 
678             return cal.getTimeInMillis();
679         }
680         catch ( ParseException e )
681         {
682             return 0;
683         }
684     }

686     private long getLastUpdated( ArchivaRepositoryMetadata metadata )
687     {
688         if ( metadata == null )
689         {
690             // Doesn't exist.
691             return 0;
692         }
693 
694         try
695         {
696             String lastUpdated = metadata.getLastUpdated();
697             if ( StringUtils.isBlank( lastUpdated ) )
698             {
699                 // Not set.
700                 return 0;
701             }
702 
703             Date lastUpdatedDate = lastUpdatedFormat.parse( lastUpdated );
704             return lastUpdatedDate.getTime();
705         }
706         catch ( ParseException e )
707         {
708             // Bad format on the last updated string.
709             return 0;
710         }
711     }

713     private long getExistingLastUpdated( File metadataFile )
714     {
715         if ( !metadataFile.exists() )
716         {
717             // Doesn't exist.
718             return 0;
719         }
720 
721         try
722         {
723             ArchivaRepositoryMetadata metadata = RepositoryMetadataReader.read( metadataFile );
724 
725             return getLastUpdated( metadata );
726         }
727         catch ( RepositoryMetadataException e )
728         {
729             // Error.
730             return 0;
731         }
732     }

734     /**
735      * Update the metadata based on the following rules.
736      * <p/>
737      * 1) If this is a SNAPSHOT reference, then utilize the proxy/repository specific
738      * metadata files to represent the current / latest SNAPSHOT available.
739      * 2) If this is a RELEASE reference, and the metadata file does not exist, then
740      * create the metadata file with contents required of the VersionedReference
741      *
742      * @deprecated
743      * @param managedRepository the managed repository where the metadata is kept.
744      * @param reference         the versioned reference to update
745      * @throws LayoutException
746      * @throws RepositoryMetadataException
747      * @throws IOException
748      * @throws ContentNotFoundException 
749      */
750     public void updateMetadata( ManagedRepositoryContent managedRepository, VersionedReference reference )
751         throws LayoutException, RepositoryMetadataException, IOException, ContentNotFoundException
752     {
753         File metadataFile = new File( managedRepository.getRepoRoot(), toPath( reference ) );
754 
755         long lastUpdated = getExistingLastUpdated( metadataFile );
756 
757         ArchivaRepositoryMetadata metadata = new ArchivaRepositoryMetadata();
758         metadata.setGroupId( reference.getGroupId() );
759         metadata.setArtifactId( reference.getArtifactId() );
760         
761         if ( VersionUtil.isSnapshot( reference.getVersion() ) )
762         {
763             // Do SNAPSHOT handling.
764             metadata.setVersion( VersionUtil.getBaseVersion( reference.getVersion() ) );
765 
766             // Gather up all of the versions found in the reference dir, and any
767             // proxied maven-metadata.xml files.
768             Set<String> snapshotVersions = gatherSnapshotVersions( managedRepository, reference );
769 
770             if ( snapshotVersions.isEmpty() )
771             {
772                 throw new ContentNotFoundException( "No snapshot versions found on reference ["
773                     + VersionedReference.toKey( reference ) + "]." );
774             }
775 
776             // sort the list to determine to aide in determining the Latest version.
777             List<String> sortedVersions = new ArrayList<String>();
778             sortedVersions.addAll( snapshotVersions );
779             Collections.sort( sortedVersions, new VersionComparator() );
780 
781             String latestVersion = sortedVersions.get( sortedVersions.size() - 1 );
782 
783             if ( VersionUtil.isUniqueSnapshot( latestVersion ) )
784             {
785                 // The latestVersion will contain the full version string "1.0-alpha-5-20070821.213044-8"
786                 // This needs to be broken down into ${base}-${timestamp}-${build_number}
787 
788                 Matcher m = VersionUtil.UNIQUE_SNAPSHOT_PATTERN.matcher( latestVersion );
789                 if ( m.matches() )
790                 {
791                     metadata.setSnapshotVersion( new SnapshotVersion() );
792                     int buildNumber = NumberUtils.toInt( m.group( 3 ), -1 );
793                     metadata.getSnapshotVersion().setBuildNumber( buildNumber );
794 
795                     Matcher mtimestamp = VersionUtil.TIMESTAMP_PATTERN.matcher( m.group( 2 ) );
796                     if ( mtimestamp.matches() )
797                     {
798                         String tsDate = mtimestamp.group( 1 );
799                         String tsTime = mtimestamp.group( 2 );
800                         
801                         long snapshotLastUpdated = toLastUpdatedLong( tsDate + tsTime );
802                         
803                         lastUpdated = Math.max( lastUpdated, snapshotLastUpdated );
804                         
805                         metadata.getSnapshotVersion().setTimestamp( m.group( 2 ) );
806                     }
807                 }
808             }
809             else if ( VersionUtil.isGenericSnapshot( latestVersion ) )
810             {
811                 // The latestVersion ends with the generic version string.
812                 // Example: 1.0-alpha-5-SNAPSHOT
813 
814                 metadata.setSnapshotVersion( new SnapshotVersion() );
815 
816                 /* Disabled due to decision in [MRM-535].
817                  * Do not set metadata.lastUpdated to file.lastModified.
818                  * 
819                  * Should this be the last updated timestamp of the file, or in the case of an 
820                  * archive, the most recent timestamp in the archive?
821                  * 
822                 ArtifactReference artifact = getFirstArtifact( managedRepository, reference );
823 
824                 if ( artifact == null )
825                 {
826                     throw new IOException( "Not snapshot artifact found to reference in " + reference );
827                 }
828 
829                 File artifactFile = managedRepository.toFile( artifact );
830 
831                 if ( artifactFile.exists() )
832                 {
833                     Date lastModified = new Date( artifactFile.lastModified() );
834                     metadata.setLastUpdatedTimestamp( lastModified );
835                 }
836                 */
837             }
838             else
839             {
840                 throw new RepositoryMetadataException( "Unable to process snapshot version <" + latestVersion
841                     + "> reference <" + reference + ">" );
842             }
843         }
844         else
845         {
846             // Do RELEASE handling.
847             metadata.setVersion( reference.getVersion() );
848         }
849 
850         // Set last updated
851         if ( lastUpdated > 0 )
852         {
853             metadata.setLastUpdatedTimestamp( toLastUpdatedDate( lastUpdated ) );
854         }
855 
856         // Save the metadata model to disk.
857         RepositoryMetadataWriter.write( metadata, metadataFile );
858         ChecksummedFile checksum = new ChecksummedFile( metadataFile );
859         checksum.fixChecksums( algorithms );
860     }

894     /**
895      * Get the first Artifact found in the provided VersionedReference location.
896      *
897      * @param managedRepository the repository to search within.
898      * @param reference         the reference to the versioned reference to search within
899      * @return the ArtifactReference to the first artifact located within the versioned reference. or null if
900      *         no artifact was found within the versioned reference.
901      * @throws IOException     if the versioned reference is invalid (example: doesn't exist, or isn't a directory)
902      * @throws LayoutException
903      */
904     public ArtifactReference getFirstArtifact( ManagedRepositoryContent managedRepository, VersionedReference reference )
905         throws LayoutException, IOException
906     {
907         String path = toPath( reference );
908 
909         int idx = path.lastIndexOf( '/' );
910         if ( idx > 0 )
911         {
912             path = path.substring( 0, idx );
913         }
914 
915         File repoDir = new File( managedRepository.getRepoRoot(), path );
916 
917         if ( !repoDir.exists() )
918         {
919             throw new IOException( "Unable to gather the list of snapshot versions on a non-existant directory: "
920                 + repoDir.getAbsolutePath() );
921         }
922 
923         if ( !repoDir.isDirectory() )
924         {
925             throw new IOException( "Unable to gather the list of snapshot versions on a non-directory: "
926                 + repoDir.getAbsolutePath() );
927         }
928 
929         File repoFiles[] = repoDir.listFiles();
930         for ( int i = 0; i < repoFiles.length; i++ )
931         {
932             if ( repoFiles[i].isDirectory() )
933             {
934                 // Skip it. it's a directory.
935                 continue;
936             }
937 
938             String relativePath = PathUtil.getRelative( managedRepository.getRepoRoot(), repoFiles[i] );
939 
940             if ( filetypes.matchesArtifactPattern( relativePath ) )
941             {
942                 ArtifactReference artifact = managedRepository.toArtifactReference( relativePath );
943 
944                 return artifact;
945             }
946         }
947 
948         // No artifact was found.
949         return null;
950     }
