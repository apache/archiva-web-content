197     public void spool( OutputContext outputContext )
198         throws IOException
199     {
200         if ( !isCollection() )
201         {
202             outputContext.setContentLength( localResource.length() );
203             outputContext.setContentType( mimeTypes.getMimeType( localResource.getName() ) );
204         }
205 
206         if ( !isCollection() && outputContext.hasStream() )
207         {
208             FileInputStream is = null;
209             try
210             {
211                 // Write content to stream
212                 is = new FileInputStream( localResource );
213                 IOUtils.copy( is, outputContext.getOutputStream() );
214             }
215             finally
216             {
217                 IOUtils.closeQuietly( is );
218             }
219         }
220         else if ( outputContext.hasStream() )
221         {
222             IndexWriter writer = new IndexWriter( this, localResource, logicalResource );
223             writer.write( outputContext );
224         }
225     }

269         {
270             String parentPath = Text.getRelativeParent( getResourcePath(), 1 );

289     public void addMember( DavResource resource, InputContext inputContext )
290         throws DavException
291     {
292         File localFile = new File( localResource, resource.getDisplayName() );
293         boolean exists = localFile.exists();
294 
295         if ( isCollection() && inputContext.hasStream() ) // New File
296         {
297             FileOutputStream stream = null;
298             try
299             {
300                 stream = new FileOutputStream( localFile );
301                 IOUtils.copy( inputContext.getInputStream(), stream );
302             }
303             catch ( IOException e )
304             {
305                 throw new DavException( HttpServletResponse.SC_INTERNAL_SERVER_ERROR, e );
306             }
307             finally
308             {
309                 IOUtils.closeQuietly( stream );
310             }
311 
312             // TODO: a bad deployment shouldn't delete an existing file - do we need to write to a temporary location first?
313             long expectedContentLength = inputContext.getContentLength();
314             long actualContentLength = localFile.length();
315             // length of -1 is given for a chunked request or unknown length, in which case we accept what was uploaded
316             if ( expectedContentLength >= 0 && expectedContentLength != actualContentLength )
317             {
318                 String msg =
319                     "Content Header length was " + expectedContentLength + " but was " + actualContentLength;
320                 log.debug( "Upload failed: " + msg );
321 
322                 FileUtils.deleteQuietly( localFile );
323                 throw new DavException( HttpServletResponse.SC_BAD_REQUEST, msg );
324             }
325 
326             queueRepositoryTask( localFile );
327 
328             log.debug(
329                 "File '" + resource.getDisplayName() + ( exists ? "' modified " : "' created " ) + "(current user '" +
330                     this.principal + "')" );
331 
332             triggerAuditEvent( resource, exists ? AuditEvent.MODIFY_FILE : AuditEvent.CREATE_FILE );
333         }
334         else if ( !inputContext.hasStream() && isCollection() ) // New directory
335         {
336             localFile.mkdir();
337 
338             log.debug( "Directory '" + resource.getDisplayName() + "' (current user '" + this.principal + "')" );
339 
340             triggerAuditEvent( resource, AuditEvent.CREATE_DIR );
341         }
342         else
343         {
344             String msg = "Could not write member " + resource.getResourcePath() + " at " + getResourcePath() +
345                 " as this is not a DAV collection";
346             log.debug( msg );
347             throw new DavException( HttpServletResponse.SC_BAD_REQUEST, msg );
348         }
349     }

351     public DavResourceIterator getMembers()
352     {
353         List<DavResource> list = new ArrayList<DavResource>();
354         if ( exists() && isCollection() )
355         {
356             for ( String item : localResource.list() )
357             {
358                 try
359                 {
360                     if ( !item.startsWith( HIDDEN_PATH_PREFIX ) )
361                     {
362                         String path = locator.getResourcePath() + '/' + item;
363                         DavResourceLocator resourceLocator = locator.getFactory().createResourceLocator(
364                             locator.getPrefix(), path );
365                         DavResource resource = factory.createResource( resourceLocator, session );
366 
367                         if ( resource != null )
368                         {
369                             list.add( resource );
370                         }
371                         log.debug( "Resource '" + item + "' retrieved by '" + this.principal + "'" );
372                     }
373                 }
374                 catch ( DavException e )
375                 {
376                     // Should not occur
377                 }
378             }
379         }
380         return new DavResourceIteratorImpl( list );
381     }

383     public void removeMember( DavResource member )
384         throws DavException
385     {
386         File resource = checkDavResourceIsArchivaDavResource( member ).getLocalResource();
387 
388         if ( resource.exists() )
389         {
390             try
391             {
392                 if ( resource.isDirectory() )
393                 {
394                     FileUtils.deleteDirectory( resource );
395 
396                     triggerAuditEvent( member, AuditEvent.REMOVE_DIR );
397                 }
398                 else
399                 {
400                     if ( !resource.delete() )
401                     {
402                         throw new IOException( "Could not remove file" );
403                     }
404 
405                     triggerAuditEvent( member, AuditEvent.REMOVE_FILE );
406                 }
407                 log.debug( ( resource.isDirectory() ? "Directory '" : "File '" ) + member.getDisplayName() +
408                     "' removed (current user '" + this.principal + "')" );
409             }
410             catch ( IOException e )
411             {
412                 throw new DavException( HttpServletResponse.SC_INTERNAL_SERVER_ERROR );
413             }
414         }
415         else
416         {
417             throw new DavException( HttpServletResponse.SC_NOT_FOUND );
418         }

561     public void unlock( String lockToken )
562         throws DavException
563     {
564         ActiveLock lock = getLock( Type.WRITE, Scope.EXCLUSIVE );
565         if ( lock == null )
566         {
567             throw new DavException( HttpServletResponse.SC_PRECONDITION_FAILED );
568         }
569         else if ( lock.isLockedByToken( lockToken ) )
570         {
571             lockManager.releaseLock( lockToken, this );
572         }
573         else
574         {
575             throw new DavException( DavServletResponse.SC_LOCKED );
576         }

594     /**
595      * Fill the set of properties
596      */
597     protected DavPropertySet initProperties()
598     {
599         if ( !exists() )
600         {
601             properties = new DavPropertySet();
602         }
603 
604         if ( properties != null )
605         {
606             return properties;
607         }
608 
609         DavPropertySet properties = new DavPropertySet();
610 
611         // set (or reset) fundamental properties
612         if ( getDisplayName() != null )
613         {
614             properties.add( new DefaultDavProperty( DavPropertyName.DISPLAYNAME, getDisplayName() ) );
615         }
616         if ( isCollection() )
617         {
618             properties.add( new ResourceType( ResourceType.COLLECTION ) );
619             // Windows XP support
620             properties.add( new DefaultDavProperty( DavPropertyName.ISCOLLECTION, "1" ) );
621         }
622         else
623         {
624             properties.add( new ResourceType( ResourceType.DEFAULT_RESOURCE ) );
625 
626             // Windows XP support
627             properties.add( new DefaultDavProperty( DavPropertyName.ISCOLLECTION, "0" ) );
628         }
629 
630         // Need to get the ISO8601 date for properties
631         DateTime dt = new DateTime( localResource.lastModified() );
632         DateTimeFormatter fmt = ISODateTimeFormat.dateTime();
633         String modifiedDate = fmt.print( dt );
634 
635         properties.add( new DefaultDavProperty( DavPropertyName.GETLASTMODIFIED, modifiedDate ) );
636 
637         properties.add( new DefaultDavProperty( DavPropertyName.CREATIONDATE, modifiedDate ) );
638 
639         properties.add( new DefaultDavProperty( DavPropertyName.GETCONTENTLENGTH, localResource.length() ) );
640 
641         this.properties = properties;
642 
643         return properties;
644     }

657     private void triggerAuditEvent( String remoteIP, String principal, String repositoryId, String resource,
658                                     String action )
659     {
660         AuditEvent event = new AuditEvent( repositoryId, principal, resource, action );
661         event.setRemoteIP( remoteIP );
662 
663         for ( AuditListener listener : auditListeners )
664         {
665             listener.auditEvent( event );
666         }
667         
668         // identify as artifact deployment/upload
669         if( action.equals( AuditEvent.CREATE_FILE ) )
670         {
671             action = AuditEvent.UPLOAD_FILE;
672         }
673         
674         String user = principal;
675         if( principal == null )
676         {
677             user = "guest";
678         }
679         
680         ArchivaAuditLogs auditLogs = new ArchivaAuditLogs();
681         auditLogs.setArtifact( resource );
682         auditLogs.setEvent( action );
683         auditLogs.setEventDate( Calendar.getInstance().getTime() );
684         auditLogs.setRepositoryId( repositoryId );
685         auditLogs.setUsername( user );
686         
687         auditLogsDao.saveAuditLogs( auditLogs );
688     }
