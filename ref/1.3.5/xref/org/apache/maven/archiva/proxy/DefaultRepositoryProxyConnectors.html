140     public File fetchFromProxies( ManagedRepositoryContent repository, ArtifactReference artifact )
141         throws ProxyDownloadException
142     {
143         File localFile = toLocalFile( repository, artifact );
144 
145         Properties requestProperties = new Properties();
146         requestProperties.setProperty( "filetype", "artifact" );
147         requestProperties.setProperty( "version", artifact.getVersion() );
148         requestProperties.setProperty( "managedRepositoryId", repository.getId() );
149 
150         List<ProxyConnector> connectors = getProxyConnectors( repository );
151         Map<String, Exception> previousExceptions = new LinkedHashMap<String, Exception>();
152         for ( ProxyConnector connector : connectors )
153         {
154             if ( connector.isDisabled() )
155             {
156                 continue;
157             }
158 
159             RemoteRepositoryContent targetRepository = connector.getTargetRepository();
160             requestProperties.setProperty( "remoteRepositoryId", targetRepository.getId() );
161 
162             String targetPath = targetRepository.toPath( artifact );
163 
164             try
165             {
166                 File downloadedFile =
167                     transferFile( connector, targetRepository, targetPath, repository, localFile, requestProperties,
168                                   true );
169 
170                 if ( fileExists( downloadedFile ) )
171                 {
172                     log.debug( "Successfully transferred: " + downloadedFile.getAbsolutePath() );
173                     return downloadedFile;
174                 }
175             }
176             catch ( NotFoundException e )
177             {
178                 log.debug( "Artifact " + Keys.toKey( artifact ) + " not found on repository \""
179                     + targetRepository.getRepository().getId() + "\"." );
180             }
181             catch ( NotModifiedException e )
182             {
183                 log.debug( "Artifact " + Keys.toKey( artifact ) + " not updated on repository \""
184                     + targetRepository.getRepository().getId() + "\"." );
185             }
186             catch ( ProxyException e )
187             {
188                 validatePolicies( this.downloadErrorPolicies, connector.getPolicies(), requestProperties, artifact,
189                                   targetRepository, localFile, e, previousExceptions );
190             }
191         }
192 
193         if ( !previousExceptions.isEmpty() )
194         {
195             throw new ProxyDownloadException( "Failures occurred downloading from some remote repositories",
196                                               previousExceptions );
197         }
198 
199         log.debug( "Exhausted all target repositories, artifact " + Keys.toKey( artifact ) + " not found." );
200 
201         return null;
202     }

204     public File fetchFromProxies( ManagedRepositoryContent repository, String path )
205     {
206         File localFile = new File( repository.getRepoRoot(), path );
207 
208         // no update policies for these paths
209         if ( localFile.exists() )
210         {
211             return null;
212         }
213 
214         Properties requestProperties = new Properties();
215         requestProperties.setProperty( "filetype", "resource" );
216         requestProperties.setProperty( "managedRepositoryId", repository.getId() );
217 
218         List<ProxyConnector> connectors = getProxyConnectors( repository );
219         for ( ProxyConnector connector : connectors )
220         {
221             if ( connector.isDisabled() )
222             {
223                 continue;
224             }
225 
226             RemoteRepositoryContent targetRepository = connector.getTargetRepository();
227             requestProperties.setProperty( "remoteRepositoryId", targetRepository.getId() );
228 
229             String targetPath = path;
230 
231             try
232             {
233                 File downloadedFile =
234                     transferFile( connector, targetRepository, targetPath, repository, localFile, requestProperties,
235                                   false );
236 
237                 if ( fileExists( downloadedFile ) )
238                 {
239                     log.debug( "Successfully transferred: " + downloadedFile.getAbsolutePath() );
240                     return downloadedFile;
241                 }
242             }
243             catch ( NotFoundException e )
244             {
245                 log.debug( "Resource " + path + " not found on repository \""
246                     + targetRepository.getRepository().getId() + "\"." );
247             }
248             catch ( NotModifiedException e )
249             {
250                 log.debug( "Resource " + path + " not updated on repository \""
251                     + targetRepository.getRepository().getId() + "\"." );
252             }
253             catch ( ProxyException e )
254             {
255                 log.warn( "Transfer error from repository \"" + targetRepository.getRepository().getId()
256                     + "\" for resource " + path + ", continuing to next repository. Error message: " + e.getMessage() );
257                 log.debug( "Full stack trace", e );
258             }
259         }
260 
261         log.debug( "Exhausted all target repositories, resource " + path + " not found." );
262 
263         return null;
264     }

266     public File fetchMetatadaFromProxies( ManagedRepositoryContent repository, String logicalPath )
267     {
268         File localFile = new File( repository.getRepoRoot(), logicalPath );
269 
270         Properties requestProperties = new Properties();
271         requestProperties.setProperty( "filetype", "metadata" );
272         boolean metadataNeedsUpdating = false;
273         long originalTimestamp = getLastModified( localFile );
274 
275         List<ProxyConnector> connectors = getProxyConnectors( repository );
276         for ( ProxyConnector connector : connectors )
277         {
278             if ( connector.isDisabled() )
279             {
280                 continue;
281             }
282 
283             RemoteRepositoryContent targetRepository = connector.getTargetRepository();
284 
285             File localRepoFile = toLocalRepoFile( repository, targetRepository, logicalPath );
286             long originalMetadataTimestamp = getLastModified( localRepoFile );
287 
288             try
289             {
290                 transferFile( connector, targetRepository, logicalPath, repository, localRepoFile, requestProperties,
291                               true );
292 
293                 if ( hasBeenUpdated( localRepoFile, originalMetadataTimestamp ) )
294                 {
295                     metadataNeedsUpdating = true;
296                 }
297             }
298             catch ( NotFoundException e )
299             {
300                 log.debug( "Metadata " + logicalPath + " not found on remote repository \""
301                     + targetRepository.getRepository().getId() + "\".", e );
302             }
303             catch ( NotModifiedException e )
304             {
305                 log.debug( "Metadata " + logicalPath + " not updated on remote repository \""
306                     + targetRepository.getRepository().getId() + "\".", e );
307             }
308             catch ( ProxyException e )
309             {
310                 log.warn( "Transfer error from repository \"" + targetRepository.getRepository().getId()
311                     + "\" for versioned Metadata " + logicalPath + ", continuing to next repository. Error message: "
312                     + e.getMessage() );
313                 log.debug( "Full stack trace", e );
314             }
315         }
316 
317         if ( hasBeenUpdated( localFile, originalTimestamp ) )
318         {
319             metadataNeedsUpdating = true;
320         }
321 
322         if ( metadataNeedsUpdating || !localFile.exists() )
323         {
324             try
325             {
326                 metadataTools.updateMetadata( repository, logicalPath );
327             }
328             catch ( RepositoryMetadataException e )
329             {
330                 log.warn( "Unable to update metadata " + localFile.getAbsolutePath() + ": " + e.getMessage(), e );
331             }
332         }
333 
334         if ( fileExists( localFile ) )
335         {
336             return localFile;
337         }
338 
339         return null;
340     }

342     private long getLastModified( File file )
343     {
344         if ( !file.exists() || !file.isFile() )
345         {
346             return 0;
347         }
348 
349         return file.lastModified();
350     }

352     private boolean hasBeenUpdated( File file, long originalLastModified )
353     {
354         if ( !file.exists() || !file.isFile() )
355         {
356             return false;
357         }
358 
359         long currentLastModified = getLastModified( file );
360         return ( currentLastModified > originalLastModified );
361     }

392     private boolean fileExists( File file )
393     {
394         if ( file == null )
395         {
396             return false;
397         }
398 
399         if ( !file.exists() )
400         {
401             return false;
402         }
403 
404         if ( !file.isFile() )
405         {
406             return false;
407         }
408 
409         return true;

412     /**
413      * Perform the transfer of the file.
414      * 
415      * @param connector the connector configuration to use.
416      * @param remoteRepository the remote repository get the resource from.
417      * @param remotePath the path in the remote repository to the resource to get.
418      * @param repository the managed repository that will hold the file
419      * @param resource the local file to place the downloaded resource into
420      * @param requestProperties the request properties to utilize for policy handling.
421      * @param executeConsumers whether to execute the consumers after proxying
422      * @return the local file that was downloaded, or null if not downloaded.
423      * @throws NotFoundException if the file was not found on the remote repository.
424      * @throws NotModifiedException if the localFile was present, and the resource was present on remote repository, but
425      *             the remote resource is not newer than the local File.
426      * @throws ProxyException if transfer was unsuccessful.
427      */
428     private File transferFile( ProxyConnector connector, RemoteRepositoryContent remoteRepository, String remotePath,
429                                ManagedRepositoryContent repository, File resource, Properties requestProperties,
430                                boolean executeConsumers )
431         throws ProxyException, NotModifiedException
432     {
433         String url = remoteRepository.getURL().getUrl();
434         if ( !url.endsWith( "/" ) )
435         {
436             url = url + "/";
437         }
438         url = url + remotePath;
439         requestProperties.setProperty( "url", url );
440 
441         // Is a whitelist defined?
442         if ( CollectionUtils.isNotEmpty( connector.getWhitelist() ) )
443         {
444             // Path must belong to whitelist.
445             if ( !matchesPattern( remotePath, connector.getWhitelist() ) )
446             {
447                 log.debug( "Path [" + remotePath
448                     + "] is not part of defined whitelist (skipping transfer from repository ["
449                     + remoteRepository.getRepository().getName() + "])." );
450                 return null;
451             }
452         }
453 
454         // Is target path part of blacklist?
455         if ( matchesPattern( remotePath, connector.getBlacklist() ) )
456         {
457             log.debug( "Path [" + remotePath + "] is part of blacklist (skipping transfer from repository ["
458                 + remoteRepository.getRepository().getName() + "])." );
459             return null;
460         }
461 
462         // Handle pre-download policy
463         try
464         {
465             validatePolicies( this.preDownloadPolicies, connector.getPolicies(), requestProperties, resource );
466         }
467         catch ( PolicyViolationException e )
468         {
469             String emsg = "Transfer not attempted on " + url + " : " + e.getMessage();
470             if ( fileExists( resource ) )
471             {
472                 log.debug( emsg + ": using already present local file." );
473                 return resource;
474             }
475 
476             log.debug( emsg );
477             return null;
478         }
479 
480         File tmpMd5 = null;
481         File tmpSha1 = null;
482         File tmpResource = null;
483 
484         File workingDirectory = createWorkingDirectory( repository );
485         try
486         {
487             Wagon wagon = null;
488             try
489             {
490                 RepositoryURL repoUrl = remoteRepository.getURL();
491                 String protocol = repoUrl.getProtocol();
492                 wagon = (Wagon) wagonFactory.getWagon( "wagon#" + protocol );
493                 if ( wagon == null )
494                 {
495                     throw new ProxyException( "Unsupported target repository protocol: " + protocol );
496                 }
497 
498                 boolean connected = connectToRepository( connector, wagon, remoteRepository );
499                 if ( connected )
500                 {
501                     tmpResource = new File( workingDirectory, resource.getName() );
502                     transferSimpleFile( wagon, remoteRepository, remotePath, repository, resource, tmpResource );
503 
504                     // TODO: these should be used to validate the download based on the policies, not always downloaded
505                     // to
506                     // save on connections since md5 is rarely used
507                     tmpSha1 =
508                         transferChecksum( wagon, remoteRepository, remotePath, repository, resource, workingDirectory,
509                                           ".sha1" );
510                     tmpMd5 =
511                         transferChecksum( wagon, remoteRepository, remotePath, repository, resource, workingDirectory,
512                                           ".md5" );
513                 }
514             }
515             catch ( NotFoundException e )
516             {
517                 urlFailureCache.cacheFailure( url );
518                 throw e;
519             }
520             catch ( NotModifiedException e )
521             {
522                 // Do not cache url here.
523                 throw e;
524             }
525             catch ( ProxyException e )
526             {
527                 urlFailureCache.cacheFailure( url );
528                 throw e;
529             }
530             finally
531             {
532                 if ( wagon != null )
533                 {
534                     try
535                     {
536                         wagon.disconnect();
537                     }
538                     catch ( ConnectionException e )
539                     {
540                         log.warn( "Unable to disconnect wagon.", e );
541                     }
542                 }
543             }
544 
545             // Handle post-download policies.
546             try
547             {
548                 validatePolicies( this.postDownloadPolicies, connector.getPolicies(), requestProperties, tmpResource );
549             }
550             catch ( PolicyViolationException e )
551             {
552                 log.warn( "Transfer invalidated from " + url + " : " + e.getMessage() );
553                 executeConsumers = false;
554                 if ( !fileExists( tmpResource ) )
555                 {
556                     resource = null;
557                 }
558             }
559 
560             if ( resource != null )
561             {
562                 synchronized ( resource.getAbsolutePath().intern() )
563                 {
564                     File directory = resource.getParentFile();
565                     moveFileIfExists( tmpMd5, directory );
566                     moveFileIfExists( tmpSha1, directory );
567                     moveFileIfExists( tmpResource, directory );
568                 }
569             }
570         }
571         finally
572         {
573             FileUtils.deleteQuietly( workingDirectory );
574         }
575 
576         if ( executeConsumers )
577         {
578             // Just-in-time update of the index and database by executing the consumers for this artifact
579             //consumers.executeConsumers( connector.getSourceRepository().getRepository(), resource );
580             queueRepositoryTask( connector.getSourceRepository().getRepository().getId(), resource );
581         }
582 
583         return resource;
584     }

617     /**
618      * <p>
619      * Quietly transfer the checksum file from the remote repository to the local file.
620      * </p>
621      * 
622      * @param wagon the wagon instance (should already be connected) to use.
623      * @param remoteRepository the remote repository to transfer from.
624      * @param remotePath the remote path to the resource to get.
625      * @param repository the managed repository that will hold the file
626      * @param localFile the local file that should contain the downloaded contents
627      * @param type the type of checksum to transfer (example: ".md5" or ".sha1")
628      * @throws ProxyException if copying the downloaded file into place did not succeed.
629      */
630     private File transferChecksum( Wagon wagon, RemoteRepositoryContent remoteRepository, String remotePath,
631                                    ManagedRepositoryContent repository, File resource, File tmpDirectory, String ext )
632         throws ProxyException
633     {
634         String url = remoteRepository.getURL().getUrl() + remotePath + ext;
635 
636         // Transfer checksum does not use the policy.
637         if ( urlFailureCache.hasFailedBefore( url ) )
638         {
639             return null;
640         }
641 
642         File destFile = new File( tmpDirectory, resource.getName() + ext );
643 
644         try
645         {
646             transferSimpleFile( wagon, remoteRepository, remotePath + ext, repository, resource, destFile );
647             log.debug( "Checksum " + url + " Downloaded: " + destFile + " to move to " + resource );
648         }
649         catch ( NotFoundException e )
650         {
651             urlFailureCache.cacheFailure( url );
652             log.debug( "Transfer failed, checksum not found: " + url );
653             // Consume it, do not pass this on.
654         }
655         catch ( NotModifiedException e )
656         {
657             log.debug( "Transfer skipped, checksum not modified: " + url );
658             // Consume it, do not pass this on.
659         }
660         catch ( ProxyException e )
661         {
662             urlFailureCache.cacheFailure( url );
663             log.warn( "Transfer failed on checksum: " + url + " : " + e.getMessage(), e );
664             // Critical issue, pass it on.
665             throw e;
666         }
667         return destFile;
668     }

670     /**
671      * Perform the transfer of the remote file to the local file specified.
672      * 
673      * @param wagon the wagon instance to use.
674      * @param remoteRepository the remote repository to use
675      * @param remotePath the remote path to attempt to get
676      * @param repository the managed repository that will hold the file
677      * @param origFile the local file to save to
678      * @return The local file that was transfered.
679      * @throws ProxyException if there was a problem moving the downloaded file into place.
680      * @throws WagonException if there was a problem tranfering the file.
681      */
682     private void transferSimpleFile( Wagon wagon, RemoteRepositoryContent remoteRepository, String remotePath,
683                                      ManagedRepositoryContent repository, File origFile, File destFile )
684         throws ProxyException
685     {
686         assert ( remotePath != null );
687 
688         // Transfer the file.
689         try
690         {
691             boolean success = false;
692 
693             if ( !origFile.exists() )
694             {
695                 log.debug( "Retrieving " + remotePath + " from " + remoteRepository.getRepository().getName() );
696                 wagon.get( remotePath, destFile );
697                 success = true;
698 
699                 // You wouldn't get here on failure, a WagonException would have been thrown.
700                 log.debug( "Downloaded successfully." );
701             }
702             else
703             {
704                 log.debug( "Retrieving " + remotePath + " from " + remoteRepository.getRepository().getName()
705                     + " if updated" );
706                 success = wagon.getIfNewer( remotePath, destFile, origFile.lastModified() );
707                 if ( !success )
708                 {
709                     throw new NotModifiedException( "Not downloaded, as local file is newer than remote side: "
710                         + origFile.getAbsolutePath() );
711                 }
712 
713                 if ( destFile.exists() )
714                 {
715                     log.debug( "Downloaded successfully." );
716                 }
717             }
718         }
719         catch ( ResourceDoesNotExistException e )
720         {
721             throw new NotFoundException( "Resource [" + remoteRepository.getURL() + "/" + remotePath
722                 + "] does not exist: " + e.getMessage(), e );
723         }
724         catch ( WagonException e )
725         {
726             // TODO: shouldn't have to drill into the cause, but TransferFailedException is often not descriptive enough
727 
728             String msg =

771     private void validatePolicies( Map<String, DownloadErrorPolicy> policies, Map<String, String> settings,
772                                    Properties request, ArtifactReference artifact, RemoteRepositoryContent content,
773                                    File localFile, ProxyException exception, Map<String, Exception> previousExceptions )
774         throws ProxyDownloadException
775     {
776         boolean process = true;
777         for ( Entry<String, ? extends DownloadErrorPolicy> entry : policies.entrySet() )
778         {
779             String key = entry.getKey();
780             DownloadErrorPolicy policy = entry.getValue();
781             String defaultSetting = policy.getDefaultOption();
782             String setting = StringUtils.defaultString( settings.get( key ), defaultSetting );
783 
784             log.debug( "Applying [" + key + "] policy with [" + setting + "]" );
785             try
786             {
787                 // all policies must approve the exception, any can cancel
788                 process = policy.applyPolicy( setting, request, localFile, exception, previousExceptions );
789                 if ( !process )
790                 {
791                     break;
792                 }
793             }
794             catch ( PolicyConfigurationException e )
795             {
796                 log.error( e.getMessage(), e );
797             }
798         }
799 
800         if ( process )
801         {
802             // if the exception was queued, don't throw it
803             if ( !previousExceptions.containsKey( content.getId() ) )
804             {
805                 throw new ProxyDownloadException(
806                                                   "An error occurred in downloading from the remote repository, and the policy is to fail immediately",
807                                                   content.getId(), exception );
808             }
809         }
810         else
811         {
812             // if the exception was queued, but cancelled, remove it
813             previousExceptions.remove( content.getId() );
814         }
815 
816         log.warn( "Transfer error from repository \"" + content.getRepository().getId() + "\" for artifact "
817             + Keys.toKey( artifact ) + ", continuing to next repository. Error message: " + exception.getMessage() );
818         log.debug( "Full stack trace", exception );
819     }

849      * @param target The final location of the downloaded file

852     private void moveTempToTarget( File temp, File target )

854     {

858         }
