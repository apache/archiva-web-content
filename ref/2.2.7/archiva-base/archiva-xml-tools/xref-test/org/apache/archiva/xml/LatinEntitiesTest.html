22  import junit.framework.TestCase;
23  import org.apache.archiva.test.utils.ArchivaBlockJUnit4ClassRunner;
24  import org.junit.Test;
25  import org.junit.runner.RunWith;
26  
27  /**
28   * LatinEntitiesTest 
29   *
30   *
31   */
32  @RunWith( ArchivaBlockJUnit4ClassRunner.class )
33  public class LatinEntitiesTest
34      extends TestCase
35  {
36      @Test
37      public void testResolveEntity()
38      {
39          // Good Entities.
40          assertEquals( "\u00a9", LatinEntities.resolveEntity( "&copy;" ) );
41          assertEquals( "\u221e", LatinEntities.resolveEntity( "&infin;" ) );
42          assertEquals( "\u00f8", LatinEntities.resolveEntity( "&oslash;" ) );
43  
44          // Bad Entities.
45          assertEquals( "", LatinEntities.resolveEntity( "" ) );
46          assertEquals( "&amp;", LatinEntities.resolveEntity( "&amp;" ) );
47          assertEquals( null, LatinEntities.resolveEntity( null ) );
48      }
49  }
