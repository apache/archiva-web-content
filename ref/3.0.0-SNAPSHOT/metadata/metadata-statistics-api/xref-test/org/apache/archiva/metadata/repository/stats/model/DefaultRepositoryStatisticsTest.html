24  import java.util.Date;

28  import static org.junit.Assert.assertEquals;

30  /**
31   * @author Martin Stockhammer <martin_s@apache.org>
32   */
33  public class DefaultRepositoryStatisticsTest
34  {
35      @Test
36      public void toProperties( ) throws Exception
37      {
38          DefaultRepositoryStatistics stats = new DefaultRepositoryStatistics();
39          Date startTime = new Date();
40          Date endTime = new Date();
41          stats.setScanStartTime( startTime );
42          stats.setScanEndTime( endTime );
43          stats.setTotalFileCount( 500 );
44          stats.setNewFileCount( 10 );
45          stats.setRepositoryId( "test-repo" );
46          stats.setTotalArtifactCount( 300 );
47          stats.setTotalArtifactFileSize( 4848484 );
48          stats.setTotalGroupCount( 4 );
49          stats.setTotalProjectCount( 6 );
50          stats.setCustomValue( "test.value.1", 55 );
51          stats.setCustomValue( "test.value.2", 44);
52          stats.setTotalCountForType( "java-source",13 );
53          stats.setTotalCountForType( "pom", 5 );
54  
55          Map<String, String> props = stats.toProperties( );
56  
57          assertEquals( "500", props.get("totalFileCount") );
58          assertEquals( "10", props.get("newFileCount"));
59          assertEquals( "300", props.get("totalArtifactCount"));
60          assertEquals( "4848484", props.get("totalArtifactFileSize"));
61          assertEquals("4", props.get("totalGroupCount"));
62          assertEquals("6", props.get("totalProjectCount"));
63          assertEquals("55",props.get("count-custom-test.value.1" ));
64          assertEquals("44", props.get("count-custom-test.value.2"));
65          assertEquals("13", props.get("count-type-java-source"));
66          assertEquals("5", props.get("count-type-pom"));
67          assertEquals( String.valueOf(startTime.getTime()), props.get("scanStartTime"));
68          assertEquals( String.valueOf(endTime.getTime()), props.get("scanEndTime"));
69      }
70  
71      @Test
72      public void fromProperties( ) throws Exception
73      {
74          DefaultRepositoryStatistics stats = new DefaultRepositoryStatistics( );
75          Date startTime = new Date();
76          Date endTime = new Date();
77          Map<String,String> props = new HashMap<>(  );
78          props.put("totalFileCount","501");
79          props.put("newFileCount","11");
80          props.put("totalArtifactCount","301");
81          props.put("totalArtifactFileSize","473565557");
82          props.put("totalGroupCount","5");
83          props.put("totalProjectCount","7");
84          props.put("count-custom-test.value.1","56");
85          props.put("count-custom-test.value.2","45");
86          props.put("count-type-java-source","14");
87          props.put("count-type-pom","6");
88          props.put("scanStartTime", String.valueOf(startTime.getTime()));
89          props.put("scanEndTime", String.valueOf(endTime.getTime()));
90  
91          stats.fromProperties( props );
92  
93          assertEquals(501,stats.getTotalFileCount());
94          assertEquals(11,stats.getNewFileCount());
95          assertEquals(301, stats.getTotalArtifactCount());
96          assertEquals(473565557, stats.getTotalArtifactFileSize());
97          assertEquals(5, stats.getTotalGroupCount());
98          assertEquals(7, stats.getTotalProjectCount());
99          assertEquals(56, stats.getCustomValue( "test.value.1" ));
100         assertEquals(45, stats.getCustomValue( "test.value.2" ));
101         assertEquals(14, stats.getTotalCountForType( "java-source" ));
102         assertEquals(6, stats.getTotalCountForType( "pom" ));
103         assertEquals(startTime, stats.getScanStartTime());
104         assertEquals( endTime, stats.getScanEndTime() );
105     }
