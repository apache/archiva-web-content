111     public Path getProjectBase() {
112         if (this.projectBase.get()==null) {
113             String pathVal = System.getProperty("mvn.project.base.dir");
114             Path baseDir;
115             if (StringUtils.isEmpty(pathVal)) {
116                 baseDir= Paths.get("").toAbsolutePath();
117             } else {
118                 baseDir = Paths.get(pathVal).toAbsolutePath();
119             }
120             this.projectBase.compareAndSet(null, baseDir);
121         }
122         return this.projectBase.get();
123     }

125     public Path getAppserverBase() {
126         if (appserverBase.get()==null)
127         {
128             String pathVal = System.getProperty( "appserver.base" );
129             Path basePath;
130             if ( StringUtils.isNotEmpty( pathVal ) )
131             {
132                 basePath = Paths.get( pathVal );
133             }
134             else
135             {
136                 log.warn("Using relative path to working directory, appserver.base was not set!");
137                 basePath = Paths.get( "target/appserver-base" );
138             }
139             appserverBase.set( basePath );
140         }
141         return appserverBase.get();
142     }

241     protected String createVersionMetadata(String groupId, String artifactId, String version, String timestamp, String buildNumber, String lastUpdated) {
242         StringBuilder buf = new StringBuilder();
243         buf.append("<?xml version=\"1.0\" encoding=\"UTF-8\"?>\n\n");
244         buf.append("<metadata>\n");
245         buf.append("  <groupId>").append(groupId).append("</groupId>\n");
246         buf.append("  <artifactId>").append(artifactId).append("</artifactId>\n");
247         buf.append("  <version>").append(version).append("</version>\n");
248         boolean hasSnapshot = StringUtils.isNotBlank(timestamp) || StringUtils.isNotBlank(buildNumber);
249         boolean hasLastUpdated = StringUtils.isNotBlank(lastUpdated);
250         if (hasSnapshot || hasLastUpdated) {
251             buf.append("  <versioning>\n");
252             if (hasSnapshot) {
253                 buf.append("    <snapshot>\n");
254                 buf.append("      <buildNumber>").append(buildNumber).append("</buildNumber>\n");
255                 buf.append("      <timestamp>").append(timestamp).append("</timestamp>\n");
256                 buf.append("    </snapshot>\n");
257             }
258             if (hasLastUpdated) {
259                 buf.append("    <lastUpdated>").append(lastUpdated).append("</lastUpdated>\n");
260             }
261             buf.append("  </versioning>\n");
262         }
263         buf.append("</metadata>");
264         return buf.toString();
265     }

576     protected WebResponse getWebResponse( WebRequest webRequest ) //, boolean followRedirect )
577         throws Exception
578     {
579 
580         MockHttpServletRequest request = new MockHttpServletRequest();
581         request.setRequestURI( webRequest.getUrl().getPath() );
582         request.addHeader( "User-Agent", "Apache Archiva unit test" );
583 
584         request.setMethod( webRequest.getHttpMethod().name() );
585 
586         if ( webRequest.getHttpMethod() == HttpMethod.PUT )
587         {
588             PutMethodWebRequest putRequest = PutMethodWebRequest.class.cast( webRequest );
589             request.setContentType( putRequest.contentType );
590             request.setContent( IOUtils.toByteArray( putRequest.inputStream ) );
591         }
592 
593         if ( webRequest instanceof MkColMethodWebRequest )
594         {
595             request.setMethod( "MKCOL" );
596         }
597 
598         final MockHttpServletResponse response = execute( request );
599 
600         if ( response.getStatus() == HttpServletResponse.SC_MOVED_PERMANENTLY
601             || response.getStatus() == HttpServletResponse.SC_MOVED_TEMPORARILY )
602         {
603             String location = response.getHeader( "Location" );
604             log.debug( "follow redirect to {}", location );
605             return getWebResponse( new GetMethodWebRequest( location ) );
606         }
607 
608         return new WebResponse( null, null, 1 )
609         {
610             @Override
611             public String getContentAsString()
612             {
613                 try
614                 {
615                     return response.getContentAsString();
616                 }
617                 catch ( UnsupportedEncodingException e )
618                 {
619                     throw new RuntimeException( e.getMessage(), e );
620                 }
621             }
622 
623             @Override
624             public int getStatusCode()
625             {
626                 return response.getStatus();
627             }
628 
629             @Override
630             public String getResponseHeaderValue( String headerName )
631             {
632                 return response.getHeader( headerName );
633             }
634         };
635     }

637     protected MockHttpServletResponse execute( HttpServletRequest request )
638         throws Exception
639     {
640         MockHttpServletResponse response = new MockHttpServletResponse()
641         {
642             @Override
643             public String getContentAsString()
644                 throws UnsupportedEncodingException
645             {
646                 String errorMessage = getErrorMessage();
647                 return ( errorMessage != null ) ? errorMessage : super.getContentAsString();
648             }
649         };
650         this.unauthenticatedRepositoryServlet.service( request, response );
651         return response;
652     }
