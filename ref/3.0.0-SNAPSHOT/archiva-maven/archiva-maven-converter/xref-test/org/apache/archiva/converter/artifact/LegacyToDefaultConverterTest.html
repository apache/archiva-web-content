112     private void copyDirectoryStructure( Path sourceDirectory, Path destinationDirectory )
113         throws IOException
114     {
115         if ( !Files.exists(sourceDirectory) )
116         {
117             throw new IOException( "Source directory doesn't exists (" + sourceDirectory.toAbsolutePath()+ ")." );
118         }
119 
120         Path[] files = Files.list( sourceDirectory ).toArray( Path[]::new );
121 
122         String sourcePath = sourceDirectory.toAbsolutePath().toString();
123 
124         for ( int i = 0; i < files.length; i++ )
125         {
126             Path file = files[i];
127 
128             String dest = file.toAbsolutePath().toString();
129 
130             dest = dest.substring( sourcePath.length() + 1 );
131 
132             Path destination = destinationDirectory.resolve( dest );
133 
134             if ( Files.isRegularFile( file ) )
135             {
136                 destination = destination.getParent();
137 
138                 FileUtils.copyFileToDirectory( file.toFile(), destination.toFile() );
139             }
140             else if ( Files.isDirectory( file ) )
141             {
142                 if ( !".svn".equals( file.getFileName().toString() ) )
143                 {
144                     if ( !Files.exists(destination))
145                     {
146                         Files.createDirectories( destination );
147                     }
148                     copyDirectoryStructure( file, destination );
149                 }
150             }
151             else
152             {
153                 throw new IOException( "Unknown file type: " + file.toAbsolutePath() );
154             }
155         }
156     }

562     @Test
563     public void testUnmodifiedArtifact()
564         throws Exception, InterruptedException
565     {
566         // test the unmodified artifact is untouched
567 
568         Artifact artifact = createArtifact( "test", "unmodified-artifact", "1.0.0" );
569         Artifact pomArtifact = createPomArtifact( artifact );
570 
571         Path sourceFile = Paths.get( sourceRepository.getBasedir(), sourceRepository.pathOf( artifact ) );
572         Path sourcePomFile = Paths.get( sourceRepository.getBasedir(), sourceRepository.pathOf( pomArtifact ) );
573         Path targetFile = Paths.get( targetRepository.getBasedir(), targetRepository.pathOf( artifact ) );
574         Path targetPomFile = Paths.get( targetRepository.getBasedir(), targetRepository.pathOf( pomArtifact ) );
575 
576         assertTrue( "Check target file exists", Files.exists(targetFile) );
577         assertTrue( "Check target POM exists", Files.exists(targetPomFile) );
578 
579         Files.setLastModifiedTime( sourceFile, FileTime.from(System.currentTimeMillis(), TimeUnit.MILLISECONDS) );
580         Files.setLastModifiedTime( sourcePomFile, FileTime.from(System.currentTimeMillis(), TimeUnit.MILLISECONDS) );
581 
582         long origTime = Files.getLastModifiedTime( targetFile ).toMillis();
583         long origPomTime = Files.getLastModifiedTime( targetPomFile ).toMillis();
584 
585         // Need to guarantee last modified is not equal
586         Thread.sleep( SLEEP_MILLIS );
587 
588         artifactConverter.convert( artifact, targetRepository );
589         checkSuccess( artifactConverter );
590 
591         compareFiles( sourceFile, targetFile );
592         compareFiles( sourcePomFile, targetPomFile );

941     private Artifact createArtifact( String groupId, String artifactId, String version )
942     {
943         Matcher matcher = Artifact.VERSION_FILE_PATTERN.matcher( version );
944         String baseVersion;
945         if ( matcher.matches() )
946         {
947             baseVersion = matcher.group( 1 ) + "-SNAPSHOT";
948         }
949         else
950         {
951             baseVersion = version;
952         }
953         return createArtifact( groupId, artifactId, baseVersion, version, "jar" );
954     }

972     private static void compareFiles( Path expectedPomFile, Path pomFile )
973         throws IOException
974     {
975         String expectedContent = normalizeString(
976             org.apache.archiva.common.utils.FileUtils.readFileToString( expectedPomFile, Charset.defaultCharset() ) );
977         String targetContent =
978             normalizeString( org.apache.archiva.common.utils.FileUtils.readFileToString( pomFile, Charset.defaultCharset() ) );

980                       targetContent );

983     private static String normalizeString( String path )
984     {
985         return path.trim().replaceAll( "\r\n", "\n" ).replace( '\r', '\n' ).replaceAll( "<\\?xml .+\\?>",
986                                                                                         "" ).replaceAll( "^\\s+", "" );
987     }

1001     private int countWarningMessages( ArtifactConverter converter )
1002     {
1003         int count = 0;

1005         {
1006             count += values.size();
1007         }
1008         return count;

1011     private void assertHasWarningReason( ArtifactConverter converter, String reason )
1012     {
1013         assertNotNull( "Warnings should never be null.", converter.getWarnings() );
1014         assertTrue( "Expecting 1 or more Warnings", countWarningMessages( converter ) > 0 );
1015 
1016         for ( List<String> messages : converter.getWarnings().values() )
1017         {
1018             if ( messages.contains( reason ) )
1019             {
1020                 /* No need to check any further */
1021                 return;
1022             }
1023         }
1024 
1025         /* didn't find it. */
1026 
1027         for ( Map.Entry<Artifact, List<String>> entry : converter.getWarnings().entrySet() )
1028         {
1029             Artifact artifact = (Artifact) entry.getKey();
1030             System.out.println(
1031                 "-Artifact: " + artifact.getGroupId() + ":" + artifact.getArtifactId() + ":" + artifact.getVersion() );
1032             List<String> messages = entry.getValue();
1033             for ( String message : messages )
1034             {
1035                 System.out.println( "  " + message );
1036             }
1037         }
1038         fail( "Unable to find message <" + reason + "> in warnings." );
1039     }
