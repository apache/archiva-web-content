179     private void assertGetVersions( String artifactId, List<String> expectedVersions )
180         throws Exception
181     {
182         ProjectReference reference = new ProjectReference();
183         reference.setGroupId( "org.apache.archiva.metadata.tests" );
184         reference.setArtifactId( artifactId );
185 
186         // Use the test metadata-repository, which is already setup for
187         // These kind of version tests.
188         Path repoDir = Paths.get( "src/test/repositories/metadata-repository" );
189         (( EditableManagedRepository)repoContent.getRepository()).setLocation( repoDir.toAbsolutePath().toUri() );
190 
191         // Request the versions.
192         Set<String> testedVersionSet = repoContent.getVersions( reference );
193 
194         // Sort the list (for asserts)
195         List<String> testedVersions = new ArrayList<>();
196         testedVersions.addAll( testedVersionSet );
197         Collections.sort( testedVersions, new VersionComparator() );
198 
199         // Test the expected array of versions, to the actual tested versions
200         assertEquals( "available versions", expectedVersions, testedVersions );
201     }
202 
203     private void assertVersions( String artifactId, String version, String[] expectedVersions )
204         throws Exception
205     {
206         VersionedReference reference = new VersionedReference();
207         reference.setGroupId( "org.apache.archiva.metadata.tests" );
208         reference.setArtifactId( artifactId );
209         reference.setVersion( version );
210 
211         // Use the test metadata-repository, which is already setup for
212         // These kind of version tests.
213         Path repoDir = Paths.get( "src/test/repositories/metadata-repository" );
214         ((EditableManagedRepository)repoContent.getRepository()).setLocation( repoDir.toAbsolutePath().toUri() );
215 
216         // Request the versions.
217         Set<String> testedVersionSet = repoContent.getVersions( reference );
218 
219         // Sort the list (for asserts later)
220         List<String> testedVersions = new ArrayList<>();
221         testedVersions.addAll( testedVersionSet );
222         Collections.sort( testedVersions, new VersionComparator() );
223 
224         // Test the expected array of versions, to the actual tested versions
225         assertEquals( "Assert Versions: length/size", expectedVersions.length, testedVersions.size() );
226 
227         for ( int i = 0; i < expectedVersions.length; i++ )
228         {
229             String actualVersion = testedVersions.get( i );
230             assertEquals( "Versions[" + i + "]", expectedVersions[i], actualVersion );
231         }
232     }
