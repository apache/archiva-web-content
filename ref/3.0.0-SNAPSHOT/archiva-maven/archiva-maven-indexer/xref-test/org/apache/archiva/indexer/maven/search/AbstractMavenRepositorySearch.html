176     protected ManagedRepositoryConfiguration createRepositoryConfig( String repository )
177     {
178         ManagedRepositoryConfiguration repositoryConfig = new ManagedRepositoryConfiguration();
179         repositoryConfig.setId( repository );
180         repositoryConfig.setLocation( org.apache.archiva.common.utils.FileUtils.getBasedir() + "/target/repos/" + repository );
181         Path f = Paths.get( repositoryConfig.getLocation() );
182         if ( !Files.exists(f) )
183         {
184             try
185             {
186                 Files.createDirectories( f );
187             }
188             catch ( IOException e )
189             {
190                 log.error("Could not create directories for {}", f);
191             }
192         }
193         repositoryConfig.setLayout( "default" );
194         repositoryConfig.setName( repository );
195         repositoryConfig.setScanned( true );
196         repositoryConfig.setSnapshots( false );
197         repositoryConfig.setReleases( true );
198         repositoryConfig.setIndexDir(DEFAULT_INDEX_PATH);
199 
200         return repositoryConfig;
201     }

207     protected void createIndex( String repository, List<Path> filesToBeIndexed, boolean scan, Path indexDir, boolean copyFiles)
208         throws Exception
209     {
210         Repository rRepo = repositoryRegistry.getRepository(repository);
211         IndexCreationFeature icf = rRepo.getFeature(IndexCreationFeature.class).get();
212 
213 
214         ArchivaIndexingContext archivaCtx = rRepo.getIndexingContext();
215         IndexingContext context = archivaCtx.getBaseContext(IndexingContext.class);
216 
217         if ( archivaCtx != null )
218         {
219             archivaCtx.close(true);
220         }
221 
222         Path repoDir = Paths.get(org.apache.archiva.common.utils.FileUtils.getBasedir()).resolve("target").resolve("repos").resolve(repository);
223 
224         Path indexerDirectory = repoDir.resolve(".indexer" );
225 
226         if ( Files.exists(indexerDirectory) )
227         {
228             FileUtils.deleteDirectory( indexerDirectory );
229         }
230 
231         assertFalse( Files.exists(indexerDirectory) );
232 
233         Path lockFile = repoDir.resolve(".indexer/write.lock" );
234         if ( Files.exists(lockFile) )
235         {
236             Files.delete(lockFile);
237         }
238         assertFalse( Files.exists(lockFile) );
239         if (indexDir==null) {
240             Path indexDirectory =
241                     Paths.get(org.apache.archiva.common.utils.FileUtils.getBasedir(), "target/index/test-" + Long.toString(System.currentTimeMillis()));
242             indexDirectory.toFile().deleteOnExit();
243             FileUtils.deleteDirectory(indexDirectory);
244             icf.setIndexPath(indexDirectory.toUri());
245         } else {
246 
247             icf.setIndexPath(indexDir.toUri());
248         }
249         if (copyFiles) {
250             Path repo = Paths.get(org.apache.archiva.common.utils.FileUtils.getBasedir(), "src/test/" + repository);
251             assertTrue(Files.exists(repo));
252             org.apache.commons.io.FileUtils.copyDirectory(repo.toFile(), repoDir.toFile());
253         }
254 
255 
256 
257 
258         archivaConfigControl.reset();
259         archivaConfig.addListener( EasyMock.anyObject( ConfigurationListener.class ) );
260         EasyMock.expect( archivaConfig.getConfiguration() ).andReturn(config).anyTimes();
261         archivaConfig.save(EasyMock.anyObject(Configuration.class));
262         EasyMock.expectLastCall().anyTimes();
263         archivaConfigControl.replay();
264         repositoryRegistry.reload();
265         archivaConfigControl.reset();
266 
267         rRepo = repositoryRegistry.getRepository(repository);
268         icf = rRepo.getFeature(IndexCreationFeature.class).get();
269 
270 
271         archivaCtx = rRepo.getIndexingContext();
272         context = archivaCtx.getBaseContext(IndexingContext.class);
273 
274 
275         // minimize datas in memory
276 //        context.getIndexWriter().setMaxBufferedDocs( -1 );
277 //        context.getIndexWriter().setRAMBufferSizeMB( 1 );
278         for ( Path artifactFile : filesToBeIndexed )
279         {
280             assertTrue( "file not exists " + artifactFile, Files.exists(artifactFile) );
281             ArtifactContext ac = artifactContextProducer.getArtifactContext( context, artifactFile.toFile() );
282 
283             if ( artifactFile.toString().endsWith( ".pom" ) )
284             {
285                 ac.getArtifactInfo().setFileExtension( "pom" );
286                 ac.getArtifactInfo().setPackaging( "pom" );
287                 ac.getArtifactInfo().setClassifier( "pom" );
288             }
289             indexer.addArtifactToIndex( ac, context );
290             context.updateTimestamp( true );
291         }
292 
293         if ( scan )
294         {
295             DefaultScannerListener listener = new DefaultScannerListener( context, indexerEngine, true, new ArtifactScanListener());
296             ScanningRequest req = new ScanningRequest(context, listener );
297             scanner.scan( req );
298             context.commit();
299         }
300         // force flushing
301         context.commit();
302         //  context.getIndexWriter().commit();
303         context.setSearchable( true );
304 
305     }

339     public String niceDisplay( SearchResults searchResults )
340         throws Exception
341     {
342         StringBuilder sb = new StringBuilder();
343         for ( SearchResultHit hit : searchResults.getHits() )
344         {
345             sb.append( hit.toString() ).append( SystemUtils.LINE_SEPARATOR );
346         }
347         return sb.toString();
