176     @Override
177     public ProjectVersionMetadata readProjectVersionMetadata( ReadMetadataRequest readMetadataRequest )
178         throws RepositoryStorageMetadataNotFoundException, RepositoryStorageMetadataInvalidException,
179         RepositoryStorageRuntimeException
180     {
181         try
182         {
183             ManagedRepository managedRepository = repositoryRegistry.getManagedRepository(readMetadataRequest.getRepositoryId());
184             boolean isReleases = managedRepository.getActiveReleaseSchemes().contains(ReleaseScheme.RELEASE);
185             boolean isSnapshots = managedRepository.getActiveReleaseSchemes().contains(ReleaseScheme.SNAPSHOT);
186             String artifactVersion = readMetadataRequest.getProjectVersion();
187             // olamy: in case of browsing via the ui we can mix repos (parent of a SNAPSHOT can come from release repo)
188             if ( !readMetadataRequest.isBrowsingRequest() )
189             {
190                 if ( VersionUtil.isSnapshot( artifactVersion ) )
191                 {
192                     // skygo trying to improve speed by honoring managed configuration MRM-1658
193                     if ( isReleases && !isSnapshots )
194                     {
195                         throw new RepositoryStorageRuntimeException( "lookforsnaponreleaseonly",
196                                                                      "managed repo is configured for release only" );
197                     }
198                 }
199                 else
200                 {
201                     if ( !isReleases && isSnapshots)
202                     {
203                         throw new RepositoryStorageRuntimeException( "lookforsreleaseonsneponly",
204                                                                      "managed repo is configured for snapshot only" );
205                     }
206                 }
207             }
208             Path basedir = Paths.get( managedRepository.getLocation() );
209             if ( VersionUtil.isSnapshot( artifactVersion ) )
210             {
211                 Path metadataFile = pathTranslator.toFile( basedir, readMetadataRequest.getNamespace(),
212                                                            readMetadataRequest.getProjectId(), artifactVersion,
213                                                            METADATA_FILENAME );
214                 try
215                 {
216                     ArchivaRepositoryMetadata metadata = MavenMetadataReader.read( metadataFile );
217 
218                     // re-adjust to timestamp if present, otherwise retain the original -SNAPSHOT filename
219                     SnapshotVersion snapshotVersion = metadata.getSnapshotVersion();
220                     if ( snapshotVersion != null )
221                     {
222                         artifactVersion =
223                             artifactVersion.substring( 0, artifactVersion.length() - 8 ); // remove SNAPSHOT from end
224                         artifactVersion =
225                             artifactVersion + snapshotVersion.getTimestamp() + "-" + snapshotVersion.getBuildNumber();
226                     }
227                 }
228                 catch ( XMLException e )
229                 {
230                     // unable to parse metadata - LOGGER it, and continue with the version as the original SNAPSHOT version
231                     LOGGER.warn( "Invalid metadata: {} - {}", metadataFile, e.getMessage() );
232                 }
233             }
234 
235             // TODO: won't work well with some other layouts, might need to convert artifact parts to ID by path translator
236             String id = readMetadataRequest.getProjectId() + "-" + artifactVersion + ".pom";
237             Path file =
238                 pathTranslator.toFile( basedir, readMetadataRequest.getNamespace(), readMetadataRequest.getProjectId(),
239                                        readMetadataRequest.getProjectVersion(), id );
240 
241             if ( !Files.exists(file) )
242             {
243                 // metadata could not be resolved
244                 throw new RepositoryStorageMetadataNotFoundException(
245                     "The artifact's POM file '" + file.toAbsolutePath() + "' was missing" );
246             }
247 
248             // TODO: this is a workaround until we can properly resolve using proxies as well - this doesn't cache
249             //       anything locally!
250             List<RemoteRepository> remoteRepositories = new ArrayList<>();
251             Map<String, NetworkProxy> networkProxies = new HashMap<>();
252 
253             Map<String, List<ProxyConnector>> proxyConnectorsMap = proxyConnectorAdmin.getProxyConnectorAsMap();
254             List<ProxyConnector> proxyConnectors = proxyConnectorsMap.get( readMetadataRequest.getRepositoryId() );
255             if ( proxyConnectors != null )
256             {
257                 for ( ProxyConnector proxyConnector : proxyConnectors )
258                 {
259                     RemoteRepository remoteRepoConfig =
260                         repositoryRegistry.getRemoteRepository( proxyConnector.getTargetRepoId() );
261 
262                     if ( remoteRepoConfig != null )
263                     {
264                         remoteRepositories.add( remoteRepoConfig );
265 
266                         NetworkProxy networkProxyConfig =
267                             networkProxyAdmin.getNetworkProxy( proxyConnector.getProxyId() );
268 
269                         if ( networkProxyConfig != null )
270                         {
271                             // key/value: remote repo ID/proxy info
272                             networkProxies.put( proxyConnector.getTargetRepoId(), networkProxyConfig );
273                         }
274                     }
275                 }
276             }
277 
278             // That's a browsing request so we can a mix of SNAPSHOT and release artifacts (especially with snapshots which
279             // can have released parent pom
280             if ( readMetadataRequest.isBrowsingRequest() )
281             {
282                 remoteRepositories.addAll( repositoryRegistry.getRemoteRepositories() );
283             }
284 
285             ModelBuildingRequest req =
286                 new DefaultModelBuildingRequest().setProcessPlugins( false ).setPomFile( file.toFile() ).setTwoPhaseBuilding(
287                     false ).setValidationLevel( ModelBuildingRequest.VALIDATION_LEVEL_MINIMAL );
288 
289             //MRM-1607. olamy this will resolve jdk profiles on the current running archiva jvm
290             req.setSystemProperties( System.getProperties() );
291 
292             // MRM-1411
293             req.setModelResolver(
294                 new RepositoryModelResolver( managedRepository, pathTranslator, wagonFactory, remoteRepositories,
295                                              networkProxies, managedRepository ));
296 
297             Model model;
298             try
299             {
300                 model = builder.build( req ).getEffectiveModel();
301             }
302             catch ( ModelBuildingException e )
303             {
304                 String msg = "The artifact's POM file '" + file + "' was invalid: " + e.getMessage();
305 
306                 List<ModelProblem> modelProblems = e.getProblems();
307                 for ( ModelProblem problem : modelProblems )
308                 {
309                     // MRM-1411, related to MRM-1335
310                     // this means that the problem was that the parent wasn't resolved!
311                     // olamy really hackhish but fail with java profile so use error message
312                     // || ( StringUtils.startsWith( problem.getMessage(), "Failed to determine Java version for profile" ) )
313                     // but setTwoPhaseBuilding(true) fix that
314                     if ( ( (problem.getException() instanceof FileNotFoundException
315                         || problem.getException() instanceof NoSuchFileException
316                     ) && e.getModelId() != null &&
317                         !e.getModelId().equals( problem.getModelId() ) ) )
318                     {
319                         LOGGER.warn( "The artifact's parent POM file '{}' cannot be resolved. "
320                                          + "Using defaults for project version metadata..", file );
321 
322                         ProjectVersionMetadata metadata = new ProjectVersionMetadata();
323                         metadata.setId( readMetadataRequest.getProjectVersion() );
324 
325                         MavenProjectFacet facet = new MavenProjectFacet();
326                         facet.setGroupId( readMetadataRequest.getNamespace() );
327                         facet.setArtifactId( readMetadataRequest.getProjectId() );
328                         facet.setPackaging( "jar" );
329                         metadata.addFacet( facet );
330 
331                         String errMsg =
332                             "Error in resolving artifact's parent POM file. " + ( problem.getException() == null
333                                 ? problem.getMessage()
334                                 : problem.getException().getMessage() );
335                         RepositoryProblemFacet repoProblemFacet = new RepositoryProblemFacet();
336                         repoProblemFacet.setRepositoryId( readMetadataRequest.getRepositoryId() );
337                         repoProblemFacet.setId( readMetadataRequest.getRepositoryId() );
338                         repoProblemFacet.setMessage( errMsg );
339                         repoProblemFacet.setProblem( errMsg );
340                         repoProblemFacet.setProject( readMetadataRequest.getProjectId() );
341                         repoProblemFacet.setVersion( readMetadataRequest.getProjectVersion() );
342                         repoProblemFacet.setNamespace( readMetadataRequest.getNamespace() );
343 
344                         metadata.addFacet( repoProblemFacet );
345 
346                         return metadata;
347                     }
348                 }
349 
350                 throw new RepositoryStorageMetadataInvalidException( "invalid-pom", msg, e );
351             }
352 
353             // Check if the POM is in the correct location
354             boolean correctGroupId = readMetadataRequest.getNamespace().equals( model.getGroupId() );
355             boolean correctArtifactId = readMetadataRequest.getProjectId().equals( model.getArtifactId() );
356             boolean correctVersion = readMetadataRequest.getProjectVersion().equals( model.getVersion() );
357             if ( !correctGroupId || !correctArtifactId || !correctVersion )
358             {
359                 StringBuilder message = new StringBuilder( "Incorrect POM coordinates in '" + file + "':" );
360                 if ( !correctGroupId )
361                 {
362                     message.append( "\nIncorrect group ID: " ).append( model.getGroupId() );
363                 }
364                 if ( !correctArtifactId )
365                 {
366                     message.append( "\nIncorrect artifact ID: " ).append( model.getArtifactId() );
367                 }
368                 if ( !correctVersion )
369                 {
370                     message.append( "\nIncorrect version: " ).append( model.getVersion() );
371                 }
372 
373                 throw new RepositoryStorageMetadataInvalidException( "mislocated-pom", message.toString() );
374             }
375 
376             ProjectVersionMetadata metadata = new ProjectVersionMetadata();
377             metadata.setCiManagement( convertCiManagement( model.getCiManagement() ) );
378             metadata.setDescription( model.getDescription() );
379             metadata.setId( readMetadataRequest.getProjectVersion() );
380             metadata.setIssueManagement( convertIssueManagement( model.getIssueManagement() ) );
381             metadata.setLicenses( convertLicenses( model.getLicenses() ) );
382             metadata.setMailingLists( convertMailingLists( model.getMailingLists() ) );
383             metadata.setDependencies( convertDependencies( model.getDependencies() ) );
384             metadata.setName( model.getName() );
385             metadata.setOrganization( convertOrganization( model.getOrganization() ) );
386             metadata.setScm( convertScm( model.getScm() ) );
387             metadata.setUrl( model.getUrl() );
388             metadata.setProperties( model.getProperties() );
389 
390             MavenProjectFacet facet = new MavenProjectFacet();
391             facet.setGroupId( model.getGroupId() != null ? model.getGroupId() : model.getParent().getGroupId() );
392             facet.setArtifactId( model.getArtifactId() );
393             facet.setPackaging( model.getPackaging() );
394             if ( model.getParent() != null )
395             {
396                 MavenProjectParent parent = new MavenProjectParent();
397                 parent.setGroupId( model.getParent().getGroupId() );
398                 parent.setArtifactId( model.getParent().getArtifactId() );
399                 parent.setVersion( model.getParent().getVersion() );
400                 facet.setParent( parent );
401             }
402             metadata.addFacet( facet );
403 
404             return metadata;
405         }
406         catch ( RepositoryAdminException e )
407         {
408             throw new RepositoryStorageRuntimeException( "repo-admin", e.getMessage(), e );
409         }
410     }

526     private static Collection<String> getSortedFiles( Path dir, Filter<String> filter )
527     {
528 
529         try(Stream<Path> stream = Files.list(dir)) {
530             final Predicate<Path> dFilter = new DirectoryFilter( filter );
531             return stream.filter(Files::isDirectory)
532                     .filter(dFilter)
533                     .map(path -> path.getFileName().toString())
534                     .sorted().collect(Collectors.toList());
535 
536         } catch (IOException e) {
537             LOGGER.error("Could not read directory list {}: {}", dir, e.getMessage(),e);
538             return Collections.emptyList();
539         }
540     }

550     @Override
551     public Collection<String> listNamespaces( String repoId, String namespace, Filter<String> filter )
552         throws RepositoryStorageRuntimeException
553     {
554         Path dir = pathTranslator.toFile( getRepositoryBasedir( repoId ), namespace );
555         if (!(Files.exists(dir) && Files.isDirectory(dir))) {
556             return Collections.emptyList();
557         }
558         // scan all the directories which are potential namespaces. Any directories known to be projects are excluded
559         Predicate<Path> dFilter = new DirectoryFilter(filter);
560         try(Stream<Path> stream = Files.list(dir)) {
561             return stream.filter(dFilter).filter(path -> !isProject(path, filter)).map(path -> path.getFileName().toString())
562                     .sorted().collect(Collectors.toList());
563         } catch (IOException e) {
564             LOGGER.error("Could not read directory {}: {}", dir, e.getMessage(), e);
565             return Collections.emptyList();
566         }
567     }

569     @Override
570     public Collection<String> listProjects( String repoId, String namespace, Filter<String> filter )
571         throws RepositoryStorageRuntimeException
572     {
573         Path dir = pathTranslator.toFile( getRepositoryBasedir( repoId ), namespace );
574         if (!(Files.exists(dir) && Files.isDirectory(dir))) {
575             return Collections.emptyList();
576         }
577         // scan all directories in the namespace, and only include those that are known to be projects
578         final Predicate<Path> dFilter = new DirectoryFilter(filter);
579         try(Stream<Path> stream = Files.list(dir)) {
580             return stream.filter(dFilter).filter(path -> isProject(path, filter)).map(path -> path.getFileName().toString())
581                     .sorted().collect(Collectors.toList());
582         } catch (IOException e) {
583             LOGGER.error("Could not read directory {}: {}", dir, e.getMessage(), e);
584             return Collections.emptyList();
585         }

589     @Override
590     public Collection<String> listProjectVersions( String repoId, String namespace, String projectId,
591                                                    Filter<String> filter )
592         throws RepositoryStorageRuntimeException
593     {
594         Path dir = pathTranslator.toFile( getRepositoryBasedir( repoId ), namespace, projectId );
595         if (!(Files.exists(dir) && Files.isDirectory(dir))) {
596             return Collections.emptyList();
597         }
598 
599         // all directories in a project directory can be considered a version
600         return getSortedFiles( dir, filter );

603     @Override
604     public Collection<ArtifactMetadata> readArtifactsMetadata( ReadMetadataRequest readMetadataRequest )
605         throws RepositoryStorageRuntimeException
606     {
607         Path dir = pathTranslator.toFile( getRepositoryBasedir( readMetadataRequest.getRepositoryId() ),
608                                           readMetadataRequest.getNamespace(), readMetadataRequest.getProjectId(),
609                                           readMetadataRequest.getProjectVersion() );
610         if (!(Files.exists(dir) && Files.isDirectory(dir))) {
611             return Collections.emptyList();
612         }
613 
614         // all files that are not metadata and not a checksum / signature are considered artifacts
615         final Predicate<Path> dFilter = new ArtifactDirectoryFilter(readMetadataRequest.getFilter());
616         try(Stream<Path> stream = Files.list(dir)) {
617             // Returns a map TRUE -> (success values), FALSE -> (Exceptions)
618             Map<Boolean, List<Try<ArtifactMetadata>>> result = stream.filter(dFilter).map(path -> {
619                         try {
620                             return Try.success(getArtifactFromFile(readMetadataRequest.getRepositoryId(), readMetadataRequest.getNamespace(),
621                                     readMetadataRequest.getProjectId(), readMetadataRequest.getProjectVersion(),
622                                     path));
623                         } catch (Exception e) {
624                             LOGGER.debug("Could not create metadata for {}:  {}", path, e.getMessage(), e);
625                             return Try.<ArtifactMetadata>failure(e);
626                         }
627                     }
628             ).collect(Collectors.groupingBy(Try::isSuccess));
629             if (result.containsKey(Boolean.FALSE) && result.get(Boolean.FALSE).size()>0 && (!result.containsKey(Boolean.TRUE) || result.get(Boolean.TRUE).size()==0)) {
630                 LOGGER.error("Could not get artifact metadata. Directory: {}. Number of errors {}.", dir, result.get(Boolean.FALSE).size());
631                 Try<ArtifactMetadata> failure = result.get(Boolean.FALSE).get(0);
632                 LOGGER.error("Sample exception {}", failure.getError().getMessage(), failure.getError());
633                 throw new RepositoryStorageRuntimeException(readMetadataRequest.getRepositoryId(), "Could not retrieve metadata of the files");
634             } else {
635                 if (!result.containsKey(Boolean.TRUE) || result.get(Boolean.TRUE) == null) {
636                     return Collections.emptyList();
637                 }
638                 return result.get(Boolean.TRUE).stream().map(tr -> tr.get()).collect(Collectors.toList());
639             }
640         } catch (IOException e) {
641             LOGGER.error("Could not read directory {}: {}", dir, e.getMessage(), e);
642         }
643         return Collections.emptyList();
644 
645     }

672     @Override
673     public void applyServerSideRelocation( ManagedRepositoryContent managedRepository, ArtifactReference artifact )
674         throws ProxyDownloadException
675     {
676         if ( "pom".equals( artifact.getType() ) )
677         {
678             return;
679         }
680 
681         // Build the artifact POM reference
682         ArtifactReference pomReference = new ArtifactReference();
683         pomReference.setGroupId( artifact.getGroupId() );
684         pomReference.setArtifactId( artifact.getArtifactId() );
685         pomReference.setVersion( artifact.getVersion() );
686         pomReference.setType( "pom" );
687 
688         RepositoryProxyConnectors connectors =
689             applicationContext.getBean( "repositoryProxyConnectors#default", RepositoryProxyConnectors.class );
690 
691         // Get the artifact POM from proxied repositories if needed
692         connectors.fetchFromProxies( managedRepository, pomReference );
693 
694         // Open and read the POM from the managed repo
695         Path pom = managedRepository.toFile( pomReference );
696 
697         if ( !Files.exists(pom) )
698         {
699             return;
700         }
701 
702         try
703         {
704             // MavenXpp3Reader leaves the file open, so we need to close it ourselves.
705 
706             Model model = null;
707             try (Reader reader = Files.newBufferedReader( pom, Charset.defaultCharset() ))
708             {
709                 model = MAVEN_XPP_3_READER.read( reader );
710             }
711 
712             DistributionManagement dist = model.getDistributionManagement();
713             if ( dist != null )
714             {
715                 Relocation relocation = dist.getRelocation();
716                 if ( relocation != null )
717                 {
718                     // artifact is relocated : update the repositoryPath
719                     if ( relocation.getGroupId() != null )
720                     {
721                         artifact.setGroupId( relocation.getGroupId() );
722                     }
723                     if ( relocation.getArtifactId() != null )
724                     {
725                         artifact.setArtifactId( relocation.getArtifactId() );
726                     }
727                     if ( relocation.getVersion() != null )
728                     {
729                         artifact.setVersion( relocation.getVersion() );
730                     }
731                 }
732             }
733         }
734         catch ( IOException e )
735         {
736             // Unable to read POM : ignore.
737         }
738         catch ( XmlPullParserException e )
739         {
740             // Invalid POM : ignore
741         }
742     }

745     @Override
746     public String getFilePath( String requestPath, org.apache.archiva.repository.ManagedRepository managedRepository )
747     {
748         // managedRepository can be null
749         // extract artifact reference from url
750         // groupId:artifactId:version:packaging:classifier
751         //org/apache/archiva/archiva-checksum/1.4-M4-SNAPSHOT/archiva-checksum-1.4-M4-SNAPSHOT.jar
752         String logicalResource = null;
753         String requestPathInfo = StringUtils.defaultString( requestPath );
754 
755         //remove prefix ie /repository/blah becomes /blah
756         requestPathInfo = removePrefix( requestPathInfo );
757 
758         // Remove prefixing slash as the repository id doesn't contain it;
759         if ( requestPathInfo.startsWith( "/" ) )
760         {
761             requestPathInfo = requestPathInfo.substring( 1 );
762         }
763 
764         int slash = requestPathInfo.indexOf( '/' );
765         if ( slash > 0 )
766         {
767             logicalResource = requestPathInfo.substring( slash );
768 
769             if ( logicalResource.endsWith( "/.." ) )
770             {
771                 logicalResource += "/";
772             }
773 
774             if ( logicalResource != null && logicalResource.startsWith( "//" ) )
775             {
776                 logicalResource = logicalResource.substring( 1 );
777             }
778 
779             if ( logicalResource == null )
780             {
781                 logicalResource = "/";
782             }
783         }
784         else
785         {
786             logicalResource = "/";
787         }
788         return logicalResource;
789 
790     }

792     @Override
793     public String getFilePathWithVersion( final String requestPath, ManagedRepositoryContent managedRepositoryContent )
794         throws XMLException, RelocationException
795     {
796 
797         if ( StringUtils.endsWith( requestPath, METADATA_FILENAME ) )
798         {
799             return getFilePath( requestPath, managedRepositoryContent.getRepository() );
800         }
801 
802         String filePath = getFilePath( requestPath, managedRepositoryContent.getRepository() );
803 
804         ArtifactReference artifactReference = null;
805         try
806         {
807             artifactReference = pathParser.toArtifactReference( filePath );
808         }
809         catch ( LayoutException e )
810         {
811             return filePath;
812         }
813 
814         if ( StringUtils.endsWith( artifactReference.getVersion(), VersionUtil.SNAPSHOT ) )
815         {
816             // read maven metadata to get last timestamp
817             Path metadataDir = Paths.get( managedRepositoryContent.getRepoRoot(), filePath ).getParent();
818             if ( !Files.exists(metadataDir) )
819             {
820                 return filePath;
821             }
822             Path metadataFile = metadataDir.resolve( METADATA_FILENAME );
823             if ( !Files.exists(metadataFile) )
824             {
825                 return filePath;
826             }
827             ArchivaRepositoryMetadata archivaRepositoryMetadata = MavenMetadataReader.read( metadataFile );
828             int buildNumber = archivaRepositoryMetadata.getSnapshotVersion().getBuildNumber();
829             String timestamp = archivaRepositoryMetadata.getSnapshotVersion().getTimestamp();
830 
831             // MRM-1846
832             if ( buildNumber < 1 && timestamp == null )
833             {
834                 return filePath;
835             }
836 
837             // org/apache/archiva/archiva-checksum/1.4-M4-SNAPSHOT/archiva-checksum-1.4-M4-SNAPSHOT.jar
838             // ->  archiva-checksum-1.4-M4-20130425.081822-1.jar
839 
840             filePath = StringUtils.replace( filePath, //
841                                             artifactReference.getArtifactId() //
842                                                 + "-" + artifactReference.getVersion(), //
843                                             artifactReference.getArtifactId() //
844                                                 + "-" + StringUtils.remove( artifactReference.getVersion(),
845                                                                             "-" + VersionUtil.SNAPSHOT ) //
846                                                 + "-" + timestamp //
847                                                 + "-" + buildNumber );
848 
849             throw new RelocationException( "/repository/" + managedRepositoryContent.getRepository().getId() +
850                                                ( StringUtils.startsWith( filePath, "/" ) ? "" : "/" ) + filePath,
851                                            RelocationException.RelocationType.TEMPORARY );
852 
853         }
854 
855         return filePath;
856     }

863      * FIXME remove
864      *
865      * @param href
866      * @return
867      */
868     private static String removePrefix( final String href )
869     {
870         String[] parts = StringUtils.split( href, '/' );
871         parts = (String[]) ArrayUtils.subarray( parts, 1, parts.length );
872         if ( parts == null || parts.length == 0 )
873         {
874             return "/";
875         }
876 
877         String joinedString = StringUtils.join( parts, '/' );
878         if ( href.endsWith( "/" ) )
879         {
880             joinedString = joinedString + "/";
881         }
882 
883         return joinedString;
884     }

886     private static void populateArtifactMetadataFromFile( ArtifactMetadata metadata, Path file ) throws IOException {
887         metadata.setWhenGathered( new Date() );
888         metadata.setFileLastModified( Files.getLastModifiedTime(file).toMillis() );
889         ChecksummedFile checksummedFile = new ChecksummedFile( file );
890         try
891         {
892             metadata.setMd5( checksummedFile.calculateChecksum( ChecksumAlgorithm.MD5 ) );
893         }
894         catch ( IOException e )
895         {
896             LOGGER.error( "Unable to checksum file {}: {},MD5", file, e.getMessage() );
897         }
898         try
899         {
900             metadata.setSha1( checksummedFile.calculateChecksum( ChecksumAlgorithm.SHA1 ) );
901         }
902         catch ( IOException e )
903         {
904             LOGGER.error( "Unable to checksum file {}: {},SHA1", file, e.getMessage() );
905         }
906         metadata.setSize( Files.size(file) );
907     }

909     private boolean isProject( Path dir, Filter<String> filter )
910     {
911         // scan directories for a valid project version subdirectory, meaning this must be a project directory
912         final Predicate<Path> dFilter = new DirectoryFilter(filter);
913         try(Stream<Path> stream = Files.list(dir)) {
914             boolean projFound = stream.filter(dFilter)
915                     .anyMatch(path -> isProjectVersion(path));
916             if (projFound) {
917                 return true;
918             }
919         } catch (IOException e) {
920             LOGGER.error("Could not read directory list {}: {}", dir, e.getMessage(), e);
921         }
922 
923         // if a metadata file is present, check if this is the "artifactId" directory, marking it as a project
924         ArchivaRepositoryMetadata metadata = readMetadata( dir );
925         if ( metadata != null && dir.getFileName().toString().equals( metadata.getArtifactId() ) )
926         {
927             return true;
928         }
929 
930         return false;
931     }

933     private boolean isProjectVersion( Path dir )
934     {
935         final String artifactId = dir.getParent().getFileName().toString();
936         final String projectVersion = dir.getFileName().toString();
937 
938         // check if there is a POM artifact file to ensure it is a version directory
939 
940         Predicate<Path> filter;
941         if ( VersionUtil.isSnapshot( projectVersion ) )
942         {
943             filter = new PomFilenameFilter(artifactId, projectVersion);
944         }
945         else
946         {
947             final String pomFile = artifactId + "-" + projectVersion + ".pom";
948             filter = new PomFileFilter(pomFile);
949         }
950         try(Stream<Path> stream = Files.list(dir)) {
951             if (stream.filter(Files::isRegularFile).anyMatch(filter)){
952                 return true;
953             }
954         } catch (IOException e) {
955             LOGGER.error("Could not list directory {}: {}", dir, e.getMessage(), e);
956         }
957 
958         // if a metadata file is present, check if this is the "version" directory, marking it as a project version
959         ArchivaRepositoryMetadata metadata = readMetadata( dir );
960         if ( metadata != null && projectVersion.equals( metadata.getVersion() ) )
961         {
962             return true;
963         }
964 
965         return false;
966     }

1016     private static class ArtifactDirectoryFilter
1017         implements Predicate<Path>
1018     {
1019         private final Filter<String> filter;
1020 
1021         private ArtifactDirectoryFilter( Filter<String> filter )
1022         {
1023             this.filter = filter;
1024         }
1025 
1026         @Override
1027         public boolean test( Path dir )
1028         {
1029             final String name = dir.getFileName().toString();
1030             // TODO compare to logic in maven-repository-layer
1031             if ( !filter.accept( name ) )
1032             {
1033                 return false;
1034             }
1035             else if ( name.startsWith( "." ) )
1036             {
1037                 return false;
1038             }
1039             else if ( name.endsWith( ".md5" ) || name.endsWith( ".sha1" ) || name.endsWith( ".asc" ) )
1040             {
1041                 return false;
1042             }
1043             else if ( Arrays.binarySearch(IGNORED_FILES, name)>=0 )
1044             {
1045                 return false;
1046             }
1047             else if ( Files.isDirectory(dir) )
1048             {
1049                 return false;
1050             }
1051             // some files from remote repositories can have name like maven-metadata-archiva-vm-all-public.xml
1052             else if ( StringUtils.startsWith( name, METADATA_FILENAME_START ) && StringUtils.endsWith( name, ".xml" ) )
1053             {
1054                 return false;
1055             }
1056 
1057             return true;
1058 
1059         }
