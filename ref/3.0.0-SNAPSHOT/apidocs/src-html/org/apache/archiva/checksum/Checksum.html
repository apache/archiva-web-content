022import org.apache.commons.io.IOUtils;
023import org.apache.commons.io.output.NullOutputStream;
024
025import java.io.FileNotFoundException;
026import java.io.IOException;
027import java.io.InputStream;
028import java.nio.ByteBuffer;
029import java.nio.MappedByteBuffer;
030import java.nio.channels.FileChannel;
031import java.nio.file.Path;
032import java.nio.file.StandardOpenOption;
033import java.security.DigestInputStream;
034import java.security.MessageDigest;
035import java.security.NoSuchAlgorithmException;
036import java.util.List;
037
038/**
039 * Checksum - simple checksum hashing routines.
040 */
041public class Checksum
042{
043    private static final int BUFFER_SIZE = 32768;
044    private byte[] result = new byte[0];
045
046    public static void update( List<Checksum> checksums, Path file )
047        throws ChecksumValidationException
048    {
049        long fileSize;
050        try (FileChannel channel = FileChannel.open(file, StandardOpenOption.READ )) {
051            fileSize = channel.size();
052            long pos = 0;
053            while (pos<fileSize)
054            {
055                long bufferSize = Math.min(BUFFER_SIZE, fileSize-pos);
056                MappedByteBuffer buffer = channel.map( FileChannel.MapMode.READ_ONLY, pos, bufferSize);
057                for ( Checksum checksum : checksums )
058                {
059                    checksum.update( buffer );
060                    buffer.rewind();
061                }
062                fileSize = channel.size();
063                pos += BUFFER_SIZE;
064            }
065            for (Checksum checksum : checksums) {
066                checksum.finish();
067            }
068        } catch(FileNotFoundException e)
069        {
070            throw new ChecksumValidationException( ChecksumValidationException.ValidationError.FILE_NOT_FOUND, "File that should be parsed, not found: "+e.getMessage(), e );
071        } catch(IOException e) {
072            throw new ChecksumValidationException( ChecksumValidationException.ValidationError.READ_ERROR, "Parsing of file failed: "+e.getMessage(), e );
073        }
074    }
075
076    private final MessageDigest md;
077
078    private ChecksumAlgorithm checksumAlgorithm;
079
080    public Checksum( ChecksumAlgorithm checksumAlgorithm )
081    {
082        this.checksumAlgorithm = checksumAlgorithm;
083        try
084        {
085            md = MessageDigest.getInstance( checksumAlgorithm.getAlgorithm() );
086        }
087        catch ( NoSuchAlgorithmException e )
088        {
089            // Not really possible, but here none-the-less
090            throw new IllegalStateException(
091                "Unable to initialize MessageDigest algorithm " + checksumAlgorithm.getAlgorithm() + " : "
092                    + e.getMessage(), e );
093        }
094    }
095
096    public String getChecksum()
097    {
098        if (this.result.length==0) {
099            finish();
100        }
101        return Hex.encode( this.result );
102    }
103
104    public byte[] getChecksumBytes() {
105        if (this.result.length==0) {
106            finish();
107        }
108        return this.result;
109    }
110
111    public ChecksumAlgorithm getAlgorithm()
112    {
113        return this.checksumAlgorithm;
114    }
115
116    public void reset()
117    {
118        md.reset();
119        this.result = new byte[0];
120    }
121
122    public Checksum update( byte[] buffer, int offset, int size )
123    {
124        if (this.result.length!=0) {
125            reset();
126        }
127        md.update( buffer, 0, size );
128        return this;
129    }
130
131    public Checksum update( ByteBuffer buffer)
132    {
133        if (this.result.length!=0) {
134            reset();
135        }
136        md.update( buffer );
137        return this;
138    }
139
140    public Checksum finish() {
141        this.result = md.digest();
142        return this;
143    }
144
145    public void update( Path file )
146        throws IOException
147    {
148        long fileSize;
149        try (FileChannel channel = FileChannel.open(file, StandardOpenOption.READ )) {
150            fileSize = channel.size();
151            long pos = 0;
152            while (pos<fileSize)
153            {
154                long bufferSize = Math.min(BUFFER_SIZE, fileSize-pos);
155                MappedByteBuffer buffer = channel.map( FileChannel.MapMode.READ_ONLY, pos, bufferSize);
156                update( buffer );
157                buffer.rewind();
158                fileSize = channel.size();
159                pos += BUFFER_SIZE;
160            }
161            finish();
162        }
163    }
164
165    public boolean compare(byte[] cmp) {
166        if (this.result == null || this.result.length==0) {
167            finish();
168        }
169        return md.isEqual( this.result, cmp );
170    }
171
172    public boolean compare(String hexString) {
173        if (this.result == null || this.result.length==0) {
174            finish();
175        }
176        return md.isEqual(this.result, Hex.decode( hexString ));
177    }
178}
