097    private URI getURIFromString(String uriStr) throws RepositoryException {
098        URI uri;
099        try {
100            if (StringUtils.isEmpty(uriStr)) {
101                return new URI("");
102            }
103            if (uriStr.startsWith("/")) {
104                // only absolute paths are prepended with file scheme
105                uri = new URI("file://" + uriStr);
106            } else {
107                uri = new URI(uriStr);
108            }
109            if (uri.getScheme() != null && !"file".equals(uri.getScheme())) {
110                log.error("Bad URI scheme found: {}, URI={}", uri.getScheme(), uri);
111                throw new RepositoryException("The uri " + uriStr + " is not valid. Only file:// URI is allowed for maven.");
112            }
113        } catch (URISyntaxException e) {
114            String newCfg = "file://" + uriStr;
115            try {
116                uri = new URI(newCfg);
117            } catch (URISyntaxException e1) {
118                log.error("Could not create URI from {} -> ", uriStr, newCfg);
119                throw new RepositoryException("The config entry " + uriStr + " cannot be converted to URI.");
120            }
121        }
122        log.debug("Setting location uri: {}", uri);
123        return uri;
124    }

133    @Override
134    public void updateManagedInstance(EditableManagedRepository repo, ManagedRepositoryConfiguration cfg) throws RepositoryException {
135        try {
136            repo.setLocation(getURIFromString(cfg.getLocation()));
137        } catch (UnsupportedURIException e) {
138            throw new RepositoryException("The location entry is not a valid uri: " + cfg.getLocation());
139        }
140        setBaseConfig(repo, cfg);
141        Path repoDir = repo.getLocalPath();
142        if (!Files.exists(repoDir)) {
143            log.debug("Creating repo directory {}", repoDir);
144            try {
145                Files.createDirectories(repoDir);
146            } catch (IOException e) {
147                log.error("Could not create directory {} for repository {}", repo.getLocalPath(), repo.getId(), e);
148                throw new RepositoryException("Could not create directory for repository " + repo.getLocalPath());
149            }
150        }
151        repo.setSchedulingDefinition(cfg.getRefreshCronExpression());
152        repo.setBlocksRedeployment(cfg.isBlockRedeployments());
153        repo.setScanned(cfg.isScanned());
154        if (cfg.isReleases()) {
155            repo.addActiveReleaseScheme(ReleaseScheme.RELEASE);
156        }
157        if (cfg.isSnapshots()) {
158            repo.addActiveReleaseScheme(ReleaseScheme.SNAPSHOT);
159        }
160
161        StagingRepositoryFeature stagingRepositoryFeature = repo.getFeature(StagingRepositoryFeature.class).get();
162        stagingRepositoryFeature.setStageRepoNeeded(cfg.isStageRepoNeeded());
163
164        IndexCreationFeature indexCreationFeature = repo.getFeature(IndexCreationFeature.class).get();
165        indexCreationFeature.setSkipPackedIndexCreation(cfg.isSkipPackedIndexCreation());
166        indexCreationFeature.setIndexPath(getURIFromString(cfg.getIndexDir()));
167        indexCreationFeature.setPackedIndexPath(getURIFromString(cfg.getPackedIndexDir()));
168        /* -> Should be created by MavenIndexProvider
169
170        Path indexPath;
171        if (indexCreationFeature.getIndexPath().getScheme() == null) {
172            indexPath = Paths.get(indexCreationFeature.getIndexPath().getPath());
173        } else {
174            indexPath = Paths.get(indexCreationFeature.getIndexPath());
175        }
176        Path absoluteIndexPath;
177        if (indexPath.isAbsolute()) {
178            absoluteIndexPath = indexPath;
179        } else {
180            absoluteIndexPath = PathUtil.getPathFromUri(repo.getLocation()).resolve(indexCreationFeature.getIndexPath().getPath());
181        }
182        try {
183            Files.createDirectories(absoluteIndexPath);
184        } catch (IOException e) {
185            log.error("Could not create index directory {}", absoluteIndexPath);
186            throw new RepositoryException("Could not create index directory " + absoluteIndexPath);
187        }*/
188
189        ArtifactCleanupFeature artifactCleanupFeature = repo.getFeature(ArtifactCleanupFeature.class).get();
190
191        artifactCleanupFeature.setDeleteReleasedSnapshots(cfg.isDeleteReleasedSnapshots());
192        artifactCleanupFeature.setRetentionCount(cfg.getRetentionCount());
193        artifactCleanupFeature.setRetentionPeriod(Period.ofDays(cfg.getRetentionPeriod()));
194    }

211    private String convertUriToPath(URI uri) {
212        if (uri.getScheme() == null) {
213            return uri.getPath();
214        } else if ("file".equals(uri.getScheme())) {
215            return Paths.get(uri).toString();
216        } else {
217            return uri.toString();
218        }
219    }

221    @Override
222    public void updateRemoteInstance(EditableRemoteRepository repo, RemoteRepositoryConfiguration cfg) throws RepositoryException {
223        setBaseConfig(repo, cfg);
224        repo.setCheckPath(cfg.getCheckPath());
225        repo.setSchedulingDefinition(cfg.getRefreshCronExpression());
226        try {
227            repo.setLocation(new URI(cfg.getUrl()));
228        } catch (UnsupportedURIException | URISyntaxException e) {
229            log.error("Could not set remote url " + cfg.getUrl());
230            throw new RepositoryException("The url config is not a valid uri: " + cfg.getUrl());
231        }
232        repo.setTimeout(Duration.ofSeconds(cfg.getTimeout()));
233        RemoteIndexFeature remoteIndexFeature = repo.getFeature(RemoteIndexFeature.class).get();
234        remoteIndexFeature.setDownloadRemoteIndex(cfg.isDownloadRemoteIndex());
235        remoteIndexFeature.setDownloadRemoteIndexOnStartup(cfg.isDownloadRemoteIndexOnStartup());
236        remoteIndexFeature.setDownloadTimeout(Duration.ofSeconds(cfg.getRemoteDownloadTimeout()));
237        remoteIndexFeature.setProxyId(cfg.getRemoteDownloadNetworkProxyId());
238        if (cfg.isDownloadRemoteIndex()) {
239            try {
240                remoteIndexFeature.setIndexUri(new URI(cfg.getRemoteIndexUrl()));
241            } catch (URISyntaxException e) {
242                log.error("Could not set remote index url " + cfg.getRemoteIndexUrl());
243                remoteIndexFeature.setDownloadRemoteIndex(false);
244                remoteIndexFeature.setDownloadRemoteIndexOnStartup(false);
245            }
246        }
247        for ( Object key : cfg.getExtraHeaders().keySet() ) {
248            repo.addExtraHeader( key.toString(), cfg.getExtraHeaders().get(key).toString() );
249        }
250        for ( Object key : cfg.getExtraParameters().keySet() ) {
251            repo.addExtraParameter( key.toString(), cfg.getExtraParameters().get(key).toString() );
252        }
253        PasswordCredentials credentials = new PasswordCredentials("", new char[0]);
254        if (cfg.getPassword() != null && cfg.getUsername() != null) {
255            credentials.setPassword(cfg.getPassword().toCharArray());
256            credentials.setUsername(cfg.getUsername());
257            repo.setCredentials(credentials);
258        } else {
259            credentials.setPassword(new char[0]);
260        }
261        IndexCreationFeature indexCreationFeature = repo.getFeature(IndexCreationFeature.class).get();
262        if (cfg.getIndexDir() != null) {
263            indexCreationFeature.setIndexPath(getURIFromString(cfg.getIndexDir()));
264        }
265        if (cfg.getPackedIndexDir() != null) {
266            indexCreationFeature.setPackedIndexPath(getURIFromString(cfg.getPackedIndexDir()));
267        }
268        log.debug("Updated remote instance {}", repo);
269    }

271    @Override
272    public RemoteRepositoryConfiguration getRemoteConfiguration(RemoteRepository remoteRepository) throws RepositoryException {
273        if (!(remoteRepository instanceof MavenRemoteRepository)) {
274            log.error("Wrong remote repository type " + remoteRepository.getClass().getName());
275            throw new RepositoryException("The given repository type cannot be handled by the maven provider: " + remoteRepository.getClass().getName());
276        }
277        RemoteRepositoryConfiguration cfg = new RemoteRepositoryConfiguration();
278        cfg.setType(remoteRepository.getType().toString());
279        cfg.setId(remoteRepository.getId());
280        cfg.setName(remoteRepository.getName());
281        cfg.setDescription(remoteRepository.getDescription());
282        cfg.setUrl(remoteRepository.getLocation().toString());
283        cfg.setTimeout((int) remoteRepository.getTimeout().toMillis() / 1000);
284        cfg.setCheckPath(remoteRepository.getCheckPath());
285        RepositoryCredentials creds = remoteRepository.getLoginCredentials();
286        if (creds != null) {
287            if (creds instanceof PasswordCredentials) {
288                PasswordCredentials pCreds = (PasswordCredentials) creds;
289                cfg.setPassword(new String(pCreds.getPassword()));
290                cfg.setUsername(pCreds.getUsername());
291            }
292        }
293        cfg.setLayout(remoteRepository.getLayout());
294        cfg.setExtraParameters(remoteRepository.getExtraParameters());
295        cfg.setExtraHeaders(remoteRepository.getExtraHeaders());
296        cfg.setRefreshCronExpression(remoteRepository.getSchedulingDefinition());
297
298        IndexCreationFeature indexCreationFeature = remoteRepository.getFeature(IndexCreationFeature.class).get();
299        cfg.setIndexDir(convertUriToPath(indexCreationFeature.getIndexPath()));
300        cfg.setPackedIndexDir(convertUriToPath(indexCreationFeature.getPackedIndexPath()));
301
302        RemoteIndexFeature remoteIndexFeature = remoteRepository.getFeature(RemoteIndexFeature.class).get();
303        cfg.setRemoteIndexUrl(remoteIndexFeature.getIndexUri().toString());
304        cfg.setRemoteDownloadTimeout((int) remoteIndexFeature.getDownloadTimeout().get(ChronoUnit.SECONDS));
305        cfg.setDownloadRemoteIndexOnStartup(remoteIndexFeature.isDownloadRemoteIndexOnStartup());
306        cfg.setDownloadRemoteIndex(remoteIndexFeature.isDownloadRemoteIndex());
307        cfg.setRemoteDownloadNetworkProxyId(remoteIndexFeature.getProxyId());
308
309
310        return cfg;
311
312    }

314    @Override
315    public ManagedRepositoryConfiguration getManagedConfiguration(ManagedRepository managedRepository) throws RepositoryException {
316        if (!(managedRepository instanceof MavenManagedRepository || managedRepository instanceof BasicManagedRepository)) {
317            log.error("Wrong remote repository type " + managedRepository.getClass().getName());
318            throw new RepositoryException("The given repository type cannot be handled by the maven provider: " + managedRepository.getClass().getName());
319        }
320        ManagedRepositoryConfiguration cfg = new ManagedRepositoryConfiguration();
321        cfg.setType(managedRepository.getType().toString());
322        cfg.setId(managedRepository.getId());
323        cfg.setName(managedRepository.getName());
324        cfg.setDescription(managedRepository.getDescription());
325        cfg.setLocation(convertUriToPath(managedRepository.getLocation()));
326        cfg.setLayout(managedRepository.getLayout());
327        cfg.setRefreshCronExpression(managedRepository.getSchedulingDefinition());
328        cfg.setScanned(managedRepository.isScanned());
329        cfg.setBlockRedeployments(managedRepository.blocksRedeployments());
330        StagingRepositoryFeature stagingRepositoryFeature = managedRepository.getFeature(StagingRepositoryFeature.class).get();
331        cfg.setStageRepoNeeded(stagingRepositoryFeature.isStageRepoNeeded());
332        IndexCreationFeature indexCreationFeature = managedRepository.getFeature(IndexCreationFeature.class).get();
333        cfg.setIndexDir(convertUriToPath(indexCreationFeature.getIndexPath()));
334        cfg.setPackedIndexDir(convertUriToPath(indexCreationFeature.getPackedIndexPath()));
335        cfg.setSkipPackedIndexCreation(indexCreationFeature.isSkipPackedIndexCreation());
336
337        ArtifactCleanupFeature artifactCleanupFeature = managedRepository.getFeature(ArtifactCleanupFeature.class).get();
338        cfg.setRetentionCount(artifactCleanupFeature.getRetentionCount());
339        cfg.setRetentionPeriod(artifactCleanupFeature.getRetentionPeriod().getDays());
340        cfg.setDeleteReleasedSnapshots(artifactCleanupFeature.isDeleteReleasedSnapshots());
341
342        if (managedRepository.getActiveReleaseSchemes().contains(ReleaseScheme.RELEASE)) {
343            cfg.setReleases(true);
344        } else {
345            cfg.setReleases(false);
346        }
347        if (managedRepository.getActiveReleaseSchemes().contains(ReleaseScheme.SNAPSHOT)) {
348            cfg.setSnapshots(true);
349        } else {
350            cfg.setSnapshots(false);
351        }
352        return cfg;
353
354    }

356    private ManagedRepositoryConfiguration getStageRepoConfig(ManagedRepositoryConfiguration repository) {
357        ManagedRepositoryConfiguration stagingRepository = new ManagedRepositoryConfiguration();
358        stagingRepository.setId(repository.getId() + StagingRepositoryFeature.STAGING_REPO_POSTFIX);
359        stagingRepository.setLayout(repository.getLayout());
360        stagingRepository.setName(repository.getName() + StagingRepositoryFeature.STAGING_REPO_POSTFIX);
361        stagingRepository.setBlockRedeployments(repository.isBlockRedeployments());
362        stagingRepository.setRetentionPeriod(repository.getRetentionPeriod());
363        stagingRepository.setDeleteReleasedSnapshots(repository.isDeleteReleasedSnapshots());
364        stagingRepository.setStageRepoNeeded(false);
365
366        String path = repository.getLocation();
367        int lastIndex = path.replace('\\', '/').lastIndexOf('/');
368        stagingRepository.setLocation(path.substring(0, lastIndex) + "/" + stagingRepository.getId());
369
370        if (StringUtils.isNotBlank(repository.getIndexDir())) {
371            Path indexDir = null;
372            try {
373                indexDir = Paths.get(new URI(repository.getIndexDir().startsWith("file://") ? repository.getIndexDir() : "file://" + repository.getIndexDir()));
374                if (indexDir.isAbsolute()) {
375                    Path newDir = indexDir.getParent().resolve(indexDir.getFileName() + StagingRepositoryFeature.STAGING_REPO_POSTFIX);
376                    log.debug("Changing index directory {} -> {}", indexDir, newDir);
377                    stagingRepository.setIndexDir(newDir.toString());
378                } else {
379                    log.debug("Keeping index directory {}", repository.getIndexDir());
380                    stagingRepository.setIndexDir(repository.getIndexDir());
381                }
382            } catch (URISyntaxException e) {
383                log.error("Could not parse index path as uri {}", repository.getIndexDir());
384                stagingRepository.setIndexDir("");
385            }
386            // in case of absolute dir do not use the same
387        }
388        if (StringUtils.isNotBlank(repository.getPackedIndexDir())) {
389            Path packedIndexDir = null;
390            try {
391                packedIndexDir = Paths.get(new URI(repository.getPackedIndexDir().startsWith("file://") ? repository.getPackedIndexDir() : "file://" + repository.getPackedIndexDir()));
392                if (packedIndexDir.isAbsolute()) {
393                    Path newDir = packedIndexDir.getParent().resolve(packedIndexDir.getFileName() + StagingRepositoryFeature.STAGING_REPO_POSTFIX);
394                    log.debug("Changing index directory {} -> {}", packedIndexDir, newDir);
395                    stagingRepository.setPackedIndexDir(newDir.toString());
396                } else {
397                    log.debug("Keeping index directory {}", repository.getPackedIndexDir());
398                    stagingRepository.setPackedIndexDir(repository.getPackedIndexDir());
399                }
400            } catch (URISyntaxException e) {
401                log.error("Could not parse index path as uri {}", repository.getPackedIndexDir());
402                stagingRepository.setPackedIndexDir("");
403            }
404            // in case of absolute dir do not use the same
405        }
406        stagingRepository.setRefreshCronExpression(repository.getRefreshCronExpression());
407        stagingRepository.setReleases(repository.isReleases());
408        stagingRepository.setRetentionCount(repository.getRetentionCount());
409        stagingRepository.setScanned(repository.isScanned());
410        stagingRepository.setSnapshots(repository.isSnapshots());
411        stagingRepository.setSkipPackedIndexCreation(repository.isSkipPackedIndexCreation());
412        // do not duplicate description
413        //stagingRepository.getDescription("")
414        return stagingRepository;
415    }

417    private void setBaseConfig(EditableRepository repo, AbstractRepositoryConfiguration cfg) throws RepositoryException {
418
419        URI baseUri = archivaConfiguration.getRepositoryBaseDir().toUri();
420        repo.setBaseUri(baseUri);
421
422        repo.setName(repo.getPrimaryLocale(), cfg.getName());
423        repo.setDescription(repo.getPrimaryLocale(), cfg.getDescription());
424        repo.setLayout(cfg.getLayout());
425    }
