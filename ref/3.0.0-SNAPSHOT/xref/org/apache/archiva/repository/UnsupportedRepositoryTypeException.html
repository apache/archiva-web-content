22  /**
23   * This is a fatal exception and should not happen. It is thrown if the implementation
24   * does not have certain classes to support this repository type.
25   */
26  public class UnsupportedRepositoryTypeException extends RuntimeException {
27  
28      private static final String MESSAGE = "The repository type is not supported: ";
29  
30      public UnsupportedRepositoryTypeException(RepositoryType type) {
31          super(MESSAGE+type.name());
32      }
33  
34      public UnsupportedRepositoryTypeException(RepositoryType type, Throwable cause) {
35          super(MESSAGE+type.name(), cause);
36      }
37  
38      public UnsupportedRepositoryTypeException(Throwable cause) {
39          super(cause);
40      }
41  
42      protected UnsupportedRepositoryTypeException(RepositoryType type, Throwable cause, boolean enableSuppression, boolean writableStackTrace) {
43          super(MESSAGE+type.name(), cause, enableSuppression, writableStackTrace);
44      }
45  }
